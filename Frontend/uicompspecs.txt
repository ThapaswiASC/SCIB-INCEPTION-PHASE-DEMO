      <div 
        className="progress-fill"
        style={{width: `${calculateProgress()}%`}}

    </div>
    <span className="spent-amount">
      {formatCurrency(weeklySpent, currency)} spent this week
    </span>
  </div>
</div>

### 4.6 FundTransferComponent

Component Type: Feature

Mapped From A1: Handle fund transfers to youth account

Purpose: Manages fund transfers from parent accounts to youth account with validation and confirmation

Parent: YouthAccountApp (via routing)

Children:
- FundTransferModal
- TransferFormComponent

Dependencies:
- Custom hook useFundTransfer
- API service fundTransferApi
- Form validation utilities

Library Components Used:
- ButtonComponent → Transfer and cancel buttons

TypeScript Specification

Inputs:
- sourceAccounts: Account[] (Available source accounts)
- youthAccountId: string (Target account ID)
- maxTransferAmount?: number (Maximum transfer limit)

Outputs:
- onTransferSubmit: (transferData: TransferRequest) => void (Transfer submission)
- onTransferCancel: () => void (Transfer cancellation)
- onTransferComplete: (result: TransferResult) => void (Transfer completion)

State:
- transferAmount: number (Transfer amount)
- selectedSource: string (Selected source account ID)
- isProcessing: boolean (Transfer in progress)
- transferResult: TransferResult | null (Transfer result)
- validationErrors: ValidationErrors (Form validation errors)

Methods:
- handleAmountChange() → Validates and updates amount
- handleSourceChange() → Updates selected source account
- handleTransferSubmit() → Submits transfer request
- validateTransfer() → Validates transfer data
- resetForm() → Resets form to initial state

HTML STRUCTURE (PSEUDO CODE ONLY)

<div className="fund-transfer-component">
  <div className="transfer-header">
    <h2>Add Funds to Youth Account</h2>
  </div>
  
  <form className="transfer-form" onSubmit={handleTransferSubmit}>
    <div className="form-group">
      <label htmlFor="source-account">From Account</label>
      <select 
        id="source-account"
        value={selectedSource}
        onChange={handleSourceChange}
        className="account-select"
      >
        {sourceAccounts.map(account => (
          <option key={account.id} value={account.id}>
            {account.name} - {formatCurrency(account.balance, account.currency)}
          </option>
        ))}
      </select>
    </div>
    
    <div className="form-group">
      <label htmlFor="transfer-amount">Amount</label>
      <input 
        id="transfer-amount"
        type="number"
        value={transferAmount}
        onChange={handleAmountChange}
        className="amount-input"
        min="0"
        step="0.01"
      />
      {validationErrors.amount && (
        <span className="error-message">{validationErrors.amount}</span>
      )}
    </div>
    
    <div className="form-actions">
      <ButtonComponent 
        label="Cancel"
        variant="secondary"
        onClick={onTransferCancel}
        disabled={isProcessing}
      />
      <ButtonComponent 
        label="Transfer Funds"
        variant="primary"
        type="submit"
        processing={isProcessing}
        disabled={!isFormValid()}
      />
    </div>
  </form>
  
  {transferResult && (
    <div className="transfer-result">
      <AlertComponent 
        type={transferResult.success ? "success" : "error"}
        message={transferResult.message}
        onDismiss={() => setTransferResult(null)}
      />
    </div>
  )}
</div>

    </div>
  ) : (
    <ul className="transaction-items">
      {transactions.map(transaction => (
        <li 
          key={transaction.id}
          className={`transaction-item ${selectedTransaction === transaction.id ? 'selected' : ''}`}
          onClick={() => handleTransactionClick(transaction)}
        >
          <div className="transaction-icon">
            <span className={`icon ${getTransactionIcon(transaction.type)}`} />
          </div>
          
          <div className="transaction-details">
            <div className="transaction-main">
              <span className="transaction-description">{transaction.description}</span>
              <span className={`transaction-amount ${getTransactionColor(transaction.type)}`}>
                {transaction.type === 'debit' ? '-' : '+'}{formatCurrency(transaction.amount, currency)}
              </span>
            </div>
            
            <div className="transaction-meta">
              <span className="transaction-date">{formatDate(transaction.date)}</span>
              <span className="transaction-category">{transaction.category}</span>
            </div>
          </div>
          
          <div className="transaction-status">
            <span className={`status-badge ${transaction.status}`}>
              {transaction.status}
            </span>
          </div>
        </li>
      ))}
    </ul>
  )}
</div>
