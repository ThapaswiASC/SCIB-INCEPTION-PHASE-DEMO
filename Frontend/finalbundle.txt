```typescript
import { Component, OnInit, OnDestroy } from '@angular/core';
import { Router } from '@angular/router';
import { Subject } from 'rxjs';
import { takeUntil } from 'rxjs/operators';

export interface DocumentUploadPageProps {
  sessionId?: string;
  userId: string;
  allowedFileTypes: string[];
  maxFileSize: number;
  enableCamera: boolean;
  enableQualityValidation: boolean;
}

export interface DocumentUploadPageState {
  isLoading: boolean;
  currentStep: number;
  uploadProgress: number;
  error: string | null;
  sessionData: any;
  userProfile: any;
}

@Component({
  selector: 'app-document-upload-page',
  templateUrl: './DocumentUploadPage.component.html',
  styleUrls: ['./DocumentUploadPage.component.css']
})
export class DocumentUploadPage implements OnInit, OnDestroy {
  private destroy$ = new Subject<void>();
  
  // Props
  sessionId?: string;
  userId: string = '';
  allowedFileTypes: string[] = ['pdf', 'jpg', 'png'];
  maxFileSize: number = 10485760; // 10MB
  enableCamera: boolean = true;
  enableQualityValidation: boolean = true;
  
  // State
  isLoading: boolean = false;
  currentStep: number = 1;
  uploadProgress: number = 0;
  error: string | null = null;
  sessionData: any = null;
  userProfile: any = null;
  
  constructor(
    private router: Router,
    private sessionService: SessionService,
    private userService: UserService
  ) {}
  
  ngOnInit(): void {
    this.initializeSession();
    this.loadUserProfile();
  }
  
  ngOnDestroy(): void {
    this.destroy$.next();
    this.destroy$.complete();
  }
  
  private async initializeSession(): Promise<void> {
    try {
      this.isLoading = true;
      const session = await this.sessionService.initializeSession(this.userId);
      this.sessionData = session;
      this.sessionId = session.id;
    } catch (error) {
      this.handleError('Failed to initialize session', error);
    } finally {
      this.isLoading = false;
    }
  }
  
  private async loadUserProfile(): Promise<void> {
    try {
      this.userProfile = await this.userService.getUserProfile(this.userId);
    } catch (error) {
      this.handleError('Failed to load user profile', error);
    }
  }
  
  onStepChange(step: number): void {
    this.currentStep = step;
  }
  
  onUploadProgress(progress: number): void {
    this.uploadProgress = progress;
  }
  
  onUploadComplete(): void {
    this.router.navigate(['/upload-success']);
  }
  
  onUploadError(error: string): void {
    this.handleError('Upload failed', error);
  }
  
  private handleError(message: string, error: any): void {
    console.error(message, error);
    this.error = message;
    this.isLoading = false;
  }
  
  retryOperation(): void {
    this.error = null;
    this.initializeSession();
  }
}
```

```typescript
import { Component, Input, Output, EventEmitter } from '@angular/core';

export interface SidebarStep {
  id: number;
  title: string;
  description: string;
  completed: boolean;
  active: boolean;
}

export interface SidebarComponentProps {
  currentStep: number;
  totalSteps: number;
  isCollapsed?: boolean;
}

@Component({
  selector: 'app-sidebar',
  templateUrl: './SidebarComponent.component.html',
  styleUrls: ['./SidebarComponent.component.css']
})
export class SidebarComponent {
  @Input() currentStep: number = 1;
  @Input() totalSteps: number = 4;
  @Input() isCollapsed: boolean = false;
  
  @Output() stepChange = new EventEmitter<number>();
  @Output() toggleCollapse = new EventEmitter<boolean>();
  
  steps: SidebarStep[] = [
    {
      id: 1,
      title: 'Upload Method',
      description: 'Choose upload method',
      completed: false,
      active: true
    },
    {
      id: 2,
      title: 'Document Capture',
      description: 'Capture or select document',
      completed: false,
      active: false
    },
    {
      id: 3,
      title: 'Quality Check',
      description: 'Validate document quality',
      completed: false,
      active: false
    },
    {
      id: 4,
      title: 'Upload & Process',
      description: 'Upload and process document',
      completed: false,
      active: false
    }
  ];
  
  ngOnInit(): void {
    this.updateSteps();
  }
  
  ngOnChanges(): void {
    this.updateSteps();
  }
  
  private updateSteps(): void {
    this.steps = this.steps.map(step => ({
      ...step,
      completed: step.id < this.currentStep,
      active: step.id === this.currentStep
    }));
  }
  
  onStepClick(stepId: number): void {
    if (stepId <= this.currentStep) {
      this.stepChange.emit(stepId);
    }
  }
  
  onToggleCollapse(): void {
    this.isCollapsed = !this.isCollapsed;
    this.toggleCollapse.emit(this.isCollapsed);
  }
  
  getStepIcon(step: SidebarStep): string {
    if (step.completed) return '✓';
    if (step.active) return step.id.toString();
    return step.id.toString();
  }
  
  getStepStatus(step: SidebarStep): string {
    if (step.completed) return 'completed';
    if (step.active) return 'active';
    return 'pending';
  }
}
```

```typescript
  @Input() sessionId: string = '';
  @Input() allowedFileTypes: string[] = ['pdf', 'jpg', 'png'];
  @Input() maxFileSize: number = 10485760; // 10MB
  @Input() enableCamera: boolean = true;
  @Input() enableQualityValidation: boolean = true;
  
  @Output() stepChange = new EventEmitter<number>();
  @Output() uploadProgress = new EventEmitter<number>();
  @Output() uploadComplete = new EventEmitter<void>();
  @Output() uploadError = new EventEmitter<string>();
  
  // State
  currentStep: number = 1;
  selectedMethod: 'camera' | 'file' | null = null;
  uploadedFile: File | null = null;
  capturedImage: Blob | null = null;
  uploadProgress: number = 0;
  isUploading: boolean = false;
  validationResult: any = null;
  error: string | null = null;
  
  uploadMethods: UploadMethod[] = [
    {
      id: 'camera',
      title: 'Camera Capture',
      description: 'Take a photo using your device camera',
      icon: '📷',
      enabled: this.enableCamera
    },
    {
      id: 'file',
      title: 'File Upload',
      description: 'Select a file from your device',
      icon: '📁',
      enabled: true
    }
  ];
  
  constructor(
    private uploadService: UploadService,
    private validationService: ValidationService
  ) {}
  
  ngOnInit(): void {
    this.updateUploadMethods();
  }
  
  private updateUploadMethods(): void {
    this.uploadMethods = this.uploadMethods.map(method => ({
      ...method,
      enabled: method.id === 'camera' ? this.enableCamera : true
    }));
  }
  
  onMethodSelect(method: 'camera' | 'file'): void {
    this.selectedMethod = method;
    this.currentStep = 2;
    this.stepChange.emit(this.currentStep);
    this.error = null;
  }
  
  onFileSelected(event: Event): void {
    const input = event.target as HTMLInputElement;
    if (input.files && input.files[0]) {
      const file = input.files[0];
      
      if (this.validateFile(file)) {
        this.uploadedFile = file;
        this.proceedToValidation();
      }
    }
  }
  
  onImageCaptured(imageBlob: Blob): void {
    this.capturedImage = imageBlob;
    this.proceedToValidation();
  }
  
  onCameraError(error: string): void {
    this.uploadError.emit(`Camera error: ${error}`);
  }
  
  private validateFile(file: File): boolean {
    // Check file type
    const fileExtension = file.name.split('.').pop()?.toLowerCase();
    if (!fileExtension || !this.allowedFileTypes.includes(fileExtension)) {
      this.uploadError.emit(`File type not allowed. Allowed types: ${this.allowedFileTypes.join(', ')}`);
      return false;
    }
    
    // Check file size
    if (file.size > this.maxFileSize) {
      const maxSizeMB = this.maxFileSize / (1024 * 1024);
      this.uploadError.emit(`File size too large. Maximum size: ${maxSizeMB}MB`);
      return false;
    }
    
    return true;
  }
  
  private proceedToValidation(): void {
    if (this.enableQualityValidation) {
      this.currentStep = 3;
      this.stepChange.emit(this.currentStep);
      this.performQualityValidation();
    } else {
      this.proceedToUpload();
    }
  }
  
  private async performQualityValidation(): Promise<void> {
    try {
      const fileToValidate = this.uploadedFile || this.capturedImage;
      if (!fileToValidate) return;
      
      this.validationResult = await this.validationService.validateDocument(fileToValidate);
      
      if (this.validationResult.isValid) {
        this.proceedToUpload();
      } else {
        // Show validation issues to user
        this.error = 'Document quality validation failed. Please retake or reselect.';
      }
    } catch (error) {
      this.uploadError.emit('Quality validation failed');
    }
  }
  
  onValidationRetry(): void {
    this.error = null;
    this.validationResult = null;
    this.currentStep = 2;
    this.stepChange.emit(this.currentStep);
  }
  
  onValidationProceed(): void {
    this.proceedToUpload();
  }
  
  private async proceedToUpload(): Promise<void> {
    this.currentStep = 4;
    this.stepChange.emit(this.currentStep);
    
    try {
      this.isUploading = true;
      const fileToUpload = this.uploadedFile || this.capturedImage;
      
      if (!fileToUpload) {
        throw new Error('No file to upload');
      }
      
      await this.uploadService.uploadDocument(
        this.sessionId,
        fileToUpload,
        (progress) => {
          this.uploadProgress = progress;
          this.uploadProgress.emit(progress);
        }
      );
      
      this.uploadComplete.emit();
    } catch (error) {
      this.uploadError.emit('Upload failed');
    } finally {
      this.isUploading = false;
    }
  }
  
  onRetryUpload(): void {
    this.error = null;
    this.uploadProgress = 0;
    this.proceedToUpload();
  }
  
  onStartOver(): void {
    this.selectedMethod = null;
    this.uploadedFile = null;
    this.capturedImage = null;
    this.uploadProgress = 0;
    this.validationResult = null;
    this.error = null;
    this.currentStep = 1;
    this.stepChange.emit(this.currentStep);
  }
  
  getAcceptedFileTypes(): string {
    return this.allowedFileTypes.map(type => `.${type}`).join(',');
  }
  
  formatFileSize(bytes: number): string {
    const mb = bytes / (1024 * 1024);
    return `${mb.toFixed(1)}MB`;
  }
}
```

**TypeScript Specification:**
```typescript
import { Component, Input, Output, EventEmitter, ViewChild, ElementRef, OnDestroy } from '@angular/core';

export interface CameraComponentProps {
  sessionId: string;
  facingMode?: 'user' | 'environment';
  width?: number;
  height?: number;
}

export interface CameraState {
  isInitialized: boolean;
  isCapturing: boolean;
  hasPermission: boolean;
  error: string | null;
  stream: MediaStream | null;
  capturedImage: string | null;
}

@Component({
  selector: 'app-camera',
  templateUrl: './CameraComponent.component.html',
  styleUrls: ['./CameraComponent.component.css']
})
export class CameraComponent implements OnDestroy {
  @ViewChild('videoElement', { static: false }) videoElement!: ElementRef<HTMLVideoElement>;
  @ViewChild('canvasElement', { static: false }) canvasElement!: ElementRef<HTMLCanvasElement>;
  
  @Input() sessionId: string = '';
  @Input() facingMode: 'user' | 'environment' = 'environment';
  @Input() width: number = 1280;
  @Input() height: number = 720;
  
  @Output() imageCaptured = new EventEmitter<Blob>();
  @Output() error = new EventEmitter<string>();
  
  // State
  isInitialized: boolean = false;
  isCapturing: boolean = false;
  hasPermission: boolean = false;
  errorMessage: string | null = null;
  stream: MediaStream | null = null;
  capturedImage: string | null = null;
  
  private constraints: MediaStreamConstraints = {
    video: {
      facingMode: this.facingMode,
      width: { ideal: this.width },
      height: { ideal: this.height }
    },
    audio: false
  };
  
  ngOnInit(): void {
    this.initializeCamera();
  }
  
  ngOnDestroy(): void {
    this.stopCamera();
  }
  
  private async initializeCamera(): Promise<void> {
    try {
      // Check if camera is supported
      if (!navigator.mediaDevices || !navigator.mediaDevices.getUserMedia) {
        throw new Error('Camera not supported on this device');
      }
      
      // Request camera permission
      this.stream = await navigator.mediaDevices.getUserMedia(this.constraints);
      this.hasPermission = true;
      
      // Wait for video element to be available
      setTimeout(() => {
        if (this.videoElement && this.videoElement.nativeElement) {
          this.videoElement.nativeElement.srcObject = this.stream;
          this.videoElement.nativeElement.play();
          this.isInitialized = true;
        }
      }, 100);
      
    } catch (error: any) {
      this.handleCameraError(error);
    }
  }
  
  private handleCameraError(error: any): void {
    let errorMessage = 'Camera initialization failed';
    
    if (error.name === 'NotAllowedError') {
      errorMessage = 'Camera permission denied. Please allow camera access and try again.';
    } else if (error.name === 'NotFoundError') {
      errorMessage = 'No camera found on this device.';
    } else if (error.name === 'NotReadableError') {
      errorMessage = 'Camera is already in use by another application.';
    } else if (error.name === 'OverconstrainedError') {
      errorMessage = 'Camera does not support the requested configuration.';
    }
    
    this.errorMessage = errorMessage;
    this.error.emit(errorMessage);
  }
  
  async captureImage(): Promise<void> {
    if (!this.videoElement || !this.canvasElement || !this.isInitialized) {
      this.error.emit('Camera not ready');
      return;
    }
    
    try {
      this.isCapturing = true;
      
      const video = this.videoElement.nativeElement;
      const canvas = this.canvasElement.nativeElement;
      const context = canvas.getContext('2d');
      
      if (!context) {
        throw new Error('Canvas context not available');
      }
      
      // Set canvas dimensions to match video
      canvas.width = video.videoWidth;
      canvas.height = video.videoHeight;
      
      // Draw video frame to canvas
      context.drawImage(video, 0, 0, canvas.width, canvas.height);
      
      // Convert canvas to blob
      canvas.toBlob((blob) => {
        if (blob) {
          this.capturedImage = canvas.toDataURL('image/jpeg', 0.8);
          this.imageCaptured.emit(blob);
        } else {
          this.error.emit('Failed to capture image');
        }
        this.isCapturing = false;
      }, 'image/jpeg', 0.8);
      
    } catch (error) {
      this.error.emit('Image capture failed');
      this.isCapturing = false;
    }
  }
  
  retakeImage(): void {
    this.capturedImage = null;
  }
  
  switchCamera(): void {
    this.facingMode = this.facingMode === 'user' ? 'environment' : 'user';
    this.constraints.video = {
      ...this.constraints.video as MediaTrackConstraints,
      facingMode: this.facingMode
    };
    
    this.stopCamera();
    this.initializeCamera();
  }
  
  private stopCamera(): void {
    if (this.stream) {
      this.stream.getTracks().forEach(track => track.stop());
      this.stream = null;
    }
    this.isInitialized = false;
  }
  
  requestPermission(): void {
    this.errorMessage = null;
    this.initializeCamera();
  }
  
  getCameraIcon(): string {
    return this.facingMode === 'user' ? '🤳' : '📷';
  }
  
  getSwitchCameraLabel(): string {
    return `Switch to ${this.facingMode === 'user' ? 'back' : 'front'} camera`;
  }
}
```
