### 3.1 KanbanBoardComponent

**Component Type:** Feature
**Purpose:** Primary container component that manages the overall Kanban board state and coordinates between columns

**TypeScript Specification:**

```typescript
// Inputs:
- boardId: string (board identifier)
- permissions: Permission[] (user permissions)
- loading: boolean (loading state)

// Outputs:
- onColumnUpdate: EventEmitter<{columnId: string, data: any}> (column update payload)
- onCardMove: EventEmitter<{cardId: string, fromColumn: string, toColumn: string}> (card move payload)
- onError: EventEmitter<string> (error message payload)

// State:
- columns: KanbanColumn[] (current column data)
- isLoading: boolean (loading indicator)
- errorMessage: string | null (error state)
- draggedCard: KanbanCard | null (currently dragged card)

// Methods:
- ngOnInit() → Initialize board data
- loadBoardData() → Fetch board from API
- handleCardMove(event: CdkDragDrop<KanbanCard[]>) → Process card movement
- handleError(error: string) → Handle error states
- refreshBoard() → Reload board data
```

```html

      *ngFor="let column of columns; trackBy: trackByColumnId"
```

### 3.2 KanbanColumnComponent

```typescript
// Inputs:
```

```html
  <div class="column-content" *ngIf="isExpanded">
    <div class="cards-container">
      <kanban-card-component
        *ngFor="let card of cards; trackBy: trackByCardId"
        [cardData]="card"

      </kanban-card-component>
    </div>

      <add-card-button-component
```

### 3.3 KanbanCardComponent

**Component Type:** Feature
**Purpose:** Displays card content, handles editing, and provides drag functionality

**TypeScript Specification:**

```typescript
// Inputs:
- cardData: KanbanCard (card information)
- draggable: boolean (enable drag functionality)
- editable: boolean (allow editing)
- compact: boolean (compact display mode)

// Outputs:
- onCardEdit: EventEmitter<{cardId: string, data: Partial<KanbanCard>}> (edit payload)
- onCardDelete: EventEmitter<string> (delete payload)
- onDragStart: EventEmitter<string> (drag start payload)
- onCardClick: EventEmitter<string> (card click payload)

// State:
- isEditing: boolean (edit mode state)
- isDragging: boolean (drag state)
- isHovered: boolean (hover state)
- editableTitle: string (temporary edit value)

// Methods:
- startEdit() → Enter edit mode
- saveEdit() → Save card changes
- cancelEdit() → Cancel editing
- deleteCard() → Delete card with confirmation
- onDragStart() → Handle drag start
- onDragEnd() → Handle drag end
```

**HTML Structure:**

```html
<div class="kanban-card"
     [class.editing]="isEditing"
     [class.dragging]="isDragging"
     [class.compact]="compact"
     cdkDrag
     [cdkDragData]="cardData"
     (cdkDragStarted)="onDragStart()"
     (cdkDragEnded)="onDragEnd()"
     (click)="onCardClick.emit(cardData.id)">
  
  <div class="card-header">
    <div class="card-title" *ngIf="!isEditing">
      {{cardData.title}}
    </div>
    <input-component
      *ngIf="isEditing"
      [(ngModel)]="editableTitle"
      (keyup.enter)="saveEdit()"
      (keyup.escape)="cancelEdit()"
      class="title-input">
    </input-component>
    
    <div class="card-actions">
      <button-component
        *ngIf="!isEditing && editable"
        type="ghost"
        icon="edit"
        size="small"
        (click)="startEdit(); $event.stopPropagation()">
      </button-component>
      
      <button-component
        *ngIf="!isEditing"
        type="ghost"
        icon="delete"
        size="small"
        (click)="deleteCard(); $event.stopPropagation()">
      </button-component>
    </div>
  </div>
  
  <div class="card-content" *ngIf="!compact">
    <div class="card-description" *ngIf="cardData.description">
      {{cardData.description}}
    </div>
    
    <div class="card-metadata">
      <badge-component
        *ngIf="cardData.priority"
        [type]="cardData.priority"
        [text]="cardData.priority">
      </badge-component>
      
      <div class="card-assignee" *ngIf="cardData.assignee">
        <avatar-component
          [user]="cardData.assignee"
          size="small">
        </avatar-component>
      </div>
      
      <div class="card-due-date" *ngIf="cardData.dueDate">
        {{cardData.dueDate | date:'short'}}
      </div>
    </div>
  </div>
  
  <div class="edit-actions" *ngIf="isEditing">
    <button-component type="primary" size="small" (click)="saveEdit()">Save</button-component>
    <button-component type="secondary" size="small" (click)="cancelEdit()">Cancel</button-component>
  </div>
</div>
```
