## 1. KANBAN BOARD COMPONENT

### TypeScript Specification (kanban-board.component.ts)

```typescript
import { Component, OnInit, Input, Output, EventEmitter, ChangeDetectionStrategy } from '@angular/core';
import { CdkDragDrop, moveItemInArray, transferArrayItem } from '@angular/cdk/drag-drop';
import { Observable, BehaviorSubject, combineLatest } from 'rxjs';
import { map, catchError, finalize } from 'rxjs/operators';

export interface Task {
  id: string;
  title: string;
  description?: string;
  priority: 'low' | 'medium' | 'high' | 'urgent';
  assignee?: {
    id: string;
    name: string;
    avatar?: string;
  };
  tags: string[];
  dueDate?: Date;
  createdAt: Date;
  updatedAt: Date;
  status: 'todo' | 'inprogress' | 'done';
  estimatedHours?: number;
  actualHours?: number;
}

export interface KanbanColumn {
  id: string;
  title: string;
  tasks: Task[];
  maxTasks?: number;
  color?: string;
}

export interface TaskMoveEvent {
  taskId: string;
  fromColumn: string;
  toColumn: string;
  newIndex: number;
}

@Component({
  selector: 'app-kanban-board',
  templateUrl: './kanban-board.component.html',
  styleUrls: ['./kanban-board.component.scss'],
  changeDetection: ChangeDetectionStrategy.OnPush
})
export class KanbanBoardComponent implements OnInit {
  @Input() youthAccountId!: string;
  @Input() initialColumns: KanbanColumn[] = [];
  @Input() loading = false;
  @Input() error: string | null = null;
  
  @Output() taskMoved = new EventEmitter<TaskMoveEvent>();
  @Output() taskCreated = new EventEmitter<Task>();
  @Output() taskUpdated = new EventEmitter<Task>();
  @Output() taskDeleted = new EventEmitter<string>();
  @Output() errorOccurred = new EventEmitter<string>();

  // State management
  private columnsSubject = new BehaviorSubject<KanbanColumn[]>([]);
  public columns$ = this.columnsSubject.asObservable();
  
  private loadingSubject = new BehaviorSubject<boolean>(false);
  public loading$ = this.loadingSubject.asObservable();
  
  private errorSubject = new BehaviorSubject<string | null>(null);
  public error$ = this.errorSubject.asObservable();

  // Drag and drop state
  public draggedTask: Task | null = null;
  public isDragActive = false;

  // Mobile responsive state
  public isMobile = false;
  public activeColumnIndex = 0;

  constructor(
    private kanbanService: KanbanService,
    private breakpointObserver: BreakpointObserver,
    private cdr: ChangeDetectorRef
  ) {}

  ngOnInit(): void {
    this.initializeColumns();
    this.setupResponsiveBreakpoints();
    this.loadKanbanData();
  }

  private initializeColumns(): void {
    const defaultColumns: KanbanColumn[] = [
      {
        id: 'todo',
        title: 'To Do',
        tasks: [],
        color: '#e3f2fd'
      },
      {
        id: 'inprogress',
        title: 'In Progress',
        tasks: [],
        maxTasks: 5,
        color: '#fff3e0'
      },
      {
        id: 'done',
        title: 'Done',
        tasks: [],
        color: '#e8f5e8'
      }
    ];
    
    this.columnsSubject.next(this.initialColumns.length > 0 ? this.initialColumns : defaultColumns);
  }

  private setupResponsiveBreakpoints(): void {
    this.breakpointObserver.observe(['(max-width: 768px)'])
      .subscribe(result => {
        this.isMobile = result.matches;
        this.cdr.markForCheck();
      });
  }

  private loadKanbanData(): void {
    if (!this.youthAccountId) {
      this.handleError('Youth Account ID is required');
      return;
    }

    this.loadingSubject.next(true);
    this.errorSubject.next(null);

    this.kanbanService.getDashboardData(this.youthAccountId)
      .pipe(
        map(response => this.transformApiResponseToKanbanData(response)),
        catchError(error => {
          this.handleError('Failed to load kanban data: ' + error.message);
          return [];
        }),
        finalize(() => this.loadingSubject.next(false))
      )
      .subscribe(columns => {
        this.columnsSubject.next(columns);
        this.cdr.markForCheck();
      });
  }

  private transformApiResponseToKanbanData(response: any): KanbanColumn[] {
    // Transform youth account dashboard data to kanban format
    const columns = this.columnsSubject.value;
    
    // Map transactions to tasks (example transformation)
    if (response.transactions) {
      const tasks: Task[] = response.transactions.map((transaction: any) => ({
        id: transaction.id,
        title: transaction.description || 'Task',
        description: transaction.notes,
        priority: this.mapAmountToPriority(transaction.amount),
        status: this.mapTransactionStatusToTaskStatus(transaction.status),
        createdAt: new Date(transaction.createdAt),
        updatedAt: new Date(transaction.updatedAt),
        tags: [transaction.category || 'general'],
        estimatedHours: Math.ceil(transaction.amount / 100) // Example mapping
      }));

      // Distribute tasks to columns based on status
      columns.forEach(column => {
        column.tasks = tasks.filter(task => task.status === column.id);
      });
    }

    return columns;
  }

  private mapAmountToPriority(amount: number): Task['priority'] {
    if (amount > 1000) return 'urgent';
    if (amount > 500) return 'high';
    if (amount > 100) return 'medium';
    return 'low';
  }

  private mapTransactionStatusToTaskStatus(status: string): Task['status'] {
    switch (status.toLowerCase()) {
      case 'pending': return 'todo';
      case 'processing': return 'inprogress';
      case 'completed': return 'done';
      default: return 'todo';
    }
  }

  // Drag and Drop Methods
  onTaskDragStart(task: Task): void {
    this.draggedTask = task;
    this.isDragActive = true;
  }

  onTaskDragEnd(): void {
    this.draggedTask = null;
    this.isDragActive = false;
  }

  onTaskDrop(event: CdkDragDrop<Task[]>, columnId: string): void {
    const task = event.item.data as Task;
    const fromColumnId = task.status;
    
    if (event.previousContainer === event.container) {
      // Reorder within same column
      moveItemInArray(event.container.data, event.previousIndex, event.currentIndex);
    } else {
      // Move between columns
      transferArrayItem(
        event.previousContainer.data,
        event.container.data,
        event.previousIndex,
        event.currentIndex
      );
      
      // Update task status
      task.status = columnId as Task['status'];
      task.updatedAt = new Date();
      
      // Emit task move event
      this.taskMoved.emit({
        taskId: task.id,
        fromColumn: fromColumnId,
        toColumn: columnId,
        newIndex: event.currentIndex
      });
      
      // Update backend
      this.updateTaskStatus(task);
    }
    
    this.onTaskDragEnd();
  }

  private updateTaskStatus(task: Task): void {
    this.kanbanService.updateTaskStatus(this.youthAccountId, task.id, task.status)
      .pipe(
        catchError(error => {
          this.handleError('Failed to update task status: ' + error.message);
          // Revert the change on error
          this.loadKanbanData();
          return [];
        })
      )
      .subscribe(() => {
        this.taskUpdated.emit(task);
      });
  }

  // Task Management Methods
  onTaskCreated(task: Task): void {
    this.kanbanService.createTask(this.youthAccountId, task)
      .pipe(
        catchError(error => {
          this.handleError('Failed to create task: ' + error.message);
          return [];
        })
      )
      .subscribe(createdTask => {
        this.addTaskToColumn(createdTask);
        this.taskCreated.emit(createdTask);
      });
  }

  private addTaskToColumn(task: Task): void {
    const columns = this.columnsSubject.value;
    const targetColumn = columns.find(col => col.id === task.status);
    
    if (targetColumn) {
      targetColumn.tasks.push(task);
      this.columnsSubject.next([...columns]);
      this.cdr.markForCheck();
    }
  }

  onTaskDeleted(taskId: string): void {
    this.kanbanService.deleteTask(this.youthAccountId, taskId)
      .pipe(
        catchError(error => {
          this.handleError('Failed to delete task: ' + error.message);
          return [];
        })
      )
      .subscribe(() => {
        this.removeTaskFromColumns(taskId);
        this.taskDeleted.emit(taskId);
      });
  }

  private removeTaskFromColumns(taskId: string): void {
    const columns = this.columnsSubject.value;
    columns.forEach(column => {
      column.tasks = column.tasks.filter(task => task.id !== taskId);
    });
    this.columnsSubject.next([...columns]);
    this.cdr.markForCheck();
  }

  // Mobile Methods
  onMobileColumnChange(columnIndex: number): void {
    this.activeColumnIndex = columnIndex;
  }

  getActiveColumn(): KanbanColumn | null {
    const columns = this.columnsSubject.value;
    return columns[this.activeColumnIndex] || null;
  }

  // Error Handling
  private handleError(message: string): void {
    this.errorSubject.next(message);
    this.errorOccurred.emit(message);
    console.error('KanbanBoard Error:', message);
  }

  onRetryLoad(): void {
    this.loadKanbanData();
  }

  // Accessibility Methods
  announceTaskMove(task: Task, fromColumn: string, toColumn: string): void {
    const message = `Task "${task.title}" moved from ${fromColumn} to ${toColumn}`;
    // Implement screen reader announcement
    this.announceToScreenReader(message);
  }

  private announceToScreenReader(message: string): void {
    // Implementation for screen reader announcements
    const announcement = document.createElement('div');
    announcement.setAttribute('aria-live', 'polite');
    announcement.setAttribute('aria-atomic', 'true');
    announcement.className = 'sr-only';
    announcement.textContent = message;
    document.body.appendChild(announcement);
    
    setTimeout(() => {
      document.body.removeChild(announcement);
    }, 1000);
  }

  // Keyboard Navigation
  onKeyboardNavigation(event: KeyboardEvent, task: Task): void {
    switch (event.key) {
      case 'ArrowRight':
        this.moveTaskToNextColumn(task);
        break;
      case 'ArrowLeft':
        this.moveTaskToPreviousColumn(task);
        break;
      case 'Delete':
        if (event.ctrlKey) {
          this.onTaskDeleted(task.id);
        }
        break;
    }
  }

  private moveTaskToNextColumn(task: Task): void {
    const columns = this.columnsSubject.value;
    const currentColumnIndex = columns.findIndex(col => col.id === task.status);
    
    if (currentColumnIndex < columns.length - 1) {
      const nextColumn = columns[currentColumnIndex + 1];
      this.moveTaskBetweenColumns(task, task.status, nextColumn.id);
    }
  }

  private moveTaskToPreviousColumn(task: Task): void {
    const columns = this.columnsSubject.value;
    const currentColumnIndex = columns.findIndex(col => col.id === task.status);
    
    if (currentColumnIndex > 0) {
      const previousColumn = columns[currentColumnIndex - 1];
      this.moveTaskBetweenColumns(task, task.status, previousColumn.id);
    }
  }

  private moveTaskBetweenColumns(task: Task, fromColumnId: string, toColumnId: string): void {
    const columns = this.columnsSubject.value;
    const fromColumn = columns.find(col => col.id === fromColumnId);
    const toColumn = columns.find(col => col.id === toColumnId);
    
    if (fromColumn && toColumn) {
      // Remove from source column
      const taskIndex = fromColumn.tasks.findIndex(t => t.id === task.id);
      if (taskIndex > -1) {
        fromColumn.tasks.splice(taskIndex, 1);
        
        // Add to target column
        task.status = toColumnId as Task['status'];
        task.updatedAt = new Date();
        toColumn.tasks.push(task);
        
        // Update state
        this.columnsSubject.next([...columns]);
        
        // Emit events
        this.taskMoved.emit({
          taskId: task.id,
          fromColumn: fromColumnId,
          toColumn: toColumnId,
          newIndex: toColumn.tasks.length - 1
        });
        
        // Update backend
        this.updateTaskStatus(task);
        
        // Announce to screen readers
        this.announceTaskMove(task, fromColumn.title, toColumn.title);
      }
    }
  }
}
```

```typescript
import { Component, Input, Output, EventEmitter, ChangeDetectionStrategy } from '@angular/core';
import { CdkDragDrop } from '@angular/cdk/drag-drop';
import { Task, KanbanColumn } from '../kanban-board/kanban-board.component';

@Component({
  selector: 'app-kanban-column',
  templateUrl: './kanban-column.component.html',
  styleUrls: ['./kanban-column.component.scss'],
  changeDetection: ChangeDetectionStrategy.OnPush
})
export class KanbanColumnComponent {
  @Input() column!: KanbanColumn;
  @Input() isDragActive = false;
  @Input() maxTasks?: number;
  @Input() isMobile = false;
  
  @Output() taskDropped = new EventEmitter<CdkDragDrop<Task[]>>();
  @Output() taskDragStart = new EventEmitter<Task>();
  @Output() taskDragEnd = new EventEmitter<void>();
  @Output() taskCreated = new EventEmitter<Task>();
  @Output() taskDeleted = new EventEmitter<string>();
  @Output() taskClicked = new EventEmitter<Task>();
  @Output() addTaskClicked = new EventEmitter<string>();

  public isDragOver = false;
  public showAddTaskForm = false;

  get isAtMaxCapacity(): boolean {
    return this.maxTasks ? this.column.tasks.length >= this.maxTasks : false;
  }

  get taskCountText(): string {
    const count = this.column.tasks.length;
    const maxText = this.maxTasks ? ` of ${this.maxTasks}` : '';
    return `${count}${maxText} task${count !== 1 ? 's' : ''}`;
  }

  onDragEnter(): void {
    if (!this.isAtMaxCapacity) {
      this.isDragOver = true;
    }
  }

  onDragLeave(): void {
    this.isDragOver = false;
  }

  onDrop(event: CdkDragDrop<Task[]>): void {
    this.isDragOver = false;
    
    if (this.isAtMaxCapacity && event.previousContainer !== event.container) {
      // Prevent drop if at max capacity
      return;
    }
    
    this.taskDropped.emit(event);
  }

  onTaskDragStart(task: Task): void {
    this.taskDragStart.emit(task);
  }

  onTaskDragEnd(): void {
    this.taskDragEnd.emit();
  }

  onTaskClick(task: Task): void {
    this.taskClicked.emit(task);
  }

  onTaskDelete(taskId: string): void {
    this.taskDeleted.emit(taskId);
  }

  onAddTaskClick(): void {
    this.showAddTaskForm = true;
    this.addTaskClicked.emit(this.column.id);
  }

  onTaskFormSubmit(taskData: Partial<Task>): void {
    const newTask: Task = {
      id: this.generateTaskId(),
      title: taskData.title || 'New Task',
      description: taskData.description,
      priority: taskData.priority || 'medium',
      status: this.column.id as Task['status'],
      tags: taskData.tags || [],
      createdAt: new Date(),
      updatedAt: new Date(),
      assignee: taskData.assignee,
      dueDate: taskData.dueDate,
      estimatedHours: taskData.estimatedHours
    };
    
    this.taskCreated.emit(newTask);
    this.showAddTaskForm = false;
  }

  onCancelAddTask(): void {
    this.showAddTaskForm = false;
  }

  private generateTaskId(): string {
    return `task_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`;
  }

  trackByTaskId(index: number, task: Task): string {
    return task.id;
  }
}
```

## 3. TASK CARD COMPONENT

### TypeScript Specification (task-card.component.ts)

```typescript
import { Component, Input, Output, EventEmitter, ChangeDetectionStrategy } from '@angular/core';
import { Task } from '../kanban-board/kanban-board.component';

@Component({
  selector: 'app-task-card',
  templateUrl: './task-card.component.html',
  styleUrls: ['./task-card.component.scss'],
  changeDetection: ChangeDetectionStrategy.OnPush
})
export class TaskCardComponent {
  @Input() task!: Task;
  @Input() draggable = true;
  @Input() showActions = true;
  @Input() compact = false;
  
  @Output() taskClicked = new EventEmitter<Task>();
  @Output() taskDragStart = new EventEmitter<Task>();
  @Output() taskDragEnd = new EventEmitter<void>();
  @Output() taskDeleted = new EventEmitter<string>();
  @Output() taskEdited = new EventEmitter<Task>();

  public isDragging = false;
  public showDeleteConfirm = false;

  get priorityClass(): string {
    return `priority-${this.task.priority}`;
  }

  get priorityLabel(): string {
    const labels = {
      low: 'Low Priority',
      medium: 'Medium Priority', 
      high: 'High Priority',
      urgent: 'Urgent Priority'
    };
    return labels[this.task.priority] || 'Unknown Priority';
  }

  get dueDateStatus(): 'overdue' | 'due-soon' | 'normal' | null {
    if (!this.task.dueDate) return null;
    
    const now = new Date();
    const dueDate = new Date(this.task.dueDate);
    const diffDays = Math.ceil((dueDate.getTime() - now.getTime()) / (1000 * 60 * 60 * 24));
    
    if (diffDays < 0) return 'overdue';
    if (diffDays <= 2) return 'due-soon';
    return 'normal';
  }

  get formattedDueDate(): string {
    if (!this.task.dueDate) return '';
    
    const dueDate = new Date(this.task.dueDate);
    const now = new Date();
    const diffDays = Math.ceil((dueDate.getTime() - now.getTime()) / (1000 * 60 * 60 * 24));
    
    if (diffDays === 0) return 'Due today';
    if (diffDays === 1) return 'Due tomorrow';
    if (diffDays === -1) return 'Due yesterday';
    if (diffDays < -1) return `Overdue by ${Math.abs(diffDays)} days`;
    if (diffDays > 1) return `Due in ${diffDays} days`;
    
    return dueDate.toLocaleDateString();
  }

  onClick(): void {
    if (!this.isDragging) {
      this.taskClicked.emit(this.task);
    }
  }

  onDragStart(): void {
    this.isDragging = true;
    this.taskDragStart.emit(this.task);
  }

  onDragEnd(): void {
    this.isDragging = false;
    this.taskDragEnd.emit();
  }

  onEditClick(event: Event): void {
    event.stopPropagation();
    this.taskEdited.emit(this.task);
  }

  onDeleteClick(event: Event): void {
    event.stopPropagation();
    this.showDeleteConfirm = true;
  }

  onConfirmDelete(): void {
    this.taskDeleted.emit(this.task.id);
    this.showDeleteConfirm = false;
  }

  onCancelDelete(): void {
    this.showDeleteConfirm = false;
  }

  onKeyDown(event: KeyboardEvent): void {
    switch (event.key) {
      case 'Enter':
      case ' ':
        event.preventDefault();
        this.onClick();
        break;
      case 'Delete':
        if (event.ctrlKey) {
          event.preventDefault();
          this.onDeleteClick(event);
        }
        break;
      case 'Escape':
        if (this.showDeleteConfirm) {
          this.onCancelDelete();
        }
        break;
    }
  }

  getTaskAriaLabel(): string {
    let label = `Task: ${this.task.title}`;
    
    if (this.task.description) {
      label += `. Description: ${this.task.description}`;
    }
    
    label += `. Priority: ${this.priorityLabel}`;
    
    if (this.task.assignee) {
      label += `. Assigned to: ${this.task.assignee.name}`;
    }
    
    if (this.task.dueDate) {
      label += `. ${this.formattedDueDate}`;
    }
    
    if (this.task.tags.length > 0) {
      label += `. Tags: ${this.task.tags.join(', ')}`;
    }
    
    return label;
  }
}
```

### TypeScript Specification (header.component.ts)

```typescript
import { Component, Input, Output, EventEmitter, ChangeDetectionStrategy } from '@angular/core';

export interface User {
  id: string;
  name: string;
  email: string;
  avatar?: string;
  role: string;
}

export interface Breadcrumb {
  label: string;
  url?: string;
  active: boolean;
}

@Component({
  selector: 'app-header',
  templateUrl: './header.component.html',
  styleUrls: ['./header.component.scss'],
  changeDetection: ChangeDetectionStrategy.OnPush
})
export class HeaderComponent {
  @Input() user!: User;
  @Input() breadcrumbs: Breadcrumb[] = [];
  @Input() showSearch = true;
  @Input() showNotifications = true;
  @Input() notificationCount = 0;
  
  @Output() searchSubmitted = new EventEmitter<string>();
  @Output() themeToggled = new EventEmitter<void>();
  @Output() menuToggled = new EventEmitter<void>();
  @Output() userMenuClicked = new EventEmitter<void>();
  @Output() notificationsClicked = new EventEmitter<void>();
  @Output() profileClicked = new EventEmitter<void>();
  @Output() logoutClicked = new EventEmitter<void>();

  public searchQuery = '';
  public showUserMenu = false;
  public showNotificationPanel = false;
  public currentTheme: 'light' | 'dark' = 'light';

  get userInitials(): string {
    return this.user?.name
      ?.split(' ')
      .map(n => n[0])
      .join('')
      .toUpperCase() || 'U';
  }

  onSearchSubmit(): void {
    if (this.searchQuery.trim()) {
      this.searchSubmitted.emit(this.searchQuery.trim());
    }
  }

  onSearchKeyDown(event: KeyboardEvent): void {
    if (event.key === 'Enter') {
      this.onSearchSubmit();
    } else if (event.key === 'Escape') {
      this.searchQuery = '';
    }
  }

  onMenuToggle(): void {
    this.menuToggled.emit();
  }

  onThemeToggle(): void {
    this.currentTheme = this.currentTheme === 'light' ? 'dark' : 'light';
    this.themeToggled.emit();
  }

  onUserMenuToggle(): void {
    this.showUserMenu = !this.showUserMenu;
    this.showNotificationPanel = false;
    this.userMenuClicked.emit();
  }

  onNotificationsToggle(): void {
    this.showNotificationPanel = !this.showNotificationPanel;
    this.showUserMenu = false;
    this.notificationsClicked.emit();
  }

  onProfileClick(): void {
    this.showUserMenu = false;
    this.profileClicked.emit();
  }

  onLogoutClick(): void {
    this.showUserMenu = false;
    this.logoutClicked.emit();
  }

  closeMenus(): void {
    this.showUserMenu = false;
    this.showNotificationPanel = false;
  }

  onKeyDown(event: KeyboardEvent): void {
    if (event.key === 'Escape') {
      this.closeMenus();
    }
  }
}
```

### TypeScript Specification (modal.component.ts)

```typescript
import { Component, Input, Output, EventEmitter, OnInit, OnDestroy, ChangeDetectionStrategy, ElementRef, ViewChild } from '@angular/core';
import { Task } from '../kanban-board/kanban-board.component';

export interface TaskFormData {
  title: string;
  description?: string;
  priority: Task['priority'];
  assigneeId?: string;
  tags: string[];
  dueDate?: Date;
  estimatedHours?: number;
}

@Component({
  selector: 'app-modal',
  templateUrl: './modal.component.html',
  styleUrls: ['./modal.component.scss'],
  changeDetection: ChangeDetectionStrategy.OnPush
})
export class ModalComponent implements OnInit, OnDestroy {
  @Input() isOpen = false;
  @Input() task: Task | null = null;
  @Input() columnId?: string;
  @Input() title = 'Task Details';
  @Input() size: 'small' | 'medium' | 'large' = 'medium';
  
  @Output() taskSubmitted = new EventEmitter<Task>();
  @Output() modalClosed = new EventEmitter<void>();
  @Output() taskDeleted = new EventEmitter<string>();
  
  @ViewChild('modalContent', { static: false }) modalContent!: ElementRef;
  @ViewChild('firstFocusable', { static: false }) firstFocusable!: ElementRef;

  public formData: TaskFormData = {
    title: '',
    description: '',
    priority: 'medium',
    tags: [],
    assigneeId: undefined,
    dueDate: undefined,
    estimatedHours: undefined
  };
  
  public isSubmitting = false;
  public errors: { [key: string]: string } = {};
  public showDeleteConfirm = false;
  public newTag = '';
  
  private previousActiveElement: HTMLElement | null = null;

  get isEditMode(): boolean {
    return this.task !== null;
  }

  get modalTitle(): string {
    return this.isEditMode ? `Edit Task: ${this.task?.title}` : 'Create New Task';
  }

  ngOnInit(): void {
    if (this.isOpen) {
      this.initializeForm();
      this.trapFocus();
    }
  }

  ngOnDestroy(): void {
    this.restoreFocus();
  }

  private initializeForm(): void {
    if (this.task) {
      // Edit mode - populate form with existing task data
      this.formData = {
        title: this.task.title,
        description: this.task.description || '',
        priority: this.task.priority,
        assigneeId: this.task.assignee?.id,
        tags: [...this.task.tags],
        dueDate: this.task.dueDate,
        estimatedHours: this.task.estimatedHours
      };
    } else {
      // Create mode - reset form
      this.formData = {
        title: '',
        description: '',
        priority: 'medium',
        tags: [],
        assigneeId: undefined,
        dueDate: undefined,
        estimatedHours: undefined
      };
    }
    
    this.errors = {};
  }

  private trapFocus(): void {
    this.previousActiveElement = document.activeElement as HTMLElement;
    
    // Focus first focusable element after modal opens
    setTimeout(() => {
      if (this.firstFocusable?.nativeElement) {
        this.firstFocusable.nativeElement.focus();
      }
    }, 100);
  }

  private restoreFocus(): void {
    if (this.previousActiveElement) {
      this.previousActiveElement.focus();
    }
  }

  onSubmit(): void {
    if (this.validateForm()) {
      this.isSubmitting = true;
      
      const taskData: Task = {
        id: this.task?.id || this.generateTaskId(),
        title: this.formData.title.trim(),
        description: this.formData.description?.trim(),
        priority: this.formData.priority,
        status: this.task?.status || (this.columnId as Task['status']) || 'todo',
        tags: this.formData.tags,
        createdAt: this.task?.createdAt || new Date(),
        updatedAt: new Date(),
        assignee: this.formData.assigneeId ? {
          id: this.formData.assigneeId,
          name: 'User Name', // This would come from user service
          avatar: undefined
        } : undefined,
        dueDate: this.formData.dueDate,
        estimatedHours: this.formData.estimatedHours
      };
      
      // Simulate API call delay
      setTimeout(() => {
        this.taskSubmitted.emit(taskData);
        this.isSubmitting = false;
        this.closeModal();
      }, 500);
    }
  }

  private validateForm(): boolean {
    this.errors = {};
    
    if (!this.formData.title.trim()) {
      this.errors['title'] = 'Task title is required';
    }
    
    if (this.formData.title.trim().length > 100) {
      this.errors['title'] = 'Task title must be less than 100 characters';
    }
    
    if (this.formData.description && this.formData.description.length > 500) {
      this.errors['description'] = 'Description must be less than 500 characters';
    }
    
    if (this.formData.estimatedHours && (this.formData.estimatedHours < 0 || this.formData.estimatedHours > 1000)) {
      this.errors['estimatedHours'] = 'Estimated hours must be between 0 and 1000';
    }
    
    if (this.formData.dueDate && this.formData.dueDate < new Date()) {
      this.errors['dueDate'] = 'Due date cannot be in the past';
    }
    
    return Object.keys(this.errors).length === 0;
  }

  private generateTaskId(): string {
    return `task_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`;
  }

  closeModal(): void {
    this.modalClosed.emit();
    this.restoreFocus();
  }

  onBackdropClick(event: Event): void {
    if (event.target === event.currentTarget) {
      this.closeModal();
    }
  }

  onKeyDown(event: KeyboardEvent): void {
    if (event.key === 'Escape') {
      if (this.showDeleteConfirm) {
        this.showDeleteConfirm = false;
      } else {
        this.closeModal();
      }
    }
  }

  // Tag Management
  addTag(): void {
    const tag = this.newTag.trim().toLowerCase();
    if (tag && !this.formData.tags.includes(tag)) {
      this.formData.tags.push(tag);
      this.newTag = '';
    }
  }

  removeTag(index: number): void {
    this.formData.tags.splice(index, 1);
  }

  onTagKeyDown(event: KeyboardEvent): void {
    if (event.key === 'Enter') {
      event.preventDefault();
      this.addTag();
    }
  }

  // Delete Functionality
  onDeleteClick(): void {
    this.showDeleteConfirm = true;
  }

  onConfirmDelete(): void {
    if (this.task) {
      this.taskDeleted.emit(this.task.id);
      this.closeModal();
    }
  }

  onCancelDelete(): void {
    this.showDeleteConfirm = false;
  }

  // Form Field Helpers
  getFieldError(fieldName: string): string | null {
    return this.errors[fieldName] || null;
  }

  hasFieldError(fieldName: string): boolean {
    return !!this.errors[fieldName];
  }

  clearFieldError(fieldName: string): void {
    delete this.errors[fieldName];
  }

  // Priority Options
  get priorityOptions() {
    return [
      { value: 'low', label: 'Low Priority', color: '#28a745' },
      { value: 'medium', label: 'Medium Priority', color: '#ffc107' },
      { value: 'high', label: 'High Priority', color: '#dc3545' },
      { value: 'urgent', label: 'Urgent Priority', color: '#e74c3c' }
    ];
  }

  trackByTag(index: number, tag: string): string {
    return tag;
  }
}
```
