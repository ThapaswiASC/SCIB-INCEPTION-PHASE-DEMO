#### 1. AppComponent (Page-level root container)

**TypeScript Specification:**

```typescript
import { Component, OnInit } from '@angular/core';
import { AuthService } from './core/services/auth.service';
import { UserService } from './core/services/user.service';
import { User } from './core/models/user.model';

@Component({
  selector: 'app-root',
  templateUrl: './app.component.html',
  styleUrls: ['./app.component.scss']
})
export class AppComponent implements OnInit {
  // Component State
  user: User | null = null;
  loading = true;
  error: string | null = null;
  sidebarCollapsed = false;
  activeView = 'kanban';

  constructor(
    private authService: AuthService,
    private userService: UserService
  ) {}

  ngOnInit(): void {
    this.initializeApp();
  }

  // Lifecycle Methods
  private async initializeApp(): Promise<void> {
    try {
      this.loading = true;
      await this.loadUserProfile();
      this.error = null;
    } catch (error) {
      this.error = 'Failed to initialize application';
      console.error('App initialization error:', error);
    } finally {
      this.loading = false;
    }
  }

  private async loadUserProfile(): Promise<void> {
    this.user = await this.userService.getCurrentUser();
  }

  // Event Handlers
  onSidebarToggle(collapsed: boolean): void {
    this.sidebarCollapsed = collapsed;
  }

  onViewChange(view: string): void {
    this.activeView = view;
  }

  onUserLogout(): void {
    this.authService.logout();
    this.user = null;
  }
}
```

```typescript
@Component({
  selector: 'app-sidebar-navigation',

})
export class SidebarNavigationComponent {
  // Input Properties
  @Input() activeView: string = 'kanban';
  @Input() collapsed: boolean = false;

  // Output Events
  @Output() viewChanged = new EventEmitter<string>();
  @Output() menuToggle = new EventEmitter<boolean>();

  // Component State
  menuExpanded = false;
  navigationItems = [
    { id: 'kanban', label: 'Kanban Board', icon: 'view_column' },

    { id: 'create-task', label: 'Create Task', icon: 'add' },
    { id: 'create-project', label: 'Create Project', icon: 'folder_open' },
    { id: 'invite-user', label: 'Invite User', icon: 'person_add' }
  ];

  // Event Handlers
  onViewSelect(viewId: string): void {
    if (viewId !== this.activeView) {
      this.viewChanged.emit(viewId);
    }
  }

  onMenuToggle(): void {
    this.collapsed = !this.collapsed;
    this.menuToggle.emit(this.collapsed);
  }

  onQuickAction(actionId: string): void {
    // Handle quick actions
    console.log('Quick action:', actionId);
  }
```

#### 3. HeaderComponent (Layout header with search)

**TypeScript Specification:**

```typescript
import { Component, Input, Output, EventEmitter } from '@angular/core';
import { User } from '../../core/models/user.model';

@Component({
  selector: 'app-header',
  templateUrl: './header.component.html',
  styleUrls: ['./header.component.scss']
})
export class HeaderComponent {
  // Input Properties
  @Input() user: User | null = null;
  @Input() searchQuery: string = '';

  // Output Events
  @Output() searchChanged = new EventEmitter<string>();
  @Output() userMenuToggle = new EventEmitter<boolean>();
  @Output() userLogout = new EventEmitter<void>();

  // Component State
  searchFocused = false;
  userMenuOpen = false;
  notifications = [
    { id: 1, message: 'New task assigned', time: '2 min ago', read: false },
    { id: 2, message: 'Project deadline approaching', time: '1 hour ago', read: false },
    { id: 3, message: 'Team meeting scheduled', time: '3 hours ago', read: true }
  ];

  // Event Handlers
  onSearchInput(event: Event): void {
    const target = event.target as HTMLInputElement;
    this.searchChanged.emit(target.value);
  }

  onSearchFocus(): void {
    this.searchFocused = true;
  }

  onSearchBlur(): void {
    this.searchFocused = false;
  }

  onUserMenuToggle(): void {
    this.userMenuOpen = !this.userMenuOpen;
    this.userMenuToggle.emit(this.userMenuOpen);
  }

  onLogout(): void {
    this.userMenuOpen = false;
    this.userLogout.emit();
  }

  onNotificationClick(notificationId: number): void {
    const notification = this.notifications.find(n => n.id === notificationId);
    if (notification) {
      notification.read = true;
    }
  }

  get unreadNotificationCount(): number {
    return this.notifications.filter(n => !n.read).length;
  }
}
```

```typescript
import { Component, OnInit, Input, Output, EventEmitter } from '@angular/core';
import { Task } from '../../core/models/task.model';
import { KanbanColumn } from '../../core/models/kanban.model';
import { TaskService } from '../../core/services/task.service';

@Component({
  selector: 'app-kanban-board',
  templateUrl: './kanban-board.component.html',
  styleUrls: ['./kanban-board.component.scss']
})
export class KanbanBoardComponent implements OnInit {
  // Input Properties
  @Input() tasks: Task[] = [];
  @Input() loading: boolean = false;
  @Input() error: string | null = null;

  // Output Events
  @Output() taskCreated = new EventEmitter<Task>();
  @Output() taskUpdated = new EventEmitter<Task>();
  @Output() taskDeleted = new EventEmitter<string>();

  // Component State
  selectedTask: Task | null = null;
  modalOpen = false;
  draggedTask: Task | null = null;
  columns: KanbanColumn[] = [
    { id: 'todo', title: 'To Do', tasks: [], color: '#6c757d' },
    { id: 'inprogress', title: 'In Progress', tasks: [], color: '#ffc107' },
    { id: 'done', title: 'Done', tasks: [], color: '#28a745' }
  ];

  constructor(private taskService: TaskService) {}

  ngOnInit(): void {
    this.initializeColumns();
    this.loadTasks();
  }

  // Lifecycle Methods
  private initializeColumns(): void {
    this.columns = [
      { id: 'todo', title: 'To Do', tasks: [], color: '#6c757d' },
      { id: 'inprogress', title: 'In Progress', tasks: [], color: '#ffc107' },
      { id: 'done', title: 'Done', tasks: [], color: '#28a745' }
    ];
  }

  private async loadTasks(): Promise<void> {
    try {
      this.loading = true;
      const tasks = await this.taskService.getAllTasks();
      this.distributeTasks(tasks);
      this.error = null;
    } catch (error) {
      this.error = 'Failed to load tasks';
      console.error('Task loading error:', error);
    } finally {
      this.loading = false;
    }
  }

  private distributeTasks(tasks: Task[]): void {
    // Clear existing tasks
    this.columns.forEach(column => column.tasks = []);

    // Distribute tasks to columns
    tasks.forEach(task => {
      const column = this.columns.find(col => col.id === task.status);
      if (column) {
        column.tasks.push(task);
      }
    });
  }

  // Event Handlers
  onCreateTask(): void {
    this.selectedTask = null;
    this.modalOpen = true;
  }

  onTaskClick(task: Task): void {
    this.selectedTask = task;
    this.modalOpen = true;
  }

  onTaskDropped(event: { task: Task; columnId: string }): void {
    const { task, columnId } = event;
    this.moveTask(task, columnId);
  }

  onTaskDragStart(task: Task): void {
    this.draggedTask = task;
  }

  onTaskDragEnd(): void {
    this.draggedTask = null;
  }

  onModalClosed(): void {
    this.modalOpen = false;
    this.selectedTask = null;
  }

  onTaskSaved(task: Task): void {
    if (this.selectedTask) {
      this.updateTask(task);
    } else {
      this.createTask(task);
    }
    this.onModalClosed();
  }

  // Business Logic
  private async createTask(task: Task): Promise<void> {
    try {
      const createdTask = await this.taskService.createTask(task);
      this.taskCreated.emit(createdTask);
      await this.loadTasks(); // Refresh board
    } catch (error) {
      this.error = 'Failed to create task';
      console.error('Task creation error:', error);
    }
  }

  private async updateTask(task: Task): Promise<void> {
    try {
      const updatedTask = await this.taskService.updateTask(task.id, task);
      this.taskUpdated.emit(updatedTask);
      await this.loadTasks(); // Refresh board
    } catch (error) {
      this.error = 'Failed to update task';
      console.error('Task update error:', error);
    }
  }

  private async moveTask(task: Task, newColumnId: string): Promise<void> {
    try {
      const updatedTask = { ...task, status: newColumnId };
      await this.taskService.updateTask(task.id, updatedTask);
      await this.loadTasks(); // Refresh board
    } catch (error) {
      this.error = 'Failed to move task';
      console.error('Task move error:', error);
      // Revert UI change
      await this.loadTasks();
    }
  }

  async deleteTask(taskId: string): Promise<void> {
    try {
      await this.taskService.deleteTask(taskId);
      this.taskDeleted.emit(taskId);
      await this.loadTasks(); // Refresh board
    } catch (error) {
      this.error = 'Failed to delete task';
      console.error('Task deletion error:', error);
    }
  }

  // Utility Methods
  getTasksByColumn(columnId: string): Task[] {
    const column = this.columns.find(col => col.id === columnId);
    return column ? column.tasks : [];
  }

  getColumnTaskCount(columnId: string): number {
    return this.getTasksByColumn(columnId).length;
  }
}
```

```typescript
import { Component, Input, Output, EventEmitter } from '@angular/core';
import { Task } from '../../core/models/task.model';
import { KanbanColumn } from '../../core/models/kanban.model';

@Component({
  selector: 'app-kanban-column',
  templateUrl: './kanban-column.component.html',
  styleUrls: ['./kanban-column.component.scss']
})
export class KanbanColumnComponent {
  // Input Properties
  @Input() column!: KanbanColumn;
  @Input() tasks: Task[] = [];
  @Input() allowDrop: boolean = false;
  @Input() draggedTask: Task | null = null;

  // Output Events
  @Output() taskDropped = new EventEmitter<{ task: Task; columnId: string }>();
  @Output() taskSelected = new EventEmitter<Task>();
  @Output() taskDragStart = new EventEmitter<Task>();
  @Output() taskDragEnd = new EventEmitter<void>();
  @Output() taskDeleted = new EventEmitter<string>();

  // Component State
  dragOver = false;
  expanded = true;

  // Event Handlers
  onTaskClick(task: Task): void {
    this.taskSelected.emit(task);
  }

  onTaskDragStart(task: Task): void {
    this.taskDragStart.emit(task);
  }

  onTaskDragEnd(): void {
    this.taskDragEnd.emit();
  }

  onTaskDelete(taskId: string): void {
    this.taskDeleted.emit(taskId);
  }

  // Drag and Drop Handlers
  onDragOver(event: DragEvent): void {
    if (this.allowDrop) {
      event.preventDefault();
      this.dragOver = true;
    }
  }

  onDragLeave(event: DragEvent): void {
    // Only set dragOver to false if we're leaving the column entirely
    const rect = (event.currentTarget as HTMLElement).getBoundingClientRect();
    const x = event.clientX;
    const y = event.clientY;
    
    if (x < rect.left || x > rect.right || y < rect.top || y > rect.bottom) {
      this.dragOver = false;
    }
  }

  onDrop(event: DragEvent): void {
    event.preventDefault();
    this.dragOver = false;
    
    if (this.draggedTask && this.draggedTask.status !== this.column.id) {
      this.taskDropped.emit({
        task: this.draggedTask,
        columnId: this.column.id
      });
    }
  }

  onToggleExpanded(): void {
    this.expanded = !this.expanded;
  }

  // Utility Methods
  getTaskCount(): number {
    return this.tasks.length;
  }

  getColumnColor(): string {
    return this.column.color || '#6c757d';
  }

  trackByTaskId(index: number, task: Task): string {
    return task.id;
  }
}
```

```typescript
import { Component, Input, Output, EventEmitter } from '@angular/core';
import { Task } from '../../core/models/task.model';

@Component({
  selector: 'app-task-card',
  templateUrl: './task-card.component.html',
  styleUrls: ['./task-card.component.scss']
})
export class TaskCardComponent {
  // Input Properties
  @Input() task!: Task;
  @Input() draggable: boolean = true;
  @Input() selected: boolean = false;

  // Output Events
  @Output() taskClick = new EventEmitter<Task>();
  @Output() taskDragStart = new EventEmitter<Task>();
  @Output() taskDragEnd = new EventEmitter<void>();
  @Output() taskDeleted = new EventEmitter<string>();

  // Component State
  hover = false;
  dragging = false;
  menuOpen = false;

  // Event Handlers
  onCardClick(event: Event): void {
    // Prevent click when clicking on action buttons
    if ((event.target as HTMLElement).closest('.task-actions')) {
      return;
    }
    this.taskClick.emit(this.task);
  }

  onDragStart(event: DragEvent): void {
    if (!this.draggable) return;
    
    this.dragging = true;
    this.taskDragStart.emit(this.task);
    
    // Set drag data
    if (event.dataTransfer) {
      event.dataTransfer.effectAllowed = 'move';
      event.dataTransfer.setData('text/plain', this.task.id);
    }
  }

  onDragEnd(): void {
    this.dragging = false;
    this.taskDragEnd.emit();
  }

  onMouseEnter(): void {
    this.hover = true;
  }

  onMouseLeave(): void {
    this.hover = false;
    this.menuOpen = false;
  }

  onMenuToggle(event: Event): void {
    event.stopPropagation();
    this.menuOpen = !this.menuOpen;
  }

  onEditTask(event: Event): void {
    event.stopPropagation();
    this.menuOpen = false;
    this.taskClick.emit(this.task);
  }

  onDeleteTask(event: Event): void {
    event.stopPropagation();
    this.menuOpen = false;
    if (confirm('Are you sure you want to delete this task?')) {
      this.taskDeleted.emit(this.task.id);
    }
  }

  // Utility Methods
  getPriorityColor(): string {
    switch (this.task.priority) {
      case 'high': return '#dc3545';
      case 'medium': return '#ffc107';
      case 'low': return '#28a745';
      default: return '#6c757d';
    }
  }

  getPriorityLabel(): string {
    return this.task.priority ? this.task.priority.charAt(0).toUpperCase() + this.task.priority.slice(1) : 'Normal';
  }

  getInitials(name: string): string {
    return name
      .split(' ')
      .map(word => word.charAt(0))
      .join('')
      .toUpperCase()
      .slice(0, 2);
  }

  isOverdue(): boolean {
    if (!this.task.dueDate) return false;
    return new Date(this.task.dueDate) < new Date() && this.task.status !== 'done';
  }

  getDueDateLabel(): string {
    if (!this.task.dueDate) return '';
    
    const dueDate = new Date(this.task.dueDate);
    const today = new Date();
    const diffTime = dueDate.getTime() - today.getTime();
    const diffDays = Math.ceil(diffTime / (1000 * 60 * 60 * 24));
    
    if (diffDays === 0) return 'Due today';
    if (diffDays === 1) return 'Due tomorrow';
    if (diffDays === -1) return 'Due yesterday';
    if (diffDays < 0) return `Overdue by ${Math.abs(diffDays)} days`;
    if (diffDays <= 7) return `Due in ${diffDays} days`;
    
    return dueDate.toLocaleDateString();
  }
}
```

#### 7. TaskModalComponent (Feature task creation/editing)

**TypeScript Specification:**

```typescript
import { Component, Input, Output, EventEmitter, OnInit, OnChanges } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { Task } from '../../core/models/task.model';
import { User } from '../../core/models/user.model';
import { UserService } from '../../core/services/user.service';

@Component({
  selector: 'app-task-modal',
  templateUrl: './task-modal.component.html',
  styleUrls: ['./task-modal.component.scss']
})
export class TaskModalComponent implements OnInit, OnChanges {
  // Input Properties
  @Input() task: Task | null = null;
  @Input() visible: boolean = false;
  @Input() mode: 'create' | 'edit' = 'create';

  // Output Events
  @Output() taskSaved = new EventEmitter<Task>();
  @Output() modalClosed = new EventEmitter<void>();

  // Component State
  taskForm!: FormGroup;
  submitting = false;
  validationErrors: string[] = [];
  availableUsers: User[] = [];
  availableTags: string[] = ['Frontend', 'Backend', 'Design', 'Testing', 'Bug', 'Feature', 'Enhancement'];
  
  priorityOptions = [
    { value: 'low', label: 'Low', color: '#28a745' },
    { value: 'medium', label: 'Medium', color: '#ffc107' },
    { value: 'high', label: 'High', color: '#dc3545' }
  ];

  statusOptions = [
    { value: 'todo', label: 'To Do' },
    { value: 'inprogress', label: 'In Progress' },
    { value: 'done', label: 'Done' }
  ];

  constructor(
    private formBuilder: FormBuilder,
    private userService: UserService
  ) {
    this.initializeForm();
  }

  ngOnInit(): void {
    this.loadAvailableUsers();
  }

  ngOnChanges(): void {
    if (this.visible) {
      this.resetForm();
      if (this.task && this.mode === 'edit') {
        this.populateForm();
      }
    }
  }

  // Form Initialization
  private initializeForm(): void {
    this.taskForm = this.formBuilder.group({
      title: ['', [Validators.required, Validators.minLength(3), Validators.maxLength(100)]],
      description: ['', [Validators.maxLength(500)]],
      priority: ['medium', [Validators.required]],
      status: ['todo', [Validators.required]],
      assigneeId: [''],
      dueDate: [''],
      tags: [[]]
    });
  }

  private resetForm(): void {
    this.taskForm.reset({
      title: '',
      description: '',
      priority: 'medium',
      status: 'todo',
      assigneeId: '',
      dueDate: '',
      tags: []
    });
    this.validationErrors = [];
  }

  private populateForm(): void {
    if (!this.task) return;

    this.taskForm.patchValue({
      title: this.task.title,
      description: this.task.description || '',
      priority: this.task.priority || 'medium',
      status: this.task.status,
      assigneeId: this.task.assignee?.id || '',
      dueDate: this.task.dueDate ? this.formatDateForInput(this.task.dueDate) : '',
      tags: this.task.tags || []
    });
  }

  // Data Loading
  private async loadAvailableUsers(): Promise<void> {
    try {
      this.availableUsers = await this.userService.getAllUsers();
    } catch (error) {
      console.error('Failed to load users:', error);
    }
  }

  // Event Handlers
  onSubmit(): void {
    if (this.taskForm.valid && !this.submitting) {
      this.saveTask();
    } else {
      this.validateForm();
    }
  }

  onCancel(): void {
    this.modalClosed.emit();
  }

  onBackdropClick(event: Event): void {
    if (event.target === event.currentTarget) {
      this.onCancel();
    }
  }

  onTagAdd(tag: string): void {
    const currentTags = this.taskForm.get('tags')?.value || [];
    if (!currentTags.includes(tag)) {
      this.taskForm.patchValue({
        tags: [...currentTags, tag]
      });
    }
  }

  onTagRemove(tagToRemove: string): void {
    const currentTags = this.taskForm.get('tags')?.value || [];
    this.taskForm.patchValue({
      tags: currentTags.filter((tag: string) => tag !== tagToRemove)
    });
  }

  // Business Logic
  private async saveTask(): Promise<void> {
    try {
      this.submitting = true;
      this.validationErrors = [];

      const formValue = this.taskForm.value;
      const assignee = formValue.assigneeId ? 
        this.availableUsers.find(user => user.id === formValue.assigneeId) : null;

      const taskData: Task = {
        id: this.task?.id || this.generateTaskId(),
        title: formValue.title,
        description: formValue.description,
        priority: formValue.priority,
        status: formValue.status,
        assignee: assignee || null,
        dueDate: formValue.dueDate || null,
        tags: formValue.tags || [],
        createdAt: this.task?.createdAt || new Date().toISOString(),
        updatedAt: new Date().toISOString()
      };

      this.taskSaved.emit(taskData);
    } catch (error) {
      this.validationErrors = ['Failed to save task. Please try again.'];
      console.error('Task save error:', error);
    } finally {
      this.submitting = false;
    }
  }

  private validateForm(): void {
    this.validationErrors = [];
    
    Object.keys(this.taskForm.controls).forEach(key => {
      const control = this.taskForm.get(key);
      if (control && control.invalid && control.touched) {
        switch (key) {
          case 'title':
            if (control.errors?.['required']) {
              this.validationErrors.push('Task title is required');
            } else if (control.errors?.['minlength']) {
              this.validationErrors.push('Task title must be at least 3 characters');
            } else if (control.errors?.['maxlength']) {
              this.validationErrors.push('Task title cannot exceed 100 characters');
            }
            break;
          case 'description':
            if (control.errors?.['maxlength']) {
              this.validationErrors.push('Description cannot exceed 500 characters');
            }
            break;
        }
      }
    });
  }

  // Utility Methods
  private generateTaskId(): string {
    return 'task_' + Date.now() + '_' + Math.random().toString(36).substr(2, 9);
  }

  private formatDateForInput(dateString: string): string {
    const date = new Date(dateString);
    return date.toISOString().split('T')[0];
  }

  getModalTitle(): string {
    return this.mode === 'create' ? 'Create New Task' : 'Edit Task';
  }

  getSubmitButtonText(): string {
    if (this.submitting) {
      return this.mode === 'create' ? 'Creating...' : 'Updating...';
    }
    return this.mode === 'create' ? 'Create Task' : 'Update Task';
  }

  isFieldInvalid(fieldName: string): boolean {
    const field = this.taskForm.get(fieldName);
    return !!(field && field.invalid && field.touched);
  }

  getFieldError(fieldName: string): string {
    const field = this.taskForm.get(fieldName);
    if (field && field.errors && field.touched) {
      if (field.errors['required']) return `${fieldName} is required`;
      if (field.errors['minlength']) return `${fieldName} is too short`;
      if (field.errors['maxlength']) return `${fieldName} is too long`;
    }
    return '';
  }
}
```
