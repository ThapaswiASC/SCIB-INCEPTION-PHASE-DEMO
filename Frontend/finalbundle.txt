@Component({
  selector: 'app-task-creation',
  templateUrl: './task-creation.component.html',
  styleUrls: ['./task-creation.component.scss']
})
export class TaskCreationComponent implements OnInit {
  @Input() isOpen: boolean = false;
  @Input() onClose: () => void;
  @Input() onTaskCreated: (task: Task) => void;
  
  @Output() taskCreated = new EventEmitter<Task>();
  @Output() formClosed = new EventEmitter<void>();
  
  formData: TaskFormData = {};
  isLoading: boolean = false;
  validationErrors: ValidationError[] = [];
  isSubmitting: boolean = false;
  
  constructor(
    private taskService: TaskService,
    private fb: FormBuilder
  ) {}
  
  ngOnInit() {
    this.initializeForm();
  }
  
  openModal() {
    this.isOpen = true;
    this.resetForm();
  }
  
  closeModal() {
    this.isOpen = false;
    this.resetForm();
    this.formClosed.emit();
  }
  
  handleFormSubmit(data: TaskFormData) {
    this.isLoading = true;
    this.isSubmitting = true;
    
    this.taskService.createTask(data).subscribe({
      next: (task: Task) => {
        this.taskCreated.emit(task);
        this.closeModal();
        this.isLoading = false;
        this.isSubmitting = false;
      },
      error: (error: ApiError) => {
        this.handleApiError(error);
        this.isLoading = false;
        this.isSubmitting = false;
      }
    });
  }
  
  handleApiError(error: ApiError) {
    this.validationErrors = error.validationErrors || [];
    // Display error snackbar
  }
  
  private initializeForm() {
    this.formData = {
      title: '',
      description: '',
      priority: null,
      dueDate: null
    };
  }
  
  private resetForm() {
    this.initializeForm();
    this.validationErrors = [];
  }
}

TypeScript Specification:
```typescript
@Component({
  selector: 'app-task-form',
  templateUrl: './task-form.component.html',
  styleUrls: ['./task-form.component.scss']
})
export class TaskFormComponent implements OnInit {
  @Input() initialData: Partial<TaskFormData> = {};
  @Input() isLoading: boolean = false;
  @Input() onSubmit: (data: TaskFormData) => void;
  @Input() onCancel: () => void;
  
  @Output() formSubmit = new EventEmitter<TaskFormData>();
  @Output() formCancel = new EventEmitter<void>();
  @Output() formChange = new EventEmitter<TaskFormData>();
  
  taskForm: FormGroup;
  isDirty: boolean = false;
  isValid: boolean = false;
  touchedFields: Set<string> = new Set();
  
  priorityOptions: PriorityOption[] = [
    { value: TaskPriority.LOW, label: 'Low', color: '#28a745' },
    { value: TaskPriority.MEDIUM, label: 'Medium', color: '#ffc107' },
    { value: TaskPriority.HIGH, label: 'High', color: '#dc3545' },
    { value: TaskPriority.URGENT, label: 'Urgent', color: '#6f42c1' }
  ];
  
  today: Date = new Date();
  
  constructor(
    private fb: FormBuilder,
    private customValidators: CustomValidators
  ) {}
  
  ngOnInit() {
    this.initializeForm();
    this.setupFormSubscriptions();
  }
  
  initializeForm() {
    this.taskForm = this.fb.group({
      title: [
        this.initialData.title || '',
        [Validators.required, Validators.maxLength(255)]
      ],
      description: [
        this.initialData.description || '',
        [Validators.maxLength(2000)]
      ],
      priority: [
        this.initialData.priority || null,
        [Validators.required]
      ],
      dueDate: [
        this.initialData.dueDate || null,
        [Validators.required, this.customValidators.futureDate()]
      ]
    });
  }
  
  setupFormSubscriptions() {
    this.taskForm.valueChanges.subscribe(value => {
      this.isDirty = this.taskForm.dirty;
      this.isValid = this.taskForm.valid;
      this.formChange.emit(value);
    });
  }
  
  onFormSubmit() {
    if (this.taskForm.valid) {
      const formData: TaskFormData = this.taskForm.value;
      this.formSubmit.emit(formData);
    } else {
      this.markAllFieldsAsTouched();
    }
  }
  
  onFormCancel() {
    this.formCancel.emit();
  }
  
  handleFieldChange(field: string, value: any) {
    this.touchedFields.add(field);
    this.taskForm.patchValue({ [field]: value });
  }
  
  validateForm(): boolean {
    return this.taskForm.valid;
  }
  
  resetForm() {
    this.taskForm.reset();
    this.touchedFields.clear();
    this.isDirty = false;
    this.isValid = false;
  }
  
  private markAllFieldsAsTouched() {
    Object.keys(this.taskForm.controls).forEach(key => {
      this.taskForm.get(key)?.markAsTouched();
      this.touchedFields.add(key);
    });
  }
}
```

@Component({
  selector: 'app-title-input',
  templateUrl: './title-input.component.html',
  styleUrls: ['./title-input.component.scss']
})
export class TitleInputComponent implements OnInit, ControlValueAccessor {
  @Input() maxLength: number = 255;
  @Input() required: boolean = false;
  @Input() placeholder: string = '';
  @Input() disabled: boolean = false;
  
  @Output() valueChange = new EventEmitter<string>();
  @Output() validationChange = new EventEmitter<ValidationState>();
  
  value: string = '';
  characterCount: number = 0;
  validationState: 'default' | 'error' | 'success' = 'default';
  errorMessage: string = '';
  
  private onChange = (value: string) => {};
  private onTouched = () => {};
  
  constructor() {}
  
  ngOnInit() {
    this.updateCharacterCount(this.value);
  }
  
  onInputChange(value: string) {
    this.value = value;
    this.updateCharacterCount(value);
    this.validateInput(value);
    this.valueChange.emit(value);
    this.onChange(value);
  }
  
  validateInput(value: string) {
    if (this.required && !value.trim()) {
      this.validationState = 'error';
      this.errorMessage = 'Title is required';
    } else if (value.length > this.maxLength) {
      this.validationState = 'error';
      this.errorMessage = `Title cannot exceed ${this.maxLength} characters`;
    } else {
      this.validationState = value.trim() ? 'success' : 'default';
      this.errorMessage = '';
    }
    
    this.validationChange.emit(this.validationState);
  }
  
  updateCharacterCount(value: string) {
    this.characterCount = value.length;
  }
  
  // ControlValueAccessor implementation
  writeValue(value: string): void {
    this.value = value || '';
    this.updateCharacterCount(this.value);
  }
  
  registerOnChange(fn: (value: string) => void): void {
    this.onChange = fn;
  }
  
  registerOnTouched(fn: () => void): void {
    this.onTouched = fn;
  }
  
  setDisabledState(isDisabled: boolean): void {
    this.disabled = isDisabled;
  }
}

    this.value = value || '';
    this.updateCharacterCount(this.value);
  }

    this.onChange = fn;

  registerOnTouched(fn: () => void): void {

@Component({
  selector: 'app-priority-select',
  templateUrl: './priority-select.component.html',
  styleUrls: ['./priority-select.component.scss']
})
export class PrioritySelectComponent implements OnInit, ControlValueAccessor {
  @Input() options: PriorityOption[] = [];
  @Input() required: boolean = false;
  @Input() disabled: boolean = false;
  
  @Output() priorityChange = new EventEmitter<TaskPriority>();
  @Output() validationChange = new EventEmitter<ValidationState>();
  
  selectedPriority: TaskPriority | null = null;
  validationState: 'default' | 'error' | 'success' = 'default';
  errorMessage: string = '';
  
  private onChange = (value: TaskPriority) => {};
  private onTouched = () => {};
  
  constructor() {}
  
  ngOnInit() {
    this.validateSelection();
  }
  
  onPrioritySelect(priority: TaskPriority) {
    this.selectedPriority = priority;
    this.validateSelection();
    this.priorityChange.emit(priority);
    this.onChange(priority);
  }
  
  validateSelection() {
    if (this.required && !this.selectedPriority) {
      this.validationState = 'error';
      this.errorMessage = 'Priority is required';
    } else {
      this.validationState = this.selectedPriority ? 'success' : 'default';
      this.errorMessage = '';
    }
    
    this.validationChange.emit(this.validationState);
  }
  
  getPriorityColor(priority: TaskPriority): string {
    const option = this.options.find(opt => opt.value === priority);
    return option?.color || '#6c757d';
  }
  
  // ControlValueAccessor implementation
  writeValue(value: TaskPriority): void {
    this.selectedPriority = value;
    this.validateSelection();
  }
  
  registerOnChange(fn: (value: TaskPriority) => void): void {
    this.onChange = fn;
  }
  
  registerOnTouched(fn: () => void): void {
    this.onTouched = fn;
  }
  
  setDisabledState(isDisabled: boolean): void {
    this.disabled = isDisabled;
  }
}

@Component({
  selector: 'app-due-date-picker',
  templateUrl: './due-date-picker.component.html',
  styleUrls: ['./due-date-picker.component.scss']
})
export class DueDatePickerComponent implements OnInit, ControlValueAccessor {
  @Input() minDate: Date = new Date();
  @Input() required: boolean = false;
  @Input() disabled: boolean = false;
  @Input() placeholder: string = 'Select due date';
  
  @Output() dateChange = new EventEmitter<Date>();
  @Output() validationChange = new EventEmitter<ValidationState>();
  
  selectedDate: Date | null = null;
  validationState: 'default' | 'error' | 'success' = 'default';
  errorMessage: string = '';
  isCalendarOpen: boolean = false;
  
  private onChange = (value: Date) => {};
  private onTouched = () => {};
  
  constructor() {}
  
  ngOnInit() {
    this.validateDate(this.selectedDate);
  }
  
  onDateSelect(date: Date) {
    this.selectedDate = date;
    this.validateDate(date);
    this.dateChange.emit(date);
    this.onChange(date);
    this.closeCalendar();
  }
  
  validateDate(date: Date | null) {
    if (this.required && !date) {
      this.validationState = 'error';
      this.errorMessage = 'Due date is required';
    } else if (date && date < this.minDate) {
      this.validationState = 'error';
      this.errorMessage = 'Due date must be in the future';
    } else if (date && isNaN(date.getTime())) {
      this.validationState = 'error';
      this.errorMessage = 'Please select a valid date';
    } else {
      this.validationState = date ? 'success' : 'default';
      this.errorMessage = '';
    }
    
    this.validationChange.emit(this.validationState);
  }
  
  formatDisplayDate(date: Date | null): string {
    if (!date) return '';
    return date.toLocaleDateString('en-US', {
      year: 'numeric',
      month: 'short',
      day: 'numeric'
    });
  }
  
  openCalendar() {
    if (!this.disabled) {
      this.isCalendarOpen = true;
    }
  }
  
  closeCalendar() {
    this.isCalendarOpen = false;
    this.onTouched();
  }
  
  // ControlValueAccessor implementation
  writeValue(value: Date): void {
    this.selectedDate = value;
    this.validateDate(value);
  }
  
  registerOnChange(fn: (value: Date) => void): void {
    this.onChange = fn;
  }
  
  registerOnTouched(fn: () => void): void {
    this.onTouched = fn;
  }
  
  setDisabledState(isDisabled: boolean): void {
    this.disabled = isDisabled;
  }
}

  @Input() isValid: boolean = false;
  @Input() isLoading: boolean = false;
  @Input() submitText: string = 'Create Task';
  @Input() cancelText: string = 'Cancel';
  @Input() disabled: boolean = false;
  
  @Output() submit = new EventEmitter<void>();
  @Output() cancel = new EventEmitter<void>();
  
  get isSubmitDisabled(): boolean {
    return !this.isValid || this.isLoading || this.disabled;
  }
  
  onSubmitClick() {
    if (!this.isSubmitDisabled) {
      this.submit.emit();
    }
  }
  
  onCancelClick() {
    if (!this.isLoading) {
      this.cancel.emit();
    }
  }
  
  getSubmitButtonState(): string {
    if (this.isLoading) return 'loading';
    if (!this.isValid) return 'disabled';
    return 'enabled';
  }
}
