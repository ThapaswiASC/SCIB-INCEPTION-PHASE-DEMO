```typescript
@Component({
  selector: 'app-kanban-board',
  templateUrl: './kanban-board.component.html',
  styleUrls: ['./kanban-board.component.scss']
})
export class KanbanBoardComponent implements OnInit {
  @Input() youthAccountId: string;
  @Input() initialColumns: KanbanColumn[];
  
  @Output() taskMoved = new EventEmitter<TaskMoveEvent>();
  @Output() taskCreated = new EventEmitter<Task>();
  @Output() errorOccurred = new EventEmitter<string>();
  
  columns: KanbanColumn[] = [
    {id: 'todo', title: 'To Do', cards: []}, 
    {id: 'inprogress', title: 'In Progress', cards: []}, 
    {id: 'done', title: 'Done', cards: []}
  ];
  loading: boolean = false;
  error: string | null = null;
  draggedTask: Task | null = null;
  
  constructor(
    private kanbanService: KanbanService,
    private taskService: TaskService
  ) {}
  
  ngOnInit() {
    this.loadKanbanData();
  }
  
  loadKanbanData() {
    this.loading = true;
    this.kanbanService.getDashboardData(this.youthAccountId)
      .subscribe({
        next: (data) => {
          this.columns = data.columns;
          this.loading = false;
        },
        error: (error) => {
          this.error = 'Failed to load kanban data';
          this.loading = false;
          this.errorOccurred.emit(error.message);
        }
      });
  }
  
  onTaskMoved(event: TaskMoveEvent) {
    // Update local state
    this.updateTaskPosition(event);
    
    // Call API to persist changes
    this.taskService.updateTaskStatus(event.taskId, event.toColumn)
      .subscribe({
        next: () => {
          this.taskMoved.emit(event);
        },
        error: (error) => {
          // Revert local changes
          this.revertTaskPosition(event);
          this.handleError(error);
        }
      });
  }
  
  onTaskCreated(task: Task) {
    this.taskService.createTask(task)
      .subscribe({
        next: (createdTask) => {
          this.addTaskToColumn(createdTask);
          this.taskCreated.emit(createdTask);
        },
        error: (error) => {
          this.handleError(error);
        }
      });
  }
  
  onAddTaskClicked(columnId: string) {
    this.selectedColumnId = columnId;
    this.isModalOpen = true;
  }
  
  handleError(error: any) {
    this.error = error.message || 'An unexpected error occurred';
    this.errorOccurred.emit(this.error);
  }
  
  private updateTaskPosition(event: TaskMoveEvent) {
    // Implementation for updating task position
  }
  
  private revertTaskPosition(event: TaskMoveEvent) {
    // Implementation for reverting task position
  }
  
  private addTaskToColumn(task: Task) {
    const column = this.columns.find(col => col.id === task.status);
    if (column) {
      column.cards.push(task);
    }
  }
}
```

```typescript
@Component({
  selector: 'app-kanban-column',
  templateUrl: './kanban-column.component.html',
  styleUrls: ['./kanban-column.component.scss']
})
export class KanbanColumnComponent implements OnInit {
  @Input() column: KanbanColumn;
  @Input() tasks: Task[];
  @Input() columnId: string;
  @Input() allowDrop: boolean = true;
  
  @Output() taskDropped = new EventEmitter<TaskDropEvent>();
  @Output() addTaskClicked = new EventEmitter<string>();
  @Output() taskSelected = new EventEmitter<Task>();
  
  isDropTarget: boolean = false;
  taskCount: number = 0;
  isCollapsed: boolean = false;
  
  ngOnInit() {
    this.updateTaskCount();
  }
  
  onDrop(event: CdkDragDrop<Task[]>) {
    if (event.previousContainer !== event.container) {
      const taskDropEvent: TaskDropEvent = {
        taskId: event.item.data.id,
        targetColumnId: this.columnId,
        position: event.currentIndex
      };
      this.taskDropped.emit(taskDropEvent);
    }
  }
  
  onAddTask() {
    this.addTaskClicked.emit(this.columnId);
  }
  
  toggleCollapse() {
    this.isCollapsed = !this.isCollapsed;
  }
  
  updateTaskCount() {
    this.taskCount = this.tasks ? this.tasks.length : 0;
  }
  
  onTaskSelected(task: Task) {
    this.taskSelected.emit(task);
  }
  
  trackByTaskId(index: number, task: Task): string {
    return task.id;
  }
}
```

```typescript
@Component({
  selector: 'app-task-card',
  templateUrl: './task-card.component.html',
  styleUrls: ['./task-card.component.scss']
})
export class TaskCardComponent implements OnInit {
  @Input() task: Task;
  @Input() draggable: boolean = true;
  @Input() selectable: boolean = true;
  @Input() compact: boolean = false;
  
  @Output() taskClicked = new EventEmitter<Task>();
  @Output() taskDragStart = new EventEmitter<Task>();
  @Output() taskDragEnd = new EventEmitter<Task>();
  @Output() assigneeClicked = new EventEmitter<{task: Task, assignee: User}>();
  
  isDragging: boolean = false;
  isSelected: boolean = false;
  isHovered: boolean = false;
  
  ngOnInit() {
    // Initialize task display
  }
  
  onCardClick() {
    if (this.selectable) {
      this.isSelected = !this.isSelected;
      this.taskClicked.emit(this.task);
    }
  }
  
  onDragStart() {
    this.isDragging = true;
    this.taskDragStart.emit(this.task);
  }
  
  onDragEnd() {
    this.isDragging = false;
    this.taskDragEnd.emit(this.task);
  }
  
  onAssigneeClick(assignee: User) {
    this.assigneeClicked.emit({task: this.task, assignee});
  }
  
  formatDueDate(date: Date): string {
    if (!date) return '';
    const now = new Date();
    const diffTime = date.getTime() - now.getTime();
    const diffDays = Math.ceil(diffTime / (1000 * 60 * 60 * 24));
    
    if (diffDays < 0) return 'Overdue';
    if (diffDays === 0) return 'Due today';
    if (diffDays === 1) return 'Due tomorrow';
    return `Due in ${diffDays} days`;
  }
  
  getPriorityClass(): string {
    return `priority-${this.task.priority || 'medium'}`;
  }
}
```

```typescript
@Component({
  selector: 'app-sidebar',

export class SidebarComponent implements OnInit {
  @Input() currentPage: string;
  @Input() navigationItems: NavItem[];
  @Input() projects: Project[];
  @Input() isCollapsed: boolean = false;
  
  @Output() navigationChanged = new EventEmitter<string>();
  @Output() sidebarToggled = new EventEmitter<boolean>();
  @Output() projectSelected = new EventEmitter<Project>();
  
  activeSection: string = 'dashboard';
  expandedSections: string[] = [];
  projectsLoading: boolean = false;
  
  constructor(
    private router: Router,
    private projectService: ProjectService,
    private navigationService: NavigationService
  ) {}
  
  ngOnInit() {
    this.loadProjects();
    this.expandedSections = ['projects']; // Default expanded
  }
  
  onNavigationClick(item: NavItem) {
    this.activeSection = item.path;
    this.router.navigate([item.path]);
    this.navigationChanged.emit(item.path);
  }
  
  toggleSidebar() {
    this.isCollapsed = !this.isCollapsed;
    this.sidebarToggled.emit(this.isCollapsed);
  }
  
  toggleSection(sectionId: string) {
    const index = this.expandedSections.indexOf(sectionId);
    if (index > -1) {
      this.expandedSections.splice(index, 1);
    } else {
      this.expandedSections.push(sectionId);
    }
  }
  
  onProjectClick(project: Project) {
    this.router.navigate(['/projects', project.id]);
    this.projectSelected.emit(project);
  }
  
  loadProjects() {
    this.projectsLoading = true;
    this.projectService.getUserProjects()
      .subscribe({
        next: (projects) => {
          this.projects = projects;
          this.projectsLoading = false;
        },
        error: (error) => {
          console.error('Failed to load projects:', error);
          this.projectsLoading = false;
        }
      });
  }
}
```

```typescript
@Component({
  selector: 'app-task-modal',
  templateUrl: './task-modal.component.html',
  styleUrls: ['./task-modal.component.scss']
})
export class TaskModalComponent implements OnInit, OnChanges {
  @Input() isOpen: boolean = false;
  @Input() task: Task | null = null;
  @Input() defaultStatus: string = 'todo';
  @Input() availableAssignees: User[] = [];
  
  @Output() taskSaved = new EventEmitter<Task>();
  @Output() modalClosed = new EventEmitter<void>();
  @Output() taskDeleted = new EventEmitter<string>();
  
  taskForm: FormGroup;
  isSubmitting: boolean = false;
  validationErrors: ValidationError[] = [];
  isEditMode: boolean = false;
  
  constructor(
    private fb: FormBuilder,
    private taskService: TaskService
  ) {
    this.buildForm();
  }
  
  ngOnInit() {
    this.buildForm();
  }
  
  ngOnChanges(changes: SimpleChanges) {
    if (changes['task'] && this.task) {
      this.isEditMode = true;
      this.populateForm();
    } else if (changes['isOpen'] && this.isOpen && !this.task) {
      this.isEditMode = false;
      this.resetForm();
    }
  }
  
  buildForm() {
    this.taskForm = this.fb.group({
      title: ['', [Validators.required, Validators.maxLength(100)]],
      description: ['', Validators.maxLength(500)],
      status: [this.defaultStatus, Validators.required],
      priority: ['medium', Validators.required],
      dueDate: [''],
      assignees: [[]],
      tags: [[]]
    });
  }
  
  populateForm() {
    if (this.task) {
      this.taskForm.patchValue({
        title: this.task.title,
        description: this.task.description,
        status: this.task.status,
        priority: this.task.priority,
        dueDate: this.task.dueDate,
        assignees: this.task.assignees || [],
        tags: this.task.tags || []
      });
    }
  }
  
  onSubmit() {
    if (this.taskForm.valid) {
      this.isSubmitting = true;
      const taskData = this.taskForm.value;
      
      if (this.isEditMode) {
        this.updateTask(taskData);
      } else {
        this.createTask(taskData);
      }
    } else {
      this.validateForm();
    }
  }
  
  createTask(taskData: any) {
    this.taskService.createTask(taskData)
      .subscribe({
        next: (createdTask) => {
          this.taskSaved.emit(createdTask);
          this.onCancel();
          this.isSubmitting = false;
        },
        error: (error) => {
          this.handleError(error);
          this.isSubmitting = false;
        }
      });
  }
  
  updateTask(taskData: any) {
    if (this.task) {
      this.taskService.updateTask(this.task.id, taskData)
        .subscribe({
          next: (updatedTask) => {
            this.taskSaved.emit(updatedTask);
            this.onCancel();
            this.isSubmitting = false;
          },
          error: (error) => {
            this.handleError(error);
            this.isSubmitting = false;
          }
        });
    }
  }
  
  onCancel() {
    this.resetForm();
    this.modalClosed.emit();
  }
  
  onDelete() {
    if (this.task && confirm('Are you sure you want to delete this task?')) {
      this.taskService.deleteTask(this.task.id)
        .subscribe({
          next: () => {
            this.taskDeleted.emit(this.task!.id);
            this.onCancel();
          },
          error: (error) => {
            this.handleError(error);
          }
        });
    }
  }
  
  validateForm() {
    this.validationErrors = [];
    Object.keys(this.taskForm.controls).forEach(key => {
      const control = this.taskForm.get(key);
      if (control && control.invalid && control.touched) {
        this.validationErrors.push({
          field: key,
          message: this.getErrorMessage(key, control.errors)
        });
      }
    });
  }
  
  getErrorMessage(field: string, errors: any): string {
    if (errors['required']) {
      return `${field} is required`;
    }
    if (errors['maxlength']) {
      return `${field} is too long`;
    }
    return `${field} is invalid`;
  }
  
  resetForm() {
    this.taskForm.reset({
      status: this.defaultStatus,
      priority: 'medium',
      assignees: [],
      tags: []
    });
    this.validationErrors = [];
  }
  
  handleError(error: any) {
    console.error('Task operation failed:', error);
    // Add user-friendly error handling
  }
}
```
