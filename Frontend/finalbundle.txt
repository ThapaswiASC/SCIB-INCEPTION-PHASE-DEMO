// src/app/features/kanban/components/kanban-board/kanban-board.component.ts
import { Component, OnInit, OnDestroy, ChangeDetectionStrategy, ChangeDetectorRef } from '@angular/core';
import { Observable, Subject, BehaviorSubject, combineLatest } from 'rxjs';
import { takeUntil, map, catchError, finalize } from 'rxjs/operators';
import { CdkDragDrop, moveItemInArray, transferArrayItem } from '@angular/cdk/drag-drop';

@Component({
  selector: 'app-kanban-board',
  templateUrl: './kanban-board.component.html',
  styleUrls: ['./kanban-board.component.scss'],
  changeDetection: ChangeDetectionStrategy.OnPush
})
export class KanbanBoardComponent implements OnInit, OnDestroy {
  // Input Properties
  @Input() boardId: string = 'default';
  @Input() readonly: boolean = false;
  @Input() showAddButtons: boolean = true;
  
  // Output Events
  @Output() taskCreated = new EventEmitter<Task>();
  @Output() taskUpdated = new EventEmitter<Task>();
  @Output() taskDeleted = new EventEmitter<string>();
  @Output() columnConfigChanged = new EventEmitter<ColumnConfig>();
  
  // State Management
  private destroy$ = new Subject<void>();
  private loadingSubject = new BehaviorSubject<boolean>(false);
  private errorSubject = new BehaviorSubject<string | null>(null);
  
  // Observable Properties
  loading$ = this.loadingSubject.asObservable();
  error$ = this.errorSubject.asObservable();
  
  // Component State
  columns: KanbanColumn[] = [
    { id: 'todo', title: 'To Do', cards: [], maxCards: null, color: '#e3f2fd' },
    { id: 'inprogress', title: 'In Progress', cards: [], maxCards: 3, color: '#fff3e0' },
    { id: 'done', title: 'Done', cards: [], maxCards: null, color: '#e8f5e8' }
  ];
  
  tasks: Task[] = [];
  selectedTask: Task | null = null;
  draggedTask: Task | null = null;
  showCreateModal: boolean = false;
  searchQuery: string = '';
  filterOptions: FilterOptions = {
    priority: null,
    assignee: null,
    dueDate: null
  };
  
  constructor(
    private taskService: TaskService,
    private kanbanService: KanbanService,
    private cdr: ChangeDetectorRef,
    private notificationService: NotificationService
  ) {}
  
  ngOnInit(): void {
    this.initializeBoard();
    this.setupRealTimeUpdates();
  }
  
  ngOnDestroy(): void {
    this.destroy$.next();
    this.destroy$.complete();
  }
  
  // Initialization Methods
  private initializeBoard(): void {
    this.loadingSubject.next(true);
    
    combineLatest([
      this.taskService.getTasks(this.boardId),
      this.kanbanService.getColumnConfig(this.boardId)
    ]).pipe(
      takeUntil(this.destroy$),
      catchError(error => {
        this.errorSubject.next('Failed to load board data');
        return [];
      }),
      finalize(() => this.loadingSubject.next(false))
    ).subscribe(([tasks, columnConfig]) => {
      this.tasks = tasks;
      this.updateColumnConfig(columnConfig);
      this.distributeTasksToColumns();
      this.cdr.markForCheck();
    });
  }
  
  private setupRealTimeUpdates(): void {
    this.kanbanService.getTaskUpdates(this.boardId)
      .pipe(takeUntil(this.destroy$))
      .subscribe(update => {
        this.handleRealTimeUpdate(update);
      });
  }
  
  // Task Management Methods
  onTaskDrop(event: CdkDragDrop<Task[]>): void {
    if (event.previousContainer === event.container) {
      moveItemInArray(event.container.data, event.previousIndex, event.currentIndex);
    } else {
      const task = event.previousContainer.data[event.previousIndex];
      const newColumnId = this.getColumnIdFromContainer(event.container);
      
      if (this.canMoveTask(task, newColumnId)) {
        transferArrayItem(
          event.previousContainer.data,
          event.container.data,
          event.previousIndex,
          event.currentIndex
        );
        
        this.updateTaskColumn(task, newColumnId);
      }
    }
  }
  
  onTaskCreate(columnId: string): void {
    this.showCreateModal = true;
    this.selectedColumnId = columnId;
  }
  
  onTaskSelect(task: Task): void {
    this.selectedTask = task;
    this.router.navigate(['/tasks', task.id]);
  }
  
  onCreateTaskSubmit(taskData: CreateTaskRequest): void {
    this.taskService.createTask({
      ...taskData,
      columnId: this.selectedColumnId,
      boardId: this.boardId
    }).pipe(
      takeUntil(this.destroy$),
      catchError(error => {
        this.notificationService.showError('Failed to create task');
        return [];
      })
    ).subscribe(newTask => {
      this.tasks.push(newTask);
      this.distributeTasksToColumns();
      this.showCreateModal = false;
      this.taskCreated.emit(newTask);
      this.cdr.markForCheck();
    });
  }
  
  // Utility Methods
  private distributeTasksToColumns(): void {
    this.columns.forEach(column => {
      column.cards = this.tasks.filter(task => 
        task.columnId === column.id &&
        this.matchesFilter(task)
      );
    });
  }
  
  private matchesFilter(task: Task): boolean {
    if (this.searchQuery && !task.title.toLowerCase().includes(this.searchQuery.toLowerCase())) {
      return false;
    }
    
    if (this.filterOptions.priority && task.priority !== this.filterOptions.priority) {
      return false;
    }
    
    if (this.filterOptions.assignee && task.assigneeId !== this.filterOptions.assignee) {
      return false;
    }
    
    return true;
  }
  
  private canMoveTask(task: Task, newColumnId: string): boolean {
    const targetColumn = this.columns.find(col => col.id === newColumnId);
    if (!targetColumn) return false;
    
    if (targetColumn.maxCards && targetColumn.cards.length >= targetColumn.maxCards) {
      this.notificationService.showWarning(`Column "${targetColumn.title}" is at maximum capacity`);
      return false;
    }
    
    return true;
  }
  
  private updateTaskColumn(task: Task, newColumnId: string): void {
    const updatedTask = { ...task, columnId: newColumnId };
    
    this.taskService.updateTask(task.id, updatedTask)
      .pipe(takeUntil(this.destroy$))
      .subscribe({
        next: (updated) => {
          const index = this.tasks.findIndex(t => t.id === task.id);
          if (index !== -1) {
            this.tasks[index] = updated;
          }
          this.taskUpdated.emit(updated);
        },
        error: () => {
          this.notificationService.showError('Failed to update task');
          this.initializeBoard(); // Reload to restore state
        }
      });
  }
  
  // Search and Filter Methods
  onSearchChange(query: string): void {
    this.searchQuery = query;
    this.distributeTasksToColumns();
    this.cdr.markForCheck();
  }
  
  onFilterChange(filters: FilterOptions): void {
    this.filterOptions = filters;
    this.distributeTasksToColumns();
    this.cdr.markForCheck();
  }
  
  // Error Handling
  onRetry(): void {
    this.errorSubject.next(null);
    this.initializeBoard();
  }
  
  // Accessibility Methods
  onKeyboardNavigation(event: KeyboardEvent, task: Task): void {
    switch (event.key) {
      case 'Enter':
      case ' ':
        event.preventDefault();
        this.onTaskSelect(task);
        break;
      case 'Delete':
        event.preventDefault();
        this.confirmDeleteTask(task);
        break;
    }
  }
  
  // Track By Functions for Performance
  trackByColumnId(index: number, column: KanbanColumn): string {
    return column.id;
  }
  
  trackByTaskId(index: number, task: Task): string {
    return task.id;
  }
}

export class KanbanColumnComponent {
  @Input() column!: KanbanColumn;
  @Input() tasks: Task[] = [];
  @Input() readonly: boolean = false;
  @Input() showAddButton: boolean = true;
  
  @Output() taskDrop = new EventEmitter<CdkDragDrop<Task[]>>();
  @Output() taskSelect = new EventEmitter<Task>();
  @Output() taskCreate = new EventEmitter<void>();
  @Output() keyboardNavigation = new EventEmitter<{event: KeyboardEvent, task: Task}>();
  
  get isAtCapacity(): boolean {
    return this.column.maxCards !== null && this.tasks.length >= this.column.maxCards;
  }
  
  get capacityPercentage(): number {
    if (this.column.maxCards === null) return 0;
    return (this.tasks.length / this.column.maxCards) * 100;
  }
  
  onDrop(event: CdkDragDrop<Task[]>): void {
    this.taskDrop.emit(event);
  }
  
  onTaskClick(task: Task): void {
    if (!this.readonly) {
      this.taskSelect.emit(task);
    }
  }
  
  onAddTask(): void {
    if (!this.readonly && !this.isAtCapacity) {
      this.taskCreate.emit();
    }
  }
  
  onTaskKeydown(event: KeyboardEvent, task: Task): void {
    this.keyboardNavigation.emit({ event, task });
  }
  
  trackByTaskId(index: number, task: Task): string {
    return task.id;
  }
}

import { Component, Input, Output, EventEmitter, ChangeDetectionStrategy } from '@angular/core';

@Component({
  selector: 'app-task-card',
  templateUrl: './task-card.component.html',
  styleUrls: ['./task-card.component.scss'],
  changeDetection: ChangeDetectionStrategy.OnPush
})
export class TaskCardComponent {
  @Input() task!: Task;
  @Input() draggable: boolean = true;
  @Input() selectable: boolean = true;
  
  @Output() taskClick = new EventEmitter<Task>();
  
  get priorityColor(): string {
    const colors = {
      LOW: '#4caf50',
      MEDIUM: '#ff9800',
      HIGH: '#f44336',
      CRITICAL: '#9c27b0'
    };
    return colors[this.task.priority] || colors.MEDIUM;
  }
  
  get isOverdue(): boolean {
    if (!this.task.dueDate) return false;
    return new Date(this.task.dueDate) < new Date();
  }
  
  get dueDateFormatted(): string {
    if (!this.task.dueDate) return '';
    return new Date(this.task.dueDate).toLocaleDateString();
  }
  
  onClick(): void {
    if (this.selectable) {
      this.taskClick.emit(this.task);
    }
  }
  
  onKeydown(event: KeyboardEvent): void {
    if (event.key === 'Enter' || event.key === ' ') {
      event.preventDefault();
      this.onClick();
    }
  }
}

// src/app/modals/create-task-modal/create-task-modal.component.ts
import { Component, Input, Output, EventEmitter, OnInit, ChangeDetectionStrategy } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';

@Component({
  selector: 'app-create-task-modal',
  templateUrl: './create-task-modal.component.html',
  styleUrls: ['./create-task-modal.component.scss'],
  changeDetection: ChangeDetectionStrategy.OnPush
})
export class CreateTaskModalComponent implements OnInit {
  @Input() visible: boolean = false;
  @Input() initialColumn: string = 'todo';
  @Input() users: User[] = [];
  
  @Output() taskSubmit = new EventEmitter<CreateTaskRequest>();
  @Output() cancel = new EventEmitter<void>();
  
  taskForm!: FormGroup;
  submitting: boolean = false;
  
  priorityOptions = [
    { value: 'LOW', label: 'Low', color: '#4caf50' },
    { value: 'MEDIUM', label: 'Medium', color: '#ff9800' },
    { value: 'HIGH', label: 'High', color: '#f44336' },
    { value: 'CRITICAL', label: 'Critical', color: '#9c27b0' }
  ];
  
  constructor(private fb: FormBuilder) {}
  
  ngOnInit(): void {
    this.initializeForm();
  }
  
  private initializeForm(): void {
    this.taskForm = this.fb.group({
      title: ['', [Validators.required, Validators.maxLength(100)]],
      description: ['', [Validators.maxLength(500)]],
      priority: ['MEDIUM', [Validators.required]],
      assigneeId: [''],
      dueDate: [''],
      estimatedHours: ['', [Validators.min(0.5), Validators.max(40)]],
      tags: [[]]
    });
  }
  
  onSubmit(): void {
    if (this.taskForm.valid && !this.submitting) {
      this.submitting = true;
      
      const formValue = this.taskForm.value;
      const taskData: CreateTaskRequest = {
        ...formValue,
        columnId: this.initialColumn,
        tags: formValue.tags || []
      };
      
      this.taskSubmit.emit(taskData);
    }
  }
  
  onCancel(): void {
    this.cancel.emit();
    this.resetForm();
  }
  
  onTagAdd(tag: string): void {
    const currentTags = this.taskForm.get('tags')?.value || [];
    if (tag.trim() && !currentTags.includes(tag.trim())) {
      this.taskForm.patchValue({
        tags: [...currentTags, tag.trim()]
      });
    }
  }
  
  onTagRemove(tagToRemove: string): void {
    const currentTags = this.taskForm.get('tags')?.value || [];
    this.taskForm.patchValue({
      tags: currentTags.filter((tag: string) => tag !== tagToRemove)
    });
  }
  
  private resetForm(): void {
    this.taskForm.reset({
      priority: 'MEDIUM',
      tags: []
    });
    this.submitting = false;
  }
  
  getFieldError(fieldName: string): string | null {
    const field = this.taskForm.get(fieldName);
    if (field?.errors && field.touched) {
      if (field.errors['required']) return `${fieldName} is required`;
      if (field.errors['maxlength']) return `${fieldName} is too long`;
      if (field.errors['min']) return `Minimum value is ${field.errors['min'].min}`;
      if (field.errors['max']) return `Maximum value is ${field.errors['max'].max}`;
    }
    return null;
  }
}

// src/app/features/tasks/components/task-detail/task-detail.component.ts
import { Component, Input, OnInit, OnDestroy, ChangeDetectionStrategy, ChangeDetectorRef } from '@angular/core';
import { ActivatedRoute, Router } from '@angular/router';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { Subject, Observable, combineLatest } from 'rxjs';
import { takeUntil, switchMap, catchError, finalize } from 'rxjs/operators';

@Component({
  selector: 'app-task-detail',
  templateUrl: './task-detail.component.html',
  styleUrls: ['./task-detail.component.scss'],
  changeDetection: ChangeDetectionStrategy.OnPush
})
export class TaskDetailComponent implements OnInit, OnDestroy {
  @Input() taskId?: string;
  @Input() editable: boolean = true;
  
  private destroy$ = new Subject<void>();
  
  task: Task | null = null;
  comments: Comment[] = [];
  loading: boolean = false;
  editing: boolean = false;
  submitting: boolean = false;
  
  taskForm!: FormGroup;
  newComment: string = '';
  
  constructor(
    private route: ActivatedRoute,
    private router: Router,
    private fb: FormBuilder,
    private taskService: TaskService,
    private commentService: CommentService,
    private notificationService: NotificationService,
    private cdr: ChangeDetectorRef
  ) {}
  
  ngOnInit(): void {
    this.initializeForm();
    this.loadTaskData();
  }
  
  ngOnDestroy(): void {
    this.destroy$.next();
    this.destroy$.complete();
  }
  
  private initializeForm(): void {
    this.taskForm = this.fb.group({
      title: ['', [Validators.required, Validators.maxLength(100)]],
      description: ['', [Validators.maxLength(500)]],
      priority: ['MEDIUM', [Validators.required]],
      assigneeId: [''],
      dueDate: [''],
      estimatedHours: ['', [Validators.min(0.5), Validators.max(40)]],
      tags: [[]]
    });
  }
  
  private loadTaskData(): void {
    const id = this.taskId || this.route.snapshot.paramMap.get('id');
    if (!id) {
      this.router.navigate(['/kanban']);
      return;
    }
    
    this.loading = true;
    
    combineLatest([
      this.taskService.getTask(id),
      this.commentService.getTaskComments(id)
    ]).pipe(
      takeUntil(this.destroy$),
      catchError(error => {
        this.notificationService.showError('Failed to load task details');
        this.router.navigate(['/kanban']);
        return [];
      }),
      finalize(() => {
        this.loading = false;
        this.cdr.markForCheck();
      })
    ).subscribe(([task, comments]) => {
      this.task = task;
      this.comments = comments;
      this.populateForm(task);
    });
  }
  
  private populateForm(task: Task): void {
    this.taskForm.patchValue({
      title: task.title,
      description: task.description,
      priority: task.priority,
      assigneeId: task.assigneeId,
      dueDate: task.dueDate ? new Date(task.dueDate).toISOString().split('T')[0] : '',
      estimatedHours: task.estimatedHours,
      tags: task.tags || []
    });
  }
  
  onEdit(): void {
    this.editing = true;
  }
  
  onCancelEdit(): void {
    this.editing = false;
    if (this.task) {
      this.populateForm(this.task);
    }
  }
  
  onSave(): void {
    if (this.taskForm.valid && this.task && !this.submitting) {
      this.submitting = true;
      
      const updatedTask: Task = {
        ...this.task,
        ...this.taskForm.value,
        updatedAt: new Date().toISOString()
      };
      
      this.taskService.updateTask(this.task.id, updatedTask)
        .pipe(
          takeUntil(this.destroy$),
          finalize(() => {
            this.submitting = false;
            this.cdr.markForCheck();
          })
        )
        .subscribe({
          next: (updated) => {
            this.task = updated;
            this.editing = false;
            this.notificationService.showSuccess('Task updated successfully');
          },
          error: () => {
            this.notificationService.showError('Failed to update task');
          }
        });
    }
  }
  
  onDelete(): void {
    if (this.task && confirm('Are you sure you want to delete this task?')) {
      this.taskService.deleteTask(this.task.id)
        .pipe(takeUntil(this.destroy$))
        .subscribe({
          next: () => {
            this.notificationService.showSuccess('Task deleted successfully');
            this.router.navigate(['/kanban']);
          },
          error: () => {
            this.notificationService.showError('Failed to delete task');
          }
        });
    }
  }
  
  onAddComment(): void {
    if (this.newComment.trim() && this.task) {
      const comment: CreateCommentRequest = {
        taskId: this.task.id,
        content: this.newComment.trim()
      };
      
      this.commentService.addComment(comment)
        .pipe(takeUntil(this.destroy$))
        .subscribe({
          next: (newComment) => {
            this.comments.unshift(newComment);
            this.newComment = '';
            this.cdr.markForCheck();
          },
          error: () => {
            this.notificationService.showError('Failed to add comment');
          }
        });
    }
  }
  
  onBack(): void {
    this.router.navigate(['/kanban']);
  }
  
  get priorityColor(): string {
    const colors = {
      LOW: '#4caf50',
      MEDIUM: '#ff9800',
      HIGH: '#f44336',
      CRITICAL: '#9c27b0'
    };
    return colors[this.task?.priority || 'MEDIUM'] || colors.MEDIUM;
  }
  
  get isOverdue(): boolean {
    if (!this.task?.dueDate) return false;
    return new Date(this.task.dueDate) < new Date();
  }
  
  trackByCommentId(index: number, comment: Comment): string {
    return comment.id;
  }
}
