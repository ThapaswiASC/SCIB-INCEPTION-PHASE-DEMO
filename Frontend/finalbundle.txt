```typescript
import { Component, OnInit, Input, Output, EventEmitter } from '@angular/core';
import { CdkDragDrop, moveItemInArray, transferArrayItem } from '@angular/cdk/drag-drop';

export interface Task {
  id: string;
  title: string;
  description: string;
  priority: 'low' | 'medium' | 'high';
  assignee: User;
  dueDate: Date;
  tags: string[];
  status: 'todo' | 'inprogress' | 'done';
}

export interface KanbanColumn {
  id: string;
  title: string;
  tasks: Task[];
  count: number;
  indicator: string;
}

export interface User {
  id: string;
  name: string;
  avatar: string;
  email: string;
}

@Component({
  selector: 'app-kanban-board',
  templateUrl: './kanban-board.component.html',
  styleUrls: ['./kanban-board.component.scss']
})
export class KanbanBoardComponent implements OnInit {
  @Input() loading: boolean = false;
  @Input() error: string = '';
  @Input() dragEnabled: boolean = true;
  @Output() taskMoved = new EventEmitter<any>();
  @Output() taskCreated = new EventEmitter<Task>();
  @Output() taskUpdated = new EventEmitter<Task>();

  columns: KanbanColumn[] = [
    { id: 'todo', title: 'To Do', tasks: [], count: 0, indicator: 'pending' },
    { id: 'inprogress', title: 'In Progress', tasks: [], count: 0, indicator: 'active' },
    { id: 'done', title: 'Done', tasks: [], count: 0, indicator: 'completed' }
  ];

  draggedTask: Task | null = null;
  isLoading: boolean = false;
  errorMessage: string = '';

  ngOnInit(): void {
    this.loadKanbanData();
  }

  loadKanbanData(): void {
    this.isLoading = true;
    // API call to load dashboard data
    // GET /youth-accounts/{id}/dashboard
    // Transform data to kanban structure
    setTimeout(() => {
      this.isLoading = false;
      this.updateColumnCounts();
    }, 1000);
  }

  onTaskDrop(event: CdkDragDrop<Task[]>): void {
    if (event.previousContainer === event.container) {
      moveItemInArray(event.container.data, event.previousIndex, event.currentIndex);
    } else {
      transferArrayItem(
        event.previousContainer.data,
        event.container.data,
        event.previousIndex,
        event.currentIndex
      );
      this.taskMoved.emit({
        task: event.item.data,
        fromColumn: event.previousContainer.id,
        toColumn: event.container.id
      });
    }
    this.updateColumnCounts();
  }

  updateColumnCounts(): void {
    this.columns.forEach(column => {
      column.count = column.tasks.length;
    });
  }

  onTaskCreated(task: Task): void {
    this.columns[0].tasks.push(task);
    this.updateColumnCounts();
    this.taskCreated.emit(task);
  }

  onTaskUpdated(task: Task): void {
    // Update task in appropriate column
    this.taskUpdated.emit(task);
  }
}
```

```typescript
import { Component, Input, Output, EventEmitter } from '@angular/core';
import { Task } from '../kanban-board/kanban-board.component';

@Component({
  selector: 'app-kanban-column',
  templateUrl: './kanban-column.component.html',
  styleUrls: ['./kanban-column.component.scss']
})
export class KanbanColumnComponent {
  @Input() columnId: string = '';
  @Input() title: string = '';
  @Input() tasks: Task[] = [];
  @Input() count: number = 0;
  @Input() indicator: string = '';
  @Input() allowDrop: boolean = true;
  
  @Output() taskDropped = new EventEmitter<any>();
  @Output() addTaskClicked = new EventEmitter<string>();
  @Output() menuToggled = new EventEmitter<string>();
  @Output() taskReordered = new EventEmitter<any>();

  isDragOver: boolean = false;
  isLoading: boolean = false;
  isMenuOpen: boolean = false;

  onDragOver(event: DragEvent): void {
    if (this.allowDrop) {
      event.preventDefault();
      this.isDragOver = true;
    }
  }

  onDragLeave(): void {
    this.isDragOver = false;
  }

  onDrop(event: DragEvent): void {
    event.preventDefault();
    this.isDragOver = false;
    
    if (this.allowDrop) {
      const taskData = event.dataTransfer?.getData('application/json');
      if (taskData) {
        const task = JSON.parse(taskData);
        this.taskDropped.emit({ task, columnId: this.columnId });
      }
    }
  }

  onAddTask(): void {
    this.addTaskClicked.emit(this.columnId);
  }

  toggleMenu(): void {
    this.isMenuOpen = !this.isMenuOpen;
    this.menuToggled.emit(this.columnId);
  }

  trackByTaskId(index: number, task: Task): string {
    return task.id;
  }
}
```

```typescript
import { Component, Input, Output, EventEmitter } from '@angular/core';
import { Task, User } from '../kanban-board/kanban-board.component';

export type Priority = 'low' | 'medium' | 'high';

@Component({
  selector: 'app-task-card',
  templateUrl: './task-card.component.html',
  styleUrls: ['./task-card.component.scss']
})
export class TaskCardComponent {
  @Input() task!: Task;
  @Input() draggable: boolean = true;
  @Input() priority: Priority = 'medium';
  @Input() assignee!: User;
  @Input() dueDate!: Date;
  @Input() tags: string[] = [];
  
  @Output() taskClicked = new EventEmitter<Task>();
  @Output() dragStarted = new EventEmitter<DragEvent>();
  @Output() taskMoved = new EventEmitter<any>();
  @Output() priorityChanged = new EventEmitter<Priority>();

  isDragging: boolean = false;
  isSelected: boolean = false;
  isHovered: boolean = false;

  onClick(): void {
    this.taskClicked.emit(this.task);
  }

  onDragStart(event: DragEvent): void {
    if (this.draggable) {
      this.isDragging = true;
      event.dataTransfer?.setData('application/json', JSON.stringify(this.task));
      this.dragStarted.emit(event);
    }
  }

  onDragEnd(): void {
    this.isDragging = false;
  }

  onMouseEnter(): void {
    this.isHovered = true;
  }

  onMouseLeave(): void {
    this.isHovered = false;
  }

  getPriorityClass(): string {
    return `priority-${this.task.priority}`;
  }

  getFormattedDueDate(): string {
    if (!this.task.dueDate) return '';
    return new Date(this.task.dueDate).toLocaleDateString();
  }

  isOverdue(): boolean {
    if (!this.task.dueDate) return false;
    return new Date(this.task.dueDate) < new Date();
  }
}
```

```typescript
  @Input() collapsed: boolean = false;
  
  @Output() navigationClicked = new EventEmitter<NavItem>();
  @Output() sidebarToggled = new EventEmitter<boolean>();

  activeItem: string = '';

  ngOnInit(): void {
    this.isCollapsed = this.collapsed;
    this.activeItem = this.activeRoute;
  }

  onNavigationClick(item: NavItem): void {
    if (!item.disabled) {
      this.activeItem = item.id;
      this.navigationClicked.emit(item);
    }
  }

  toggleSidebar(): void {
    this.isCollapsed = !this.isCollapsed;
    this.sidebarToggled.emit(this.isCollapsed);
  }

  toggleSection(sectionId: string): void {
    const index = this.expandedSections.indexOf(sectionId);
    if (index > -1) {
      this.expandedSections.splice(index, 1);
    } else {
      this.expandedSections.push(sectionId);
    }
    this.sectionExpanded.emit(sectionId);
  }

  isSectionExpanded(sectionId: string): boolean {
    return this.expandedSections.includes(sectionId);
  }

  isItemActive(item: NavItem): boolean {
    return this.activeItem === item.id || this.activeRoute === item.route;
  }
}
```

```typescript
import { Component, Input, Output, EventEmitter } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { Task, User } from '../kanban-board/kanban-board.component';

export interface TaskFormData {
  title: string;
  description: string;
  priority: 'low' | 'medium' | 'high';
  assigneeId: string;
  dueDate: Date | null;
  tags: string[];
  status: 'todo' | 'inprogress' | 'done';
}

export interface ValidationError {
  field: string;
  message: string;
}

@Component({
  selector: 'app-task-modal',
  templateUrl: './task-modal.component.html',
  styleUrls: ['./task-modal.component.scss']
})
export class TaskModalComponent {
  @Input() task: Task | null = null;
  @Input() isOpen: boolean = false;
  @Input() mode: 'create' | 'edit' = 'create';
  @Input() assignees: User[] = [];
  
  @Output() taskSaved = new EventEmitter<Task>();
  @Output() modalClosed = new EventEmitter<void>();
  @Output() formSubmitted = new EventEmitter<TaskFormData>();
  @Output() validationFailed = new EventEmitter<ValidationError[]>();

  taskForm: FormGroup;
  isSubmitting: boolean = false;
  validationErrors: ValidationError[] = [];
  isDirty: boolean = false;

  constructor(private fb: FormBuilder) {
    this.taskForm = this.createForm();
  }

  ngOnInit(): void {
    if (this.task && this.mode === 'edit') {
      this.populateForm(this.task);
    }
  }

  createForm(): FormGroup {
    return this.fb.group({
      title: ['', [Validators.required, Validators.minLength(3)]],
      description: [''],
      priority: ['medium', Validators.required],
      assigneeId: [''],
      dueDate: [null],
      tags: [[]],
      status: ['todo']
    });
  }

  populateForm(task: Task): void {
    this.taskForm.patchValue({
      title: task.title,
      description: task.description,
      priority: task.priority,
      assigneeId: task.assignee?.id || '',
      dueDate: task.dueDate,
      tags: task.tags,
      status: task.status
    });
  }

  onSubmit(): void {
    if (this.taskForm.valid) {
      this.isSubmitting = true;
      const formData: TaskFormData = this.taskForm.value;
      
      // Simulate API call
      setTimeout(() => {
        const savedTask: Task = {
          id: this.task?.id || this.generateId(),
          ...formData,
          assignee: this.assignees.find(a => a.id === formData.assigneeId) || null
        } as Task;
        
        this.taskSaved.emit(savedTask);
        this.formSubmitted.emit(formData);
        this.isSubmitting = false;
        this.closeModal();
      }, 1000);
    } else {
      this.validateForm();
    }
  }

  validateForm(): void {
    this.validationErrors = [];
    
    Object.keys(this.taskForm.controls).forEach(key => {
      const control = this.taskForm.get(key);
      if (control && control.invalid && control.touched) {
        if (control.errors?.['required']) {
          this.validationErrors.push({
            field: key,
            message: `${key} is required`
          });
        }
        if (control.errors?.['minlength']) {
          this.validationErrors.push({
            field: key,
            message: `${key} must be at least ${control.errors['minlength'].requiredLength} characters`
          });
        }
      }
    });
    
    this.validationFailed.emit(this.validationErrors);
  }

  closeModal(): void {
    this.taskForm.reset();
    this.validationErrors = [];
    this.isDirty = false;
    this.modalClosed.emit();
  }

  onFormChange(): void {
    this.isDirty = true;
  }

  generateId(): string {
    return 'task_' + Math.random().toString(36).substr(2, 9);
  }

  addTag(tag: string): void {
    const currentTags = this.taskForm.get('tags')?.value || [];
    if (tag.trim() && !currentTags.includes(tag.trim())) {
      this.taskForm.patchValue({
        tags: [...currentTags, tag.trim()]
      });
    }
  }

  removeTag(tagToRemove: string): void {
    const currentTags = this.taskForm.get('tags')?.value || [];
    this.taskForm.patchValue({
      tags: currentTags.filter((tag: string) => tag !== tagToRemove)
    });
  }
}
```
