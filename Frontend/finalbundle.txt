### 3.1 KanbanDashboard (Page Component)

**Purpose**: Root page component orchestrating the entire Kanban board experience

**TypeScript Specification:**
```typescript
interface KanbanDashboardProps {
  boardId: string;
  userId: string;
}

interface KanbanDashboardState {
  board: Board | null;
  tasks: Task[];
  loading: boolean;
  error: string | null;
  wsConnected: boolean;
}

interface Board {
  id: string;
  name: string;
  description: string;
  columns: Column[];
  members: User[];
  settings: BoardSettings;
}

interface Task {
  id: string;
  title: string;
  description: string;
  status: 'todo' | 'inprogress' | 'done';
  priority: 'low' | 'medium' | 'high';
  assignee: User | null;
  dueDate: Date | null;
  createdAt: Date;
  updatedAt: Date;
}

class KanbanDashboard extends Component<KanbanDashboardProps, KanbanDashboardState> {
  private wsConnection: WebSocket | null = null;

  async componentDidMount() {
    await this.loadBoardData();
    this.initializeWebSocket();
  }

  async loadBoardData() {
    try {
      this.setState({ loading: true });
      const response = await fetch(`/api/boards/${this.props.boardId}`);
      const board = await response.json();
      this.setState({ board, loading: false });
    } catch (error) {
      this.setState({ error: error.message, loading: false });
    }
  }

  initializeWebSocket() {
    this.wsConnection = new WebSocket(`/ws/boards/${this.props.boardId}`);
    this.wsConnection.onmessage = this.handleWebSocketMessage;
    this.wsConnection.onopen = () => this.setState({ wsConnected: true });
    this.wsConnection.onclose = () => this.setState({ wsConnected: false });
  }

  handleWebSocketMessage = (event: MessageEvent) => {
    const data = JSON.parse(event.data);
    switch (data.type) {
      case 'TASK_CREATED':
      case 'TASK_UPDATED':
      case 'TASK_DELETED':
        this.loadBoardData();
        break;
    }
  };

  handleTaskCreate = async (taskData: Partial<Task>) => {
    try {
      const response = await fetch(`/api/boards/${this.props.boardId}/tasks`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(taskData)
      });
      const newTask = await response.json();
      this.setState(prev => ({ tasks: [...prev.tasks, newTask] }));
    } catch (error) {
      this.setState({ error: error.message });
    }
  };

  handleTaskUpdate = async (taskId: string, updates: Partial<Task>) => {
    try {
      const response = await fetch(`/api/boards/${this.props.boardId}/tasks/${taskId}`, {
        method: 'PUT',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(updates)
      });
      const updatedTask = await response.json();
      this.setState(prev => ({
        tasks: prev.tasks.map(task => task.id === taskId ? updatedTask : task)
      }));
    } catch (error) {
      this.setState({ error: error.message });
    }
  };

  handleTaskDelete = async (taskId: string) => {
    try {
      await fetch(`/api/boards/${this.props.boardId}/tasks/${taskId}`, {
        method: 'DELETE'
      });
      this.setState(prev => ({
        tasks: prev.tasks.filter(task => task.id !== taskId)
      }));
    } catch (error) {
      this.setState({ error: error.message });
    }
  };
}
```

**HTML Structure:**
```html
<div className="kanban-dashboard">
```

### 3.3 KanbanBoardComponent (Feature Component)

**Purpose**: Core three-column Kanban board with drag-and-drop

**TypeScript Specification:**
```typescript
import { DndProvider, useDrag, useDrop } from 'react-dnd';
import { HTML5Backend } from 'react-dnd-html5-backend';

interface KanbanBoardProps {
  board: Board;
  tasks: Task[];
  onTaskCreate: (taskData: Partial<Task>) => void;
  onTaskUpdate: (taskId: string, updates: Partial<Task>) => void;
  onTaskDelete: (taskId: string) => void;
  onTaskMove: (taskId: string, newStatus: string) => void;
}

interface KanbanBoardState {
  showCreateModal: boolean;
  selectedTask: Task | null;
  showEditModal: boolean;
  showDetailModal: boolean;
}

class KanbanBoardComponent extends Component<KanbanBoardProps, KanbanBoardState> {
  state = {
    showCreateModal: false,
    selectedTask: null,
    showEditModal: false,
    showDetailModal: false
  };

  getTasksByStatus = (status: string) => {
    return this.props.tasks.filter(task => task.status === status);
  };

  handleTaskMove = (taskId: string, newStatus: string) => {
    this.props.onTaskMove(taskId, newStatus);
  };

  openCreateModal = () => {
    this.setState({ showCreateModal: true });
  };

  closeCreateModal = () => {
    this.setState({ showCreateModal: false });
  };

  openTaskDetail = (task: Task) => {
    this.setState({ selectedTask: task, showDetailModal: true });
  };

  openTaskEdit = (task: Task) => {
    this.setState({ selectedTask: task, showEditModal: true });
  };
}
```

### 3.4 KanbanColumnComponent (Feature Component)

**Purpose**: Individual column containers (To Do, In Progress, Done)

**TypeScript Specification:**
```typescript
import { useDrop } from 'react-dnd';

interface KanbanColumnProps {
  title: string;
  status: string;
  tasks: Task[];
  onTaskMove: (taskId: string, newStatus: string) => void;
  onTaskClick: (task: Task) => void;
  onTaskEdit: (task: Task) => void;
  onTaskDelete: (taskId: string) => void;
}

const KanbanColumnComponent: React.FC<KanbanColumnProps> = ({
  title,
  status,
  tasks,
  onTaskMove,
  onTaskClick,
  onTaskEdit,
  onTaskDelete
}) => {
  const [{ isOver }, drop] = useDrop({
    accept: 'TASK',
    drop: (item: { id: string }) => {
      onTaskMove(item.id, status);
    },
    collect: (monitor) => ({
      isOver: monitor.isOver()
    })
  });

  const handleAddTask = () => {
    // Trigger task creation with pre-filled status
    onTaskCreate({ status });
  };

  return (
    <div
      ref={drop}
      className={`kanban-column ${isOver ? 'drag-over' : ''}`}
      role="region"
      aria-label={`${title} column`}
    >
      <div className="column-header">
        <h3 className="column-title">{title}</h3>
        <span className="task-count">{tasks.length}</span>
        <AavaButtonComponent
          variant="ghost"
          size="small"
          icon="plus"
          onClick={handleAddTask}
          aria-label={`Add task to ${title}`}
        />
      </div>
      
      <div className="column-content">
        {tasks.map((task) => (
          <KanbanCardComponent
            key={task.id}
            task={task}
            onClick={() => onTaskClick(task)}
            onEdit={() => onTaskEdit(task)}
            onDelete={() => onTaskDelete(task.id)}
          />
        ))}
        
        {tasks.length === 0 && (
          <div className="empty-column">
            <p>No tasks in {title.toLowerCase()}</p>
            <AavaButtonComponent
              variant="outline"
              onClick={handleAddTask}
            >
              Add First Task
            </AavaButtonComponent>
          </div>
        )}
      </div>
    </div>
  );
};
```

### 3.5 KanbanCardComponent (Feature Component)

**Purpose**: Individual task cards with rich information display

**TypeScript Specification:**
```typescript
import { useDrag } from 'react-dnd';

interface KanbanCardProps {
  task: Task;
  onClick: () => void;
  onEdit: () => void;
  onDelete: () => void;
}

const KanbanCardComponent: React.FC<KanbanCardProps> = ({
  task,
  onClick,
  onEdit,
  onDelete
}) => {
  const [{ isDragging }, drag] = useDrag({
    type: 'TASK',
    item: { id: task.id },
    collect: (monitor) => ({
      isDragging: monitor.isDragging()
    })
  });

  const getPriorityColor = (priority: string) => {
    switch (priority) {
      case 'high': return 'var(--color-error)';
      case 'medium': return 'var(--color-warning)';
      case 'low': return 'var(--color-success)';
      default: return 'var(--color-secondary)';
    }
  };

  const formatDueDate = (date: Date | null) => {
    if (!date) return null;
    const now = new Date();
    const diffTime = date.getTime() - now.getTime();
    const diffDays = Math.ceil(diffTime / (1000 * 60 * 60 * 24));
    
    if (diffDays < 0) return 'Overdue';
    if (diffDays === 0) return 'Due today';
    if (diffDays === 1) return 'Due tomorrow';
    return `Due in ${diffDays} days`;
  };

  return (
    <div
      ref={drag}
      className={`kanban-card ${isDragging ? 'dragging' : ''}`}
      onClick={onClick}
      role="button"
      tabIndex={0}
      aria-label={`Task: ${task.title}`}
      onKeyDown={(e) => {
        if (e.key === 'Enter' || e.key === ' ') {
          e.preventDefault();
          onClick();
        }
      }}
    >
      <div className="card-header">
        <div 
          className="priority-indicator"
          style={{ backgroundColor: getPriorityColor(task.priority) }}
          aria-label={`Priority: ${task.priority}`}
        />
        <AavaPopoverComponent
          trigger={(
            <button 
              className="card-menu-trigger"
              onClick={(e) => e.stopPropagation()}
              aria-label="Task actions"
            >
              ⋯
            </button>
          )}
          content={(
            <div className="card-menu">
              <button onClick={(e) => { e.stopPropagation(); onEdit(); }}>Edit</button>
              <button onClick={(e) => { e.stopPropagation(); onDelete(); }}>Delete</button>
            </div>
          )}
        />
      </div>
      
      <div className="card-content">
        <h4 className="card-title">{task.title}</h4>
        {task.description && (
          <p className="card-description">{task.description}</p>
        )}
      </div>
      
      <div className="card-footer">
        <div className="card-meta">
          {task.dueDate && (
            <span 
              className={`due-date ${formatDueDate(task.dueDate)?.includes('Overdue') ? 'overdue' : ''}`}
            >
              {formatDueDate(task.dueDate)}
            </span>
          )}
        </div>
        
        {task.assignee && (
          <AavaAvatarsComponent
            user={task.assignee}
            size="small"
            showTooltip={true}
          />
        )}
      </div>
    </div>
  );
};
```
