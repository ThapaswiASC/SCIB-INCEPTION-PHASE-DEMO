**IMPORTANT: FULL CODE MUST BE PRESERVED - DO NOT MODIFY OR SUMMARIZE**

### TaskCreationComponent (Primary Focus)

**TypeScript Specification:**
```typescript
import { Component, Input, Output, EventEmitter, OnInit, OnDestroy } from '@angular/core';
import { FormBuilder, FormGroup, Validators, AbstractControl } from '@angular/forms';
import { Subject, takeUntil, debounceTime, distinctUntilChanged } from 'rxjs';
import { TaskService } from '../../services/task.service';
import { Task, TaskPriority, TaskStatus, CreateTaskRequest } from '../../models/task.model';
import { ValidationError } from '../../models/api-response.model';

@Component({
  selector: 'app-task-creation',
  templateUrl: './task-creation.component.html',
  styleUrls: ['./task-creation.component.scss']
})
export class TaskCreationComponent implements OnInit, OnDestroy {
  @Input() isOpen: boolean = false;
  @Input() initialData?: Partial<Task>;
  @Output() onClose = new EventEmitter<void>();
  @Output() onTaskCreated = new EventEmitter<Task>();

  taskForm: FormGroup;
  isLoading: boolean = false;
  validationErrors: ValidationError[] = [];
  characterCounts = {
    title: 0,
    description: 0
  };
  
  readonly maxLengths = {
    title: 255,
    description: 2000
  };
  
  readonly priorityOptions = [
    { value: TaskPriority.LOW, label: 'Low', color: '#10B981' },
    { value: TaskPriority.MEDIUM, label: 'Medium', color: '#F59E0B' },
    { value: TaskPriority.HIGH, label: 'High', color: '#EF4444' },
    { value: TaskPriority.CRITICAL, label: 'Critical', color: '#7C2D12' }
  ];

  private destroy$ = new Subject<void>();

  constructor(
    private fb: FormBuilder,
    private taskService: TaskService
  ) {
    this.initializeForm();
  }

  ngOnInit(): void {
    this.setupFormValidation();
    this.setupCharacterCounters();
    
    if (this.initialData) {
      this.populateForm(this.initialData);
    }
  }

  ngOnDestroy(): void {
    this.destroy$.next();
    this.destroy$.complete();
  }

  private initializeForm(): void {
    this.taskForm = this.fb.group({
      title: ['', [
        Validators.required,
        Validators.maxLength(this.maxLengths.title),
        this.noWhitespaceValidator
      ]],
      description: ['', [
        Validators.maxLength(this.maxLengths.description)
      ]],
      priority: [TaskPriority.MEDIUM, [Validators.required]],
      dueDate: ['', [Validators.required, this.futureDateValidator]],
      assigneeId: [''],
      tags: [[]]
    });
  }

  private setupFormValidation(): void {
    this.taskForm.statusChanges
      .pipe(
        debounceTime(300),
        distinctUntilChanged(),
        takeUntil(this.destroy$)
      )
      .subscribe(() => {
        this.updateValidationErrors();
      });
  }

  private setupCharacterCounters(): void {
    // Title character counter
    this.taskForm.get('title')?.valueChanges
      .pipe(takeUntil(this.destroy$))
      .subscribe(value => {
        this.characterCounts.title = value ? value.length : 0;
      });

    // Description character counter
    this.taskForm.get('description')?.valueChanges
      .pipe(takeUntil(this.destroy$))
      .subscribe(value => {
        this.characterCounts.description = value ? value.length : 0;
      });
  }

  private populateForm(data: Partial<Task>): void {
    this.taskForm.patchValue({
      title: data.title || '',
      description: data.description || '',
      priority: data.priority || TaskPriority.MEDIUM,
      dueDate: data.dueDate ? new Date(data.dueDate).toISOString().split('T')[0] : '',
      assigneeId: data.assigneeId || '',
      tags: data.tags || []
    });
  }

  private noWhitespaceValidator(control: AbstractControl): {[key: string]: any} | null {
    if (control.value && control.value.trim().length === 0) {
      return { 'whitespace': true };
    }
    return null;
  }

  private futureDateValidator(control: AbstractControl): {[key: string]: any} | null {
    if (control.value) {
      const selectedDate = new Date(control.value);
      const today = new Date();
      today.setHours(0, 0, 0, 0);
      
      if (selectedDate < today) {
        return { 'pastDate': true };
      }
    }
    return null;
  }

  private updateValidationErrors(): void {
    this.validationErrors = [];
    
    Object.keys(this.taskForm.controls).forEach(key => {
      const control = this.taskForm.get(key);
      if (control && control.errors && control.touched) {
        const errors = control.errors;
        
        if (errors['required']) {
          this.validationErrors.push({
            field: key,
            message: `${this.getFieldLabel(key)} is required`,
            code: 'REQUIRED'
          });
        }
        
        if (errors['maxlength']) {
          this.validationErrors.push({
            field: key,
            message: `${this.getFieldLabel(key)} must not exceed ${errors['maxlength'].requiredLength} characters`,
            code: 'MAX_LENGTH'
          });
        }
        
        if (errors['whitespace']) {
          this.validationErrors.push({
            field: key,
            message: `${this.getFieldLabel(key)} cannot be empty or contain only whitespace`,
            code: 'WHITESPACE'
          });
        }
        
        if (errors['pastDate']) {
          this.validationErrors.push({
            field: key,
            message: 'Due date must be in the future',
            code: 'PAST_DATE'
          });
        }
      }
    });
  }

  private getFieldLabel(fieldName: string): string {
    const labels: {[key: string]: string} = {
      title: 'Title',
      description: 'Description',
      priority: 'Priority',
      dueDate: 'Due Date',
      assigneeId: 'Assignee'
    };
    return labels[fieldName] || fieldName;
  }

  onSubmit(): void {
    if (this.taskForm.valid && !this.isLoading) {
      this.isLoading = true;
      this.validationErrors = [];
      
      const formValue = this.taskForm.value;
      const createTaskRequest: CreateTaskRequest = {
        title: formValue.title.trim(),
        description: formValue.description?.trim() || '',
        priority: formValue.priority,
        dueDate: new Date(formValue.dueDate).toISOString(),
        assigneeId: formValue.assigneeId || undefined,
        tags: formValue.tags || []
      };

      this.taskService.createTask(createTaskRequest)
        .pipe(takeUntil(this.destroy$))
        .subscribe({
          next: (createdTask: Task) => {
            this.isLoading = false;
            this.onTaskCreated.emit(createdTask);
            this.resetForm();
            this.onClose.emit();
          },
          error: (error) => {
            this.isLoading = false;
            this.handleApiError(error);
          }
        });
    } else {
      this.markAllFieldsAsTouched();
      this.updateValidationErrors();
    }
  }

  onCancel(): void {
    if (this.hasUnsavedChanges()) {
      const confirmClose = confirm('You have unsaved changes. Are you sure you want to close?');
      if (confirmClose) {
        this.resetForm();
        this.onClose.emit();
      }
    } else {
      this.onClose.emit();
    }
  }

  private hasUnsavedChanges(): boolean {
    return this.taskForm.dirty && !this.taskForm.pristine;
  }

  private resetForm(): void {
    this.taskForm.reset({
      title: '',
      description: '',
      priority: TaskPriority.MEDIUM,
      dueDate: '',
      assigneeId: '',
      tags: []
    });
    this.validationErrors = [];
    this.characterCounts = { title: 0, description: 0 };
  }

  private markAllFieldsAsTouched(): void {
    Object.keys(this.taskForm.controls).forEach(key => {
      this.taskForm.get(key)?.markAsTouched();
    });
  }

  private handleApiError(error: any): void {
    if (error.status === 422 && error.error?.errors) {
      this.validationErrors = error.error.errors;
    } else if (error.status === 409) {
      this.validationErrors = [{
        field: 'title',
        message: 'A task with this title already exists',
        code: 'DUPLICATE_TITLE'
      }];
    } else {
      this.validationErrors = [{
        field: 'general',
        message: 'An error occurred while creating the task. Please try again.',
        code: 'GENERAL_ERROR'
      }];
    }
  }

  getFieldError(fieldName: string): string | null {
    const error = this.validationErrors.find(err => err.field === fieldName);
    return error ? error.message : null;
  }

  isFieldInvalid(fieldName: string): boolean {
    const control = this.taskForm.get(fieldName);
    return !!(control && control.invalid && control.touched);
  }

  getCharacterCountColor(field: 'title' | 'description'): string {
    const count = this.characterCounts[field];
    const max = this.maxLengths[field];
    const percentage = (count / max) * 100;
    
    if (percentage >= 90) return '#EF4444'; // Red
    if (percentage >= 75) return '#F59E0B'; // Orange
    return '#6B7280'; // Gray
  }
}
```

**KanbanBoardComponent:**
```typescript
@Component({
  selector: 'app-kanban-board',
  templateUrl: './kanban-board.component.html',
  styleUrls: ['./kanban-board.component.scss']
})
export class KanbanBoardComponent implements OnInit {
  @Input() tasks: Task[] = [];
  @Input() columns: Column[] = [];
  @Output() onTaskMove = new EventEmitter<{taskId: string, newStatus: TaskStatus}>();
  @Output() onTaskClick = new EventEmitter<string>();
  
  draggedTask: Task | null = null;
  columnTasks: Map<string, Task[]> = new Map();
  
  ngOnInit(): void {
    this.organizeTasksByStatus();
  }
  
  private organizeTasksByStatus(): void {
    this.columnTasks.clear();
    this.columns.forEach(column => {
      const columnTasks = this.tasks.filter(task => task.status === column.status);
      this.columnTasks.set(column.status, columnTasks);
    });
  }
  
  onDragStart(event: DragEvent, task: Task): void {
    this.draggedTask = task;
    event.dataTransfer?.setData('text/plain', task.id);
  }
  
  onDragOver(event: DragEvent): void {
    event.preventDefault();
  }
  
  onDrop(event: DragEvent, newStatus: TaskStatus): void {
    event.preventDefault();
    if (this.draggedTask && this.draggedTask.status !== newStatus) {
      this.onTaskMove.emit({
        taskId: this.draggedTask.id,
        newStatus: newStatus
      });
    }
    this.draggedTask = null;
  }
}
```

```typescript
  characterCount: number = 0;
```
