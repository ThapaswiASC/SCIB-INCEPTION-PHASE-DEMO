**TypeScript Specification:**
```typescript
@Component({
  selector: 'app-kanban-board',
  templateUrl: './kanban-board.component.html',
  styleUrls: ['./kanban-board.component.scss']
})
export class KanbanBoardComponent implements OnInit {
  @Input() columns: KanbanColumn[] = [];
  @Input() tasks: Task[] = [];
  @Input() loading: boolean = false;
  @Input() error: string | null = null;
  
  @Output() taskCreate = new EventEmitter<CreateTaskRequest>();
  @Output() taskMove = new EventEmitter<{taskId: string, columnId: string}>();
  @Output() taskSelect = new EventEmitter<Task>();
  
  selectedTask: Task | null = null;
  draggedTask: Task | null = null;
  columnConfiguration: ColumnConfig = {};
  showCreateModal: boolean = false;
  
  constructor(
    private taskService: TaskService,
    private cdr: ChangeDetectorRef
  ) {}
  
  ngOnInit(): void {
    this.initializeColumns();
    this.loadTasks();
  }
  
  initializeColumns(): void {
    this.columns = [
      {id: 'todo', title: 'To Do', cards: []},
      {id: 'inprogress', title: 'In Progress', cards: []},
      {id: 'done', title: 'Done', cards: []}
    ];
  }
  
  loadTasks(): void {
    this.loading = true;
    this.taskService.getTasks().subscribe({
      next: (tasks) => {
        this.tasks = tasks;
        this.distributeTasksToColumns();
        this.loading = false;
      },
      error: (error) => {
        this.error = 'Failed to load tasks';
        this.loading = false;
      }
    });
  }
  
  onTaskCreate(task: CreateTaskRequest): void {
    this.taskService.createTask(task).subscribe({
      next: (newTask) => {
        this.tasks.push(newTask);
        this.distributeTasksToColumns();
        this.showCreateModal = false;
      },
      error: (error) => {
        this.error = 'Failed to create task';
      }
    });
  }
  
  onTaskMove(event: {taskId: string, columnId: string}): void {
    const task = this.tasks.find(t => t.id === event.taskId);
    if (task) {
      task.status = event.columnId;
      this.distributeTasksToColumns();
    }
  }
  
  private distributeTasksToColumns(): void {
    this.columns.forEach(column => {
      column.cards = this.tasks.filter(task => task.status === column.id);
    });
  }
}
```

```typescript
@Component({
  selector: 'app-kanban-column',
  templateUrl: './kanban-column.component.html',
  styleUrls: ['./kanban-column.component.scss']
})
export class KanbanColumnComponent {
  @Input() column!: KanbanColumn;
  @Input() tasks: Task[] = [];
  @Input() maxTasks: number | null = null;
  
  @Output() taskDrop = new EventEmitter<{taskId: string, columnId: string}>();
  @Output() addTask = new EventEmitter<void>();
  
  isDropTarget: boolean = false;
  
  get taskCount(): number {
    return this.tasks.length;
  }
  
  get isMaxTasksReached(): boolean {
    return this.maxTasks !== null && this.taskCount >= this.maxTasks;
  }
  
  onDragOver(event: DragEvent): void {
    event.preventDefault();
    this.isDropTarget = true;
  }
  
  onDragLeave(event: DragEvent): void {
    this.isDropTarget = false;
  }
  
  onDrop(event: DragEvent): void {
    event.preventDefault();
    this.isDropTarget = false;
    
    const taskId = event.dataTransfer?.getData('text/plain');
    if (taskId && !this.isMaxTasksReached) {
      this.taskDrop.emit({taskId, columnId: this.column.id});
    }
  }
  
  onAddTaskClick(): void {
    if (!this.isMaxTasksReached) {
      this.addTask.emit();
    }
  }
  
  trackByTaskId(index: number, task: Task): string {
    return task.id;
  }
}
```

#### TaskCardComponent

**TypeScript Specification:**
```typescript
@Component({
  selector: 'app-task-card',
  templateUrl: './task-card.component.html',
  styleUrls: ['./task-card.component.scss']
})
export class TaskCardComponent {
  @Input() task!: Task;
  @Input() draggable: boolean = true;
  @Input() selectable: boolean = true;
  
  @Output() taskClick = new EventEmitter<Task>();
  @Output() dragStart = new EventEmitter<Task>();
  @Output() dragEnd = new EventEmitter<void>();
  
  isDragging: boolean = false;
  isSelected: boolean = false;
  
  get priorityClass(): string {
    return `priority-${this.task.priority?.toLowerCase() || 'medium'}`;
  }
  
  get priorityLabel(): string {
    const priorities = {
      high: 'High Priority',
      medium: 'Medium Priority',
      low: 'Low Priority'
    };
    return priorities[this.task.priority?.toLowerCase() as keyof typeof priorities] || 'Medium Priority';
  }
  
  onCardClick(): void {
    if (this.selectable) {
      this.isSelected = !this.isSelected;
      this.taskClick.emit(this.task);
    }
  }
  
  onDragStart(event: DragEvent): void {
    if (this.draggable) {
      this.isDragging = true;
      event.dataTransfer?.setData('text/plain', this.task.id);
      this.dragStart.emit(this.task);
    }
  }
  
  onDragEnd(): void {
    this.isDragging = false;
    this.dragEnd.emit();
  }
  
  getAssigneeInitials(name: string): string {
    return name.split(' ').map(n => n[0]).join('').toUpperCase();
  }
}
```

```typescript
@Component({
  selector: 'app-create-task-modal',
  templateUrl: './create-task-modal.component.html',
  styleUrls: ['./create-task-modal.component.scss']
})
export class CreateTaskModalComponent implements OnInit {
  @Input() isOpen: boolean = false;
  @Input() users: User[] = [];
  @Input() columns: KanbanColumn[] = [];
  
  @Output() submit = new EventEmitter<CreateTaskRequest>();
  @Output() cancel = new EventEmitter<void>();
  
  formData: CreateTaskRequest = {
    title: '',
    description: '',
    priority: 'medium',
    assigneeId: '',
    columnId: 'todo',
    dueDate: null,
    tags: []
  };
  
  validationErrors: ValidationError[] = [];
  isSubmitting: boolean = false;
  
  priorities = [
    { value: 'low', label: 'Low Priority' },
    { value: 'medium', label: 'Medium Priority' },
    { value: 'high', label: 'High Priority' }
  ];
  
  ngOnInit(): void {
    if (this.columns.length > 0) {
      this.formData.columnId = this.columns[0].id;
    }
  }
  
  onSubmit(): void {
    if (this.validateForm()) {
      this.isSubmitting = true;
      this.submit.emit({ ...this.formData });
    }
  }
  
  onCancel(): void {
    this.resetForm();
    this.cancel.emit();
  }
  
  private validateForm(): boolean {
    this.validationErrors = [];
    
    if (!this.formData.title.trim()) {
      this.validationErrors.push({
        field: 'title',
        message: 'Task title is required'
      });
    }
    
    if (this.formData.title.length > 100) {
      this.validationErrors.push({
        field: 'title',
        message: 'Task title must be less than 100 characters'
      });
    }
    
    if (this.formData.description && this.formData.description.length > 500) {
      this.validationErrors.push({
        field: 'description',
        message: 'Description must be less than 500 characters'
      });
    }
    
    return this.validationErrors.length === 0;
  }
  
  private resetForm(): void {
    this.formData = {
      title: '',
      description: '',
      priority: 'medium',
      assigneeId: '',
      columnId: this.columns.length > 0 ? this.columns[0].id : 'todo',
      dueDate: null,
      tags: []
    };
    this.validationErrors = [];
    this.isSubmitting = false;
  }
  
  getFieldError(field: string): string | null {
    const error = this.validationErrors.find(e => e.field === field);
    return error ? error.message : null;
  }
  
  hasFieldError(field: string): boolean {
    return this.validationErrors.some(e => e.field === field);
  }
}
```

```typescript
    this.toggleCollapse.emit();

    this.currentMenuItems.forEach(item => {
      item.active = item.route === this.activeRoute;
    });
  }
```
