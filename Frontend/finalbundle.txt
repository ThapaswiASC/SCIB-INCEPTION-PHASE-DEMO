```typescript
import { Component, OnInit, Input, Output, EventEmitter, ChangeDetectionStrategy } from '@angular/core';
import { CdkDragDrop, moveItemInArray, transferArrayItem } from '@angular/cdk/drag-drop';

export interface Task {
  id: string;
  title: string;
  description?: string;
  priority: 'low' | 'medium' | 'high';
  assignee?: string;
  dueDate?: Date;
  tags?: string[];
  effort?: number;
}

export interface KanbanColumn {
  id: string;
  title: string;
  tasks: Task[];
  maxTasks?: number;
}

export interface TaskMoveEvent {
  task: Task;
  fromColumn: string;
  toColumn: string;
  fromIndex: number;
  toIndex: number;
}

@Component({
  selector: 'app-kanban-board',
  templateUrl: './kanban-board.component.html',
  styleUrls: ['./kanban-board.component.scss'],
  changeDetection: ChangeDetectionStrategy.OnPush
})
export class KanbanBoardComponent implements OnInit {
  @Input() columns: KanbanColumn[] = [];
  @Input() loading: boolean = false;
  @Input() error: string | null = null;
  
  @Output() taskMoved = new EventEmitter<TaskMoveEvent>();
  @Output() taskCreated = new EventEmitter<Task>();
  @Output() taskUpdated = new EventEmitter<Task>();
  @Output() addTaskClicked = new EventEmitter<string>();

  draggedTask: Task | null = null;
  isMobile: boolean = false;
  activeColumnIndex: number = 0;

  ngOnInit(): void {
    this.checkMobileView();
    this.initializeColumns();
  }

  private initializeColumns(): void {
    if (this.columns.length === 0) {
      this.columns = [
        { id: 'todo', title: 'To Do', tasks: [] },
        { id: 'inprogress', title: 'In Progress', tasks: [] },
        { id: 'done', title: 'Done', tasks: [] }
      ];
    }
  }

  private checkMobileView(): void {
    this.isMobile = window.innerWidth < 768;
  }

  onDrop(event: CdkDragDrop<Task[]>, targetColumnId: string): void {
    const sourceColumn = this.columns.find(col => col.tasks === event.previousContainer.data);
    const targetColumn = this.columns.find(col => col.id === targetColumnId);
    
    if (!sourceColumn || !targetColumn) return;

    if (event.previousContainer === event.container) {
      moveItemInArray(event.container.data, event.previousIndex, event.currentIndex);
    } else {
      const task = sourceColumn.tasks[event.previousIndex];
      transferArrayItem(
        event.previousContainer.data,
        event.container.data,
        event.previousIndex,
        event.currentIndex
      );
      
      this.taskMoved.emit({
        task,
        fromColumn: sourceColumn.id,
        toColumn: targetColumn.id,
        fromIndex: event.previousIndex,
        toIndex: event.currentIndex
      });
    }
  }

  onAddTask(columnId: string): void {
    this.addTaskClicked.emit(columnId);
  }

  onTaskClick(task: Task): void {
    this.taskUpdated.emit(task);
  }

  switchColumn(index: number): void {
    this.activeColumnIndex = index;
  }

  getColumnTasks(columnId: string): Task[] {
    const column = this.columns.find(col => col.id === columnId);
    return column ? column.tasks : [];
  }
}
```

**TypeScript Implementation:**
```typescript
import { Component, Input, Output, EventEmitter, ChangeDetectionStrategy } from '@angular/core';
import { CdkDragDrop } from '@angular/cdk/drag-drop';
import { Task, KanbanColumn } from '../kanban-board/kanban-board.component';

@Component({
  selector: 'app-kanban-column',
  templateUrl: './kanban-column.component.html',
  styleUrls: ['./kanban-column.component.scss'],
  changeDetection: ChangeDetectionStrategy.OnPush
})
export class KanbanColumnComponent {
  @Input() column!: KanbanColumn;
  @Input() tasks: Task[] = [];
  @Input() isMobile: boolean = false;
  @Input() allowDrop: boolean = true;
  
  @Output() taskDropped = new EventEmitter<CdkDragDrop<Task[]>>();
  @Output() addTaskClicked = new EventEmitter<void>();
  @Output() taskClicked = new EventEmitter<Task>();

  isDropTarget: boolean = false;

  onDrop(event: CdkDragDrop<Task[]>): void {
    this.taskDropped.emit(event);
    this.isDropTarget = false;
  }

  onDragEnter(): void {
    if (this.allowDrop) {
      this.isDropTarget = true;
    }
  }

  onDragLeave(): void {
    this.isDropTarget = false;
  }

  onAddTask(): void {
    this.addTaskClicked.emit();
  }

  onTaskClick(task: Task): void {
    this.taskClicked.emit(task);
  }

  trackByTaskId(index: number, task: Task): string {
    return task.id;
  }

  get taskCount(): number {
    return this.tasks.length;
  }

  get isAtCapacity(): boolean {
    return this.column.maxTasks ? this.taskCount >= this.column.maxTasks : false;
  }
}
```

### TaskCardComponent

**TypeScript Implementation:**
```typescript
import { Component, Input, Output, EventEmitter, ChangeDetectionStrategy } from '@angular/core';
import { Task } from '../kanban-board/kanban-board.component';

@Component({
  selector: 'app-task-card',
  templateUrl: './task-card.component.html',
  styleUrls: ['./task-card.component.scss'],
  changeDetection: ChangeDetectionStrategy.OnPush
})
export class TaskCardComponent {
  @Input() task!: Task;
  @Input() draggable: boolean = true;
  @Input() selected: boolean = false;
  
  @Output() taskClicked = new EventEmitter<Task>();
  @Output() taskDragStart = new EventEmitter<Task>();
  @Output() taskDragEnd = new EventEmitter<Task>();

  isDragging: boolean = false;
  isHovered: boolean = false;

  onClick(): void {
    this.taskClicked.emit(this.task);
  }

  onDragStart(): void {
    this.isDragging = true;
    this.taskDragStart.emit(this.task);
  }

  onDragEnd(): void {
    this.isDragging = false;
    this.taskDragEnd.emit(this.task);
  }

  onMouseEnter(): void {
    this.isHovered = true;
  }

  onMouseLeave(): void {
    this.isHovered = false;
  }

  getPriorityClass(): string {
    return `priority-${this.task.priority}`;
  }

  getFormattedDueDate(): string {
    if (!this.task.dueDate) return '';
    return new Date(this.task.dueDate).toLocaleDateString();
  }

  isOverdue(): boolean {
    if (!this.task.dueDate) return false;
    return new Date(this.task.dueDate) < new Date();
  }
}
```

```typescript
import { Component, Input, Output, EventEmitter, OnInit, ChangeDetectionStrategy } from '@angular/core';

export interface User {
  id: string;
  name: string;
  email: string;
  avatar?: string;
  role: string;
}

export interface Notification {
  id: string;
  title: string;
  message: string;
  type: 'info' | 'warning' | 'error' | 'success';
  timestamp: Date;
  read: boolean;
}

@Component({
  selector: 'app-header',
  templateUrl: './header.component.html',
  styleUrls: ['./header.component.scss'],
  changeDetection: ChangeDetectionStrategy.OnPush
})
export class HeaderComponent implements OnInit {
  @Input() user!: User;
  @Input() notifications: Notification[] = [];
  @Input() searchEnabled: boolean = true;
  @Input() currentRoute: string = '';
  
  @Output() searchChanged = new EventEmitter<string>();
  @Output() themeToggled = new EventEmitter<void>();
  @Output() userMenuToggled = new EventEmitter<void>();
  @Output() notificationClicked = new EventEmitter<Notification>();
  @Output() menuToggled = new EventEmitter<void>();

  searchQuery: string = '';
  userMenuOpen: boolean = false;
  notificationsOpen: boolean = false;
  isMobile: boolean = false;

  ngOnInit(): void {
    this.checkMobileView();
  }

  private checkMobileView(): void {
    this.isMobile = window.innerWidth < 768;
  }

  onSearchInput(event: Event): void {
    const target = event.target as HTMLInputElement;
    this.searchQuery = target.value;
    this.searchChanged.emit(this.searchQuery);
  }

  onThemeToggle(): void {
    this.themeToggled.emit();
  }

  toggleUserMenu(): void {
    this.userMenuOpen = !this.userMenuOpen;
    this.notificationsOpen = false;
    this.userMenuToggled.emit();
  }

  toggleNotifications(): void {
    this.notificationsOpen = !this.notificationsOpen;
    this.userMenuOpen = false;
  }

  onNotificationClick(notification: Notification): void {
    this.notificationClicked.emit(notification);
  }

  onMenuToggle(): void {
    this.menuToggled.emit();
  }

  getBreadcrumbs(): string[] {
    return this.currentRoute.split('/').filter(segment => segment);
  }

  getUnreadNotificationCount(): number {
    return this.notifications.filter(n => !n.read).length;
  }

  getUserInitials(): string {
    return this.user.name
      .split(' ')
      .map(name => name.charAt(0))
      .join('')
      .toUpperCase();
  }
}
```

```typescript
import { Component, Input, Output, EventEmitter, OnInit, OnDestroy, ChangeDetectionStrategy, ElementRef, ViewChild } from '@angular/core';

@Component({
  selector: 'app-modal',
  templateUrl: './modal.component.html',
  styleUrls: ['./modal.component.scss'],
  changeDetection: ChangeDetectionStrategy.OnPush
})
export class ModalComponent implements OnInit, OnDestroy {
  @Input() isOpen: boolean = false;
  @Input() title: string = '';
  @Input() size: 'sm' | 'md' | 'lg' = 'md';
  @Input() closable: boolean = true;
  @Input() closeOnBackdrop: boolean = true;
  @Input() closeOnEscape: boolean = true;
  
  @Output() closed = new EventEmitter<void>();
  @Output() opened = new EventEmitter<void>();
  @Output() backdropClicked = new EventEmitter<void>();

  @ViewChild('modalDialog', { static: false }) modalDialog!: ElementRef;
  @ViewChild('closeButton', { static: false }) closeButton!: ElementRef;

  isAnimating: boolean = false;
  private previousActiveElement: HTMLElement | null = null;

  ngOnInit(): void {
    if (this.closeOnEscape) {
      document.addEventListener('keydown', this.handleEscapeKey.bind(this));
    }
  }

  ngOnDestroy(): void {
    document.removeEventListener('keydown', this.handleEscapeKey.bind(this));
    this.restoreFocus();
  }

  ngOnChanges(): void {
    if (this.isOpen) {
      this.openModal();
    } else {
      this.closeModal();
    }
  }

  private openModal(): void {
    this.isAnimating = true;
    this.previousActiveElement = document.activeElement as HTMLElement;
    
    // Prevent body scroll
    document.body.style.overflow = 'hidden';
    
    // Focus management
    setTimeout(() => {
      if (this.closeButton) {
        this.closeButton.nativeElement.focus();
      }
      this.isAnimating = false;
    }, 150);
    
    this.opened.emit();
  }

  private closeModal(): void {
    this.isAnimating = true;
    
    // Restore body scroll
    document.body.style.overflow = '';
    
    setTimeout(() => {
      this.isAnimating = false;
      this.restoreFocus();
    }, 150);
    
    this.closed.emit();
  }

  private restoreFocus(): void {
    if (this.previousActiveElement) {
      this.previousActiveElement.focus();
      this.previousActiveElement = null;
    }
  }

  private handleEscapeKey(event: KeyboardEvent): void {
    if (event.key === 'Escape' && this.isOpen && this.closable) {
      this.close();
    }
  }

  onBackdropClick(event: MouseEvent): void {
    if (event.target === event.currentTarget && this.closeOnBackdrop && this.closable) {
      this.backdropClicked.emit();
      this.close();
    }
  }

  close(): void {
    if (this.closable) {
      this.closed.emit();
    }
  }

  onDialogClick(event: MouseEvent): void {
    event.stopPropagation();
  }

  // Trap focus within modal
  onKeyDown(event: KeyboardEvent): void {
    if (event.key === 'Tab') {
      this.trapFocus(event);
    }
  }

  private trapFocus(event: KeyboardEvent): void {
    if (!this.modalDialog) return;
    
    const focusableElements = this.modalDialog.nativeElement.querySelectorAll(
      'button, [href], input, select, textarea, [tabindex]:not([tabindex="-1"])'
    );
    
    const firstElement = focusableElements[0] as HTMLElement;
    const lastElement = focusableElements[focusableElements.length - 1] as HTMLElement;
    
    if (event.shiftKey) {
      if (document.activeElement === firstElement) {
        lastElement.focus();
        event.preventDefault();
      }
    } else {
      if (document.activeElement === lastElement) {
        firstElement.focus();
        event.preventDefault();
      }
    }
  }
}
```
