```typescript
import { Component, Input, Output, EventEmitter, OnInit, OnDestroy } from '@angular/core';

  title: string;
  tasks: Task[];
  maxTasks?: number;

export interface Task {
  id: string;
  title: string;
  description?: string;
  assignee?: User;
  priority: 'low' | 'medium' | 'high';
  dueDate?: Date;
  tags?: string[];
  columnId: string;
  createdAt: Date;
  updatedAt: Date;
}

  name: string;
  email: string;

@Component({
  selector: 'app-kanban-board',
  templateUrl: './kanban-board.component.html',
  styleUrls: ['./kanban-board.component.scss']
})
export class KanbanBoardComponent implements OnInit, OnDestroy {
  @Input() boardId: string = '';
  @Input() initialColumns: KanbanColumn[] = [];
  @Input() readonly: boolean = false;
  @Input() loading: boolean = false;
  @Input() error: string | null = null;

  @Output() taskMoved = new EventEmitter<{taskId: string, fromColumn: string, toColumn: string}>();
  @Output() taskCreated = new EventEmitter<Task>();
  @Output() taskUpdated = new EventEmitter<Task>();
  @Output() taskDeleted = new EventEmitter<string>();
  @Output() columnUpdated = new EventEmitter<KanbanColumn>();

  columns: KanbanColumn[] = [];
  draggedTask: Task | null = null;
  isLoading: boolean = false;
  errorMessage: string | null = null;

  private destroy$ = new Subject<void>();

  ngOnInit(): void {
    this.initializeColumns();
    this.loadBoardData();
  }

  ngOnDestroy(): void {
    this.destroy$.next();
    this.destroy$.complete();
  }

  private initializeColumns(): void {
    if (this.initialColumns.length > 0) {
      this.columns = [...this.initialColumns];
    } else {
      this.columns = [
        { id: 'todo', title: 'To Do', tasks: [] },
        { id: 'inprogress', title: 'In Progress', tasks: [] },
        { id: 'done', title: 'Done', tasks: [] }
      ];
    }
  }

  private loadBoardData(): void {
    // API integration would go here
    // GET /forms/sessions/{sessionId} to load board state
  }

  onTaskDrop(event: CdkDragDrop<Task[]>): void {
    if (this.readonly) return;

    if (event.previousContainer === event.container) {
      moveItemInArray(event.container.data, event.previousIndex, event.currentIndex);
    } else {
      const task = event.previousContainer.data[event.previousIndex];
      const fromColumnId = this.getColumnIdFromContainer(event.previousContainer);
      const toColumnId = this.getColumnIdFromContainer(event.container);

      transferArrayItem(
        event.previousContainer.data,
        event.container.data,
        event.previousIndex,
        event.currentIndex
      );

      // Update task column reference
      task.columnId = toColumnId;
      task.updatedAt = new Date();

      // Emit task moved event
      this.taskMoved.emit({
        taskId: task.id,
        fromColumn: fromColumnId,
        toColumn: toColumnId
      });
    }
  }

  private getColumnIdFromContainer(container: any): string {
    // Extract column ID from container element or data
    return container.element.nativeElement.getAttribute('data-column-id') || '';
  }

  onTaskCreate(columnId: string): void {
    // Emit event to open task creation modal
    const newTask: Partial<Task> = {
      columnId: columnId,
      priority: 'medium',
      createdAt: new Date(),
      updatedAt: new Date()
    };
    // This would typically open a modal or navigate to creation form
  }

  onTaskEdit(task: Task): void {
    // Emit event to open task edit modal
    this.taskUpdated.emit(task);
  }

  onTaskDelete(taskId: string): void {
    // Remove task from columns and emit delete event
    this.columns.forEach(column => {
      const taskIndex = column.tasks.findIndex(task => task.id === taskId);
      if (taskIndex > -1) {
        column.tasks.splice(taskIndex, 1);
      }
    });
    this.taskDeleted.emit(taskId);
  }

  getColumnTasks(columnId: string): Task[] {
    const column = this.columns.find(col => col.id === columnId);
    return column ? column.tasks : [];
  }

  getConnectedDropLists(): string[] {
    return this.columns.map(column => column.id);
  }
}
```

### 3.2 KanbanColumnComponent (Feature)

**Purpose:** Manages individual kanban column display, task list rendering, and drop zone functionality.

**TypeScript Specification:**
```typescript
// kanban-column.component.ts
import { Component, Input, Output, EventEmitter, OnInit } from '@angular/core';
import { CdkDragDrop } from '@angular/cdk/drag-drop';

@Component({
  selector: 'app-kanban-column',
  templateUrl: './kanban-column.component.html',
  styleUrls: ['./kanban-column.component.scss']
})
export class KanbanColumnComponent implements OnInit {
  @Input() column!: KanbanColumn;
  @Input() tasks: Task[] = [];
  @Input() allowDrop: boolean = true;
  @Input() readonly: boolean = false;
  @Input() connectedDropLists: string[] = [];

  @Output() taskDrop = new EventEmitter<CdkDragDrop<Task[]>>();
  @Output() addTask = new EventEmitter<string>();
  @Output() taskClick = new EventEmitter<Task>();
  @Output() taskEdit = new EventEmitter<Task>();
  @Output() taskDelete = new EventEmitter<string>();

  isDragOver: boolean = false;
  taskCount: number = 0;

  ngOnInit(): void {
    this.updateTaskCount();
  }

  ngOnChanges(): void {
    this.updateTaskCount();
  }

  private updateTaskCount(): void {
    this.taskCount = this.tasks.length;
  }

  onTaskDrop(event: CdkDragDrop<Task[]>): void {
    if (!this.allowDrop || this.readonly) return;
    this.taskDrop.emit(event);
    this.isDragOver = false;
  }

  onDragEnter(): void {
    if (this.allowDrop && !this.readonly) {
      this.isDragOver = true;
    }
  }

  onDragLeave(): void {
    this.isDragOver = false;
  }

  onAddTaskClick(): void {
    if (!this.readonly) {
      this.addTask.emit(this.column.id);
    }
  }

  onTaskCardClick(task: Task): void {
    this.taskClick.emit(task);
  }

  onTaskCardEdit(task: Task): void {
    this.taskEdit.emit(task);
  }

  onTaskCardDelete(taskId: string): void {
    this.taskDelete.emit(taskId);
  }

  trackByTaskId(index: number, task: Task): string {
    return task.id;
  }

  getColumnColorClass(): string {
    return this.column.color ? `column-${this.column.color}` : 'column-default';
  }

  isMaxTasksReached(): boolean {
    return this.column.maxTasks ? this.taskCount >= this.column.maxTasks : false;
  }
}
```

### 3.3 TaskCardComponent (Feature)

**Purpose:** Displays individual task information with metadata, priority indicators, assignee avatars, and interaction capabilities.

**TypeScript Specification:**
```typescript
// task-card.component.ts
import { Component, Input, Output, EventEmitter, OnInit } from '@angular/core';

@Component({
  selector: 'app-task-card',
  templateUrl: './task-card.component.html',
  styleUrls: ['./task-card.component.scss']
})
export class TaskCardComponent implements OnInit {
  @Input() task!: Task;
  @Input() draggable: boolean = true;
  @Input() clickable: boolean = true;
  @Input() compact: boolean = false;
  @Input() showActions: boolean = true;

  @Output() taskClick = new EventEmitter<Task>();
  @Output() taskEdit = new EventEmitter<Task>();
  @Output() taskDelete = new EventEmitter<string>();
  @Output() assigneeClick = new EventEmitter<User>();
  @Output() tagClick = new EventEmitter<string>();

  isDragging: boolean = false;
  isSelected: boolean = false;
  showActionsMenu: boolean = false;

  ngOnInit(): void {
    // Initialize component
  }

  onCardClick(event: Event): void {
    if (!this.clickable) return;
    
    // Prevent click when clicking on actions or interactive elements
    const target = event.target as HTMLElement;
    if (target.closest('.task-actions') || target.closest('.assignee-avatar') || target.closest('.tag-item')) {
      return;
    }
    
    this.taskClick.emit(this.task);
  }

  onEditClick(event: Event): void {
    event.stopPropagation();
    this.taskEdit.emit(this.task);
    this.showActionsMenu = false;
  }

  onDeleteClick(event: Event): void {
    event.stopPropagation();
    if (confirm(`Are you sure you want to delete "${this.task.title}"?`)) {
      this.taskDelete.emit(this.task.id);
    }
    this.showActionsMenu = false;
  }

  onAssigneeClick(event: Event): void {
    event.stopPropagation();
    if (this.task.assignee) {
      this.assigneeClick.emit(this.task.assignee);
    }
  }

  onTagClick(event: Event, tag: string): void {
    event.stopPropagation();
    this.tagClick.emit(tag);
  }

  onDragStart(): void {
    this.isDragging = true;
  }

  onDragEnd(): void {
    this.isDragging = false;
  }

  toggleActionsMenu(event: Event): void {
    event.stopPropagation();
    this.showActionsMenu = !this.showActionsMenu;
  }

  getPriorityClass(): string {
    return `priority-${this.task.priority}`;
  }

  getPriorityIcon(): string {
    switch (this.task.priority) {
      case 'high': return '🔴';
      case 'medium': return '🟡';
      case 'low': return '🟢';
      default: return '⚪';
    }
  }

  getPriorityLabel(): string {
    return this.task.priority.charAt(0).toUpperCase() + this.task.priority.slice(1);
  }

  getAssigneeInitials(): string {
    if (!this.task.assignee) return '';
    const names = this.task.assignee.name.split(' ');
    return names.map(name => name.charAt(0)).join('').toUpperCase().slice(0, 2);
  }

  isOverdue(): boolean {
    if (!this.task.dueDate) return false;
    return new Date(this.task.dueDate) < new Date();
  }

  getDueDateLabel(): string {
    if (!this.task.dueDate) return '';
    
    const dueDate = new Date(this.task.dueDate);
    const today = new Date();
    const diffTime = dueDate.getTime() - today.getTime();
    const diffDays = Math.ceil(diffTime / (1000 * 60 * 60 * 24));
    
    if (diffDays < 0) {
      return `${Math.abs(diffDays)} days overdue`;
    } else if (diffDays === 0) {
      return 'Due today';
    } else if (diffDays === 1) {
      return 'Due tomorrow';
    } else {
      return `Due in ${diffDays} days`;
    }
  }

  formatDate(date: Date): string {
    return new Intl.DateTimeFormat('en-US', {
      month: 'short',
      day: 'numeric'
    }).format(new Date(date));
  }
}
```

### 3.4 TaskCreationModalComponent (Feature)

**Purpose:** Provides a modal interface for creating new tasks with form validation, assignee selection, and advanced options.

**TypeScript Specification:**
```typescript
// task-creation-modal.component.ts
import { Component, Input, Output, EventEmitter, OnInit, OnDestroy } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { Subject } from 'rxjs';
import { takeUntil } from 'rxjs/operators';

export interface TaskFormData {
  title: string;
  description: string;
  assigneeId: string;
  priority: 'low' | 'medium' | 'high';
  dueDate: string;
  tags: string[];
  columnId: string;
}

export interface ValidationError {
  field: string;
  message: string;
}

@Component({
  selector: 'app-task-creation-modal',
  templateUrl: './task-creation-modal.component.html',
  styleUrls: ['./task-creation-modal.component.scss']
})
export class TaskCreationModalComponent implements OnInit, OnDestroy {
  @Input() isOpen: boolean = false;
  @Input() initialData: Partial<Task> = {};
  @Input() availableAssignees: User[] = [];
  @Input() defaultColumnId: string = '';
  @Input() availableColumns: KanbanColumn[] = [];

  @Output() taskSubmit = new EventEmitter<TaskFormData>();
  @Output() modalCancel = new EventEmitter<void>();
  @Output() modalClose = new EventEmitter<void>();

  taskForm!: FormGroup;
  isSubmitting: boolean = false;
  validationErrors: ValidationError[] = [];
  showAdvancedOptions: boolean = false;
  tagInput: string = '';
  selectedTags: string[] = [];

  priorityOptions = [
    { value: 'low', label: 'Low', icon: '🟢', color: '#10b981' },
    { value: 'medium', label: 'Medium', icon: '🟡', color: '#f59e0b' },
    { value: 'high', label: 'High', icon: '🔴', color: '#dc2626' }
  ];

  private destroy$ = new Subject<void>();

  constructor(private formBuilder: FormBuilder) {}

  ngOnInit(): void {
    this.initializeForm();
    this.setupFormValidation();
  }

  ngOnDestroy(): void {
    this.destroy$.next();
    this.destroy$.complete();
  }

  ngOnChanges(): void {
    if (this.isOpen && this.taskForm) {
      this.resetForm();
      this.populateInitialData();
    }
  }

  private initializeForm(): void {
    this.taskForm = this.formBuilder.group({
      title: ['', [Validators.required, Validators.minLength(3), Validators.maxLength(100)]],
      description: ['', [Validators.maxLength(500)]],
      assigneeId: [''],
      priority: ['medium', Validators.required],
      dueDate: [''],
      columnId: [this.defaultColumnId, Validators.required]
    });
  }

  private setupFormValidation(): void {
    this.taskForm.valueChanges
      .pipe(takeUntil(this.destroy$))
      .subscribe(() => {
        this.clearValidationErrors();
      });
  }

  private populateInitialData(): void {
    if (this.initialData) {
      this.taskForm.patchValue({
        title: this.initialData.title || '',
        description: this.initialData.description || '',
        assigneeId: this.initialData.assignee?.id || '',
        priority: this.initialData.priority || 'medium',
        dueDate: this.initialData.dueDate ? this.formatDateForInput(this.initialData.dueDate) : '',
        columnId: this.initialData.columnId || this.defaultColumnId
      });
      
      this.selectedTags = this.initialData.tags ? [...this.initialData.tags] : [];
    }
  }

  private resetForm(): void {
    this.taskForm.reset();
    this.selectedTags = [];
    this.tagInput = '';
    this.validationErrors = [];
    this.isSubmitting = false;
    this.showAdvancedOptions = false;
  }

  onSubmit(): void {
    if (this.taskForm.invalid || this.isSubmitting) {
      this.markFormGroupTouched();
      return;
    }

    this.isSubmitting = true;
    this.clearValidationErrors();

    const formData: TaskFormData = {
      ...this.taskForm.value,
      tags: this.selectedTags
    };

    // Simulate API call delay
    setTimeout(() => {
      this.taskSubmit.emit(formData);
      this.isSubmitting = false;
    }, 1000);
  }

  onCancel(): void {
    if (this.hasUnsavedChanges()) {
      if (confirm('You have unsaved changes. Are you sure you want to cancel?')) {
        this.modalCancel.emit();
        this.resetForm();
      }
    } else {
      this.modalCancel.emit();
      this.resetForm();
    }
  }

  onClose(): void {
    this.modalClose.emit();
    this.resetForm();
  }

  onBackdropClick(event: Event): void {
    if (event.target === event.currentTarget) {
      this.onCancel();
    }
  }

  toggleAdvancedOptions(): void {
    this.showAdvancedOptions = !this.showAdvancedOptions;
  }

  addTag(): void {
    const tag = this.tagInput.trim().toLowerCase();
    if (tag && !this.selectedTags.includes(tag) && this.selectedTags.length < 5) {
      this.selectedTags.push(tag);
      this.tagInput = '';
    }
  }

  removeTag(tag: string): void {
    this.selectedTags = this.selectedTags.filter(t => t !== tag);
  }

  onTagInputKeydown(event: KeyboardEvent): void {
    if (event.key === 'Enter') {
      event.preventDefault();
      this.addTag();
    } else if (event.key === 'Backspace' && !this.tagInput && this.selectedTags.length > 0) {
      this.removeTag(this.selectedTags[this.selectedTags.length - 1]);
    }
  }

  getFieldError(fieldName: string): string {
    const field = this.taskForm.get(fieldName);
    if (field && field.invalid && field.touched) {
      const errors = field.errors;
      if (errors) {
        if (errors['required']) return `${this.getFieldLabel(fieldName)} is required`;
        if (errors['minlength']) return `${this.getFieldLabel(fieldName)} must be at least ${errors['minlength'].requiredLength} characters`;
        if (errors['maxlength']) return `${this.getFieldLabel(fieldName)} must not exceed ${errors['maxlength'].requiredLength} characters`;
      }
    }
    return '';
  }

  private getFieldLabel(fieldName: string): string {
    const labels: { [key: string]: string } = {
      title: 'Title',
      description: 'Description',
      assigneeId: 'Assignee',
      priority: 'Priority',
      dueDate: 'Due Date',
      columnId: 'Column'
    };
    return labels[fieldName] || fieldName;
  }

  private markFormGroupTouched(): void {
    Object.keys(this.taskForm.controls).forEach(key => {
      this.taskForm.get(key)?.markAsTouched();
    });
  }

  private clearValidationErrors(): void {
    this.validationErrors = [];
  }

  private hasUnsavedChanges(): boolean {
    return this.taskForm.dirty || this.selectedTags.length > 0;
  }

  private formatDateForInput(date: Date): string {
    return new Date(date).toISOString().split('T')[0];
  }

  getAssigneeName(assigneeId: string): string {
    const assignee = this.availableAssignees.find(a => a.id === assigneeId);
    return assignee ? assignee.name : 'Unassigned';
  }

  getPriorityOption(value: string) {
    return this.priorityOptions.find(option => option.value === value);
  }
}
```

```typescript
// header.component.ts
import { Component, Input, Output, EventEmitter, OnInit, OnDestroy } from '@angular/core';
import { FormControl } from '@angular/forms';
import { Subject } from 'rxjs';
import { debounceTime, distinctUntilChanged, takeUntil } from 'rxjs/operators';

export interface Notification {
  id: string;
  title: string;
  message: string;
  type: 'info' | 'success' | 'warning' | 'error';
  read: boolean;
  createdAt: Date;
  actionUrl?: string;
}

@Component({
  selector: 'app-header',
  templateUrl: './header.component.html',
  styleUrls: ['./header.component.scss']
})
export class HeaderComponent implements OnInit, OnDestroy {
  @Input() user!: User;
  @Input() notifications: Notification[] = [];
  @Input() showSearch: boolean = true;
  @Input() searchPlaceholder: string = 'Search tasks, projects, or people...';

  @Output() searchQuery = new EventEmitter<string>();
  @Output() notificationClick = new EventEmitter<Notification>();
  @Output() notificationMarkRead = new EventEmitter<string>();
  @Output() notificationClearAll = new EventEmitter<void>();
  @Output() profileClick = new EventEmitter<void>();
  @Output() settingsClick = new EventEmitter<void>();
  @Output() logoutClick = new EventEmitter<void>();

  searchControl = new FormControl('');
  showNotifications: boolean = false;
  showProfileMenu: boolean = false;
  searchFocused: boolean = false;

  private destroy$ = new Subject<void>();

  ngOnInit(): void {
    this.setupSearchSubscription();
  }

  ngOnDestroy(): void {
    this.destroy$.next();
    this.destroy$.complete();
  }

  private setupSearchSubscription(): void {
    this.searchControl.valueChanges
      .pipe(
        debounceTime(300),
        distinctUntilChanged(),
        takeUntil(this.destroy$)
      )
      .subscribe(query => {
        this.searchQuery.emit(query || '');
      });
  }

  onSearchFocus(): void {
    this.searchFocused = true;
  }

  onSearchBlur(): void {
    this.searchFocused = false;
  }

  clearSearch(): void {
    this.searchControl.setValue('');
    this.searchQuery.emit('');
  }

  toggleNotifications(): void {
    this.showNotifications = !this.showNotifications;
    this.showProfileMenu = false;
  }

  toggleProfileMenu(): void {
    this.showProfileMenu = !this.showProfileMenu;
    this.showNotifications = false;
  }

  closeDropdowns(): void {
    this.showNotifications = false;
    this.showProfileMenu = false;
  }

  onNotificationItemClick(notification: Notification): void {
    this.notificationClick.emit(notification);
    if (!notification.read) {
      this.notificationMarkRead.emit(notification.id);
    }
  }

  onMarkAllRead(): void {
    const unreadIds = this.notifications
      .filter(n => !n.read)
      .map(n => n.id);
    
    unreadIds.forEach(id => {
      this.notificationMarkRead.emit(id);
    });
  }

  onClearAllNotifications(): void {
    if (confirm('Are you sure you want to clear all notifications?')) {
      this.notificationClearAll.emit();
    }
  }

  onProfileMenuClick(): void {
    this.profileClick.emit();
    this.closeDropdowns();
  }

  onSettingsClick(): void {
    this.settingsClick.emit();
    this.closeDropdowns();
  }

  onLogoutClick(): void {
    if (confirm('Are you sure you want to log out?')) {
      this.logoutClick.emit();
    }
    this.closeDropdowns();
  }

  getUnreadNotificationCount(): number {
    return this.notifications.filter(n => !n.read).length;
  }

  getUserInitials(): string {
    if (!this.user?.name) return '';
    const names = this.user.name.split(' ');
    return names.map(name => name.charAt(0)).join('').toUpperCase().slice(0, 2);
  }

  getNotificationIcon(type: string): string {
    switch (type) {
      case 'success': return '✅';
      case 'warning': return '⚠️';
      case 'error': return '❌';
      default: return 'ℹ️';
    }
  }

  formatNotificationTime(date: Date): string {
    const now = new Date();
    const diffMs = now.getTime() - new Date(date).getTime();
    const diffMins = Math.floor(diffMs / 60000);
    const diffHours = Math.floor(diffMins / 60);
    const diffDays = Math.floor(diffHours / 24);

    if (diffMins < 1) return 'Just now';
    if (diffMins < 60) return `${diffMins}m ago`;
    if (diffHours < 24) return `${diffHours}h ago`;
    if (diffDays < 7) return `${diffDays}d ago`;
    
    return new Intl.DateTimeFormat('en-US', {
      month: 'short',
      day: 'numeric'
    }).format(new Date(date));
  }
}
```

```typescript
// sidebar.component.ts
import { Component, Input, Output, EventEmitter, OnInit, OnDestroy } from '@angular/core';
import { Subject } from 'rxjs';
import { takeUntil } from 'rxjs/operators';

export interface MenuItem {
  id: string;
  label: string;
  icon: string;
  route?: string;
  badge?: string | number;
  children?: MenuItem[];
  disabled?: boolean;
  divider?: boolean;
}

export interface WorkspaceInfo {
  id: string;
  name: string;
  description?: string;
  avatar?: string;
  memberCount?: number;
}

@Component({
  selector: 'app-sidebar',
  templateUrl: './sidebar.component.html',
  styleUrls: ['./sidebar.component.scss']
})
export class SidebarComponent implements OnInit, OnDestroy {
  @Input() currentRoute: string = '';
  @Input() menuItems: MenuItem[] = [];
  @Input() workspace: WorkspaceInfo | null = null;
  @Input() isCollapsed: boolean = false;
  @Input() isMobile: boolean = false;
  @Input() showOverlay: boolean = false;

  @Output() menuItemClick = new EventEmitter<MenuItem>();
  @Output() toggleCollapse = new EventEmitter<void>();
  @Output() workspaceClick = new EventEmitter<WorkspaceInfo>();
  @Output() overlayClick = new EventEmitter<void>();

  expandedItems: Set<string> = new Set();
  selectedItem: string = '';

  private destroy$ = new Subject<void>();

  // Default menu items if none provided
  defaultMenuItems: MenuItem[] = [
    {
      id: 'dashboard',
      label: 'Dashboard',
      icon: '📊',
      route: '/dashboard'
    },
    {
      id: 'kanban',
      label: 'Kanban Board',
      icon: '📋',
      route: '/kanban'
    },
    {
      id: 'tasks',
      label: 'Tasks',
      icon: '✅',
      route: '/tasks',
      badge: '12'
    },
    {
      id: 'projects',
      label: 'Projects',
      icon: '📁',
      children: [
        {
          id: 'active-projects',
          label: 'Active Projects',
          icon: '🟢',
          route: '/projects/active'
        },
        {
          id: 'archived-projects',
          label: 'Archived',
          icon: '📦',
          route: '/projects/archived'
        }
      ]
    },
    {
      id: 'team',
      label: 'Team',
      icon: '👥',
      route: '/team'
    },
    {
      id: 'divider-1',
      label: '',
      icon: '',
      divider: true
    },
    {
      id: 'analytics',
      label: 'Analytics',
      icon: '📈',
      route: '/analytics'
    },
    {
      id: 'settings',
      label: 'Settings',
      icon: '⚙️',
      route: '/settings'
    }
  ];

  ngOnInit(): void {
    this.initializeMenuItems();
    this.updateSelectedItem();
  }

  ngOnDestroy(): void {
    this.destroy$.next();
    this.destroy$.complete();
  }

  ngOnChanges(): void {
    this.updateSelectedItem();
  }

  private initializeMenuItems(): void {
    if (this.menuItems.length === 0) {
      this.menuItems = this.defaultMenuItems;
    }
  }

  private updateSelectedItem(): void {
    // Find the menu item that matches the current route
    const findSelectedItem = (items: MenuItem[]): string => {
      for (const item of items) {
        if (item.route === this.currentRoute) {
          return item.id;
        }
        if (item.children) {
          const childMatch = findSelectedItem(item.children);
          if (childMatch) {
            this.expandedItems.add(item.id);
            return childMatch;
          }
        }
      }
      return '';
    };

    this.selectedItem = findSelectedItem(this.menuItems);
  }

  onToggleCollapse(): void {
    this.toggleCollapse.emit();
  }

  onMenuItemClick(item: MenuItem, event?: Event): void {
    if (event) {
      event.preventDefault();
    }

    if (item.disabled || item.divider) {
      return;
    }

    // Handle parent items with children
    if (item.children && item.children.length > 0) {
      this.toggleItemExpansion(item.id);
      return;
    }

    // Handle regular menu items
    this.selectedItem = item.id;
    this.menuItemClick.emit(item);

    // Close sidebar on mobile after selection
    if (this.isMobile && this.showOverlay) {
      this.overlayClick.emit();
    }
  }

  onWorkspaceClick(): void {
    if (this.workspace) {
      this.workspaceClick.emit(this.workspace);
    }
  }

  onOverlayClick(): void {
    this.overlayClick.emit();
  }

  toggleItemExpansion(itemId: string): void {
    if (this.expandedItems.has(itemId)) {
      this.expandedItems.delete(itemId);
    } else {
      this.expandedItems.add(itemId);
    }
  }

  isItemExpanded(itemId: string): boolean {
    return this.expandedItems.has(itemId);
  }

  isItemSelected(itemId: string): boolean {
    return this.selectedItem === itemId;
  }

  isItemActive(item: MenuItem): boolean {
    if (item.route === this.currentRoute) {
      return true;
    }
    
    // Check if any child is active
    if (item.children) {
      return item.children.some(child => child.route === this.currentRoute);
    }
    
    return false;
  }

  getWorkspaceInitials(): string {
    if (!this.workspace?.name) return 'W';
    const names = this.workspace.name.split(' ');
    return names.map(name => name.charAt(0)).join('').toUpperCase().slice(0, 2);
  }

  trackByMenuItem(index: number, item: MenuItem): string {
    return item.id;
  }

  getItemAriaLabel(item: MenuItem): string {
    let label = item.label;
    
    if (item.badge) {
      label += `, ${item.badge} items`;
    }
    
    if (item.children && item.children.length > 0) {
      const expanded = this.isItemExpanded(item.id);
      label += `, ${expanded ? 'expanded' : 'collapsed'} submenu`;
    }
    
    if (this.isItemActive(item)) {
      label += ', current page';
    }
    
    return label;
  }
}
```

### Issues Found:

#### HIGH SEVERITY

**Missing Component Implementations in Specifications**
- **Description:** Several components defined in architecture are not implemented in specifications: AddTaskButtonComponent, BaseModalComponent, TaskDetailModalComponent, FilterFormComponent, and multiple shared components.
- **Impact:** Incomplete UI implementation will result in broken functionality and missing user interactions.
- **Affected Components:** AddTaskButtonComponent, BaseModalComponent, TaskDetailModalComponent, FilterFormComponent, LoadingSpinnerComponent, ErrorMessageComponent, DropZoneComponent

**API Endpoint Mismatch**
- **Description:** Architecture specifies dedicated kanban endpoints (/kanban/boards, /kanban/tasks) but specifications use form session endpoints (/forms/sessions).
- **Impact:** API integration will fail due to endpoint mismatch, causing data loading and persistence issues.
- **Affected Components:** KanbanBoardComponent, TaskCardComponent, TaskCreationModalComponent

#### MEDIUM SEVERITY

**Incomplete Drag-and-Drop Implementation**
- **Description:** Architecture defines comprehensive drag-and-drop state management but specifications lack detailed implementation of drag events and state handling.
- **Impact:** Core kanban functionality may not work properly without proper drag-and-drop implementation.
- **Affected Components:** KanbanBoardComponent, KanbanColumnComponent, TaskCardComponent

**Missing Error Handling Components**
- **Description:** Architecture defines ErrorMessageComponent and LoadingSpinnerComponent but these are not implemented in specifications.
- **Impact:** Poor user experience during loading states and error conditions.
- **Affected Components:** ErrorMessageComponent, LoadingSpinnerComponent

#### LOW SEVERITY

**Inconsistent Naming Conventions**
- **Description:** Some component names differ between architecture and specifications (e.g., IconButtonComponent vs icon-button usage).
- **Impact:** Minor development confusion and potential import issues.
- **Affected Components:** IconButtonComponent, various shared components

### Recommendations:

1. **Implement Missing Components:** Create specifications for AddTaskButtonComponent, BaseModalComponent, TaskDetailModalComponent, FilterFormComponent, and all shared components defined in architecture

2. **Align API Integration:** Either update architecture to use /forms/sessions endpoints or modify specifications to use dedicated /kanban endpoints as defined in architecture

3. **Complete Drag-and-Drop Implementation:** Add comprehensive drag-and-drop state management, event handlers, and visual feedback across KanbanBoardComponent, KanbanColumnComponent, and TaskCardComponent
