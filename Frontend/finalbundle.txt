```typescript
@Component({
  selector: 'app-task-card',
  templateUrl: './task-card.component.html',
  styleUrls: ['./task-card.component.scss']
})
export class TaskCardComponent implements OnInit {
  @Input() task: Task;
  @Input() draggable: boolean = true;

  @Output() taskClick = new EventEmitter<Task>();
  @Output() dragStart = new EventEmitter<Task>();

  isDragging: boolean = false;
  isHovered: boolean = false;

  ngOnInit() {
    // Initialize component
  }

  onTaskClick() {
    this.taskClick.emit(this.task);
  }

  onDragStart() {
    this.isDragging = true;
    this.dragStart.emit(this.task);
  }

  onDragEnd() {
    this.isDragging = false;
  }

  onMouseEnter() {
    this.isHovered = true;
  }

  onMouseLeave() {
    this.isHovered = false;
  }

  getPriorityVariant(priority: string): string {
    const variants = {
      'low': 'info',
      'medium': 'warning',
      'high': 'danger',
      'urgent': 'danger'
    };
    return variants[priority] || 'secondary';
  }

  isOverdue(dueDate: string): boolean {
    return new Date(dueDate) < new Date();
  }

  isDueSoon(dueDate: string): boolean {
    const due = new Date(dueDate);
    const now = new Date();
    const diffDays = (due.getTime() - now.getTime()) / (1000 * 3600 * 24);
    return diffDays <= 3 && diffDays > 0;
  }
}
```

### 3.4 TaskModalComponent

**Component Type:** Feature
**Purpose:** Provides form interface for task creation and editing with validation

**TypeScript Specification:**

```typescript
@Component({
  selector: 'app-task-modal',
  templateUrl: './task-modal.component.html',
  styleUrls: ['./task-modal.component.scss']
})
export class TaskModalComponent implements OnInit, OnChanges {
  @Input() isOpen: boolean = false;
  @Input() task: Task | null = null;
  @Input() defaultColumn: string = 'todo';

  @Output() submit = new EventEmitter<Task>();
  @Output() close = new EventEmitter<void>();

  taskForm: FormGroup;
  isSubmitting: boolean = false;
  validationErrors: ValidationError[] = [];
  availableAssignees: User[] = [];

  constructor(private fb: FormBuilder) {
    this.taskForm = this.fb.group({
      title: ['', [Validators.required, Validators.minLength(3)]],
      description: [''],
      priority: [''],
      assigneeId: [''],
      dueDate: [''],
      tags: ['']
    });
  }

  ngOnInit() {
    this.loadAssignees();
  }

  ngOnChanges(changes: SimpleChanges) {
    if (changes['task'] && this.task) {
      this.populateForm();
    }
    if (changes['isOpen'] && this.isOpen) {
      this.resetForm();
    }
  }

  onSubmit() {
    if (this.taskForm.valid) {
      this.isSubmitting = true;
      const formData = this.taskForm.value;
      const task: Task = {
        ...formData,
        id: this.task?.id || this.generateId(),
        tags: formData.tags ? formData.tags.split(',').map(tag => tag.trim()) : [],
        column: this.defaultColumn,
        createdAt: this.task?.createdAt || new Date().toISOString(),
        updatedAt: new Date().toISOString()
      };
      this.submit.emit(task);
    } else {
      this.validateForm();
    }
  }

  onCancel() {
    this.close.emit();
  }

  private loadAssignees() {
    // Load available assignees from API
  }

  private populateForm() {
    if (this.task) {
      this.taskForm.patchValue({
        title: this.task.title,
        description: this.task.description,
        priority: this.task.priority,
        assigneeId: this.task.assignee?.id,
        dueDate: this.task.dueDate,
        tags: this.task.tags?.join(', ')
      });
    }
  }

  private resetForm() {
    this.taskForm.reset();
    this.validationErrors = [];
    this.isSubmitting = false;
  }

  private validateForm() {
    this.validationErrors = [];
    Object.keys(this.taskForm.controls).forEach(key => {
      const control = this.taskForm.get(key);
      if (control && control.invalid) {
        control.markAsTouched();
      }
    });
  }

  private generateId(): string {
    return 'task_' + Date.now() + '_' + Math.random().toString(36).substr(2, 9);
  }
}
```

```typescript
@Component({

export class HeaderComponent implements OnInit {
  @Input() user: User;

  @Output() search = new EventEmitter<string>();
  @Output() themeToggle = new EventEmitter<void>();
  @Output() notificationClick = new EventEmitter<void>();
  @Output() menuToggle = new EventEmitter<void>();

  searchQuery: string = '';
  theme: 'light' | 'dark' = 'light';
  notificationCount: number = 0;
  isMobileMenuOpen: boolean = false;

    this.notificationCount = this.notifications?.filter(n => !n.read).length || 0;
  }

  onSearchInput(query: string) {
    this.searchQuery = query;
    this.search.emit(query);
  }

  onThemeToggle() {
    this.theme = this.theme === 'light' ? 'dark' : 'light';
    this.themeToggle.emit();
  }

  onNotificationClick() {
    this.notificationClick.emit();
  }

  onMobileMenuToggle() {
    this.isMobileMenuOpen = !this.isMobileMenuOpen;
    this.menuToggle.emit();
  }

  getBreadcrumbs() {
    // Generate breadcrumb navigation based on current route
    const routeMap = {

}
```

```typescript
  @Output() toggle = new EventEmitter<void>();
  @Output() projectSelect = new EventEmitter<Project>();

  isCollapsed: boolean = false;
  expandedSections: string[] = ['main'];
  isMobile: boolean = false;

  ngOnInit() {
    this.checkMobile();
  }

  onNavigate(route: string) {
    this.navigate.emit(route);
  }

  onToggle() {
    if (this.isMobile) {
      this.toggle.emit();
    } else {
      this.isCollapsed = !this.isCollapsed;
    }
  }

  toggleSection(section: string) {
    const index = this.expandedSections.indexOf(section);
    if (index > -1) {
      this.expandedSections.splice(index, 1);
    } else {
      this.expandedSections.push(section);
    }
  }

  isRouteActive(route: string): boolean {
    return this.activeRoute === route;
  }

  isSectionExpanded(section: string): boolean {
    return this.expandedSections.includes(section);
  }

  onAddProject() {
    // Emit event to open project creation modal
  }

  private checkMobile() {
    this.isMobile = window.innerWidth <= 768;
  }
}
```
