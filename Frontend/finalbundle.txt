- Purpose: Top navigation and search

- Purpose: Task creation and editing modal

```typescript
@Component({
  selector: 'app-kanban-column',
  templateUrl: './kanban-column.component.html',
  styleUrls: ['./kanban-column.component.scss']
})
export class KanbanColumnComponent {
  @Input() column: KanbanColumn = {} as KanbanColumn;
  @Input() tasks: Task[] = [];
  @Input() allowDrop: boolean = true;
  
  @Output() taskDropped = new EventEmitter<CdkDragDrop<Task[]>>();
  @Output() addTaskClicked = new EventEmitter<string>();
  
  isDropTarget: boolean = false;
  
  get taskCount(): number {
    return this.tasks.length;
  }
  
  onDrop(event: CdkDragDrop<Task[]>): void {
    this.taskDropped.emit(event);
  }
  
  onAddTask(): void {
    this.addTaskClicked.emit(this.column.id);
  }
}
```

```typescript
@Component({
  selector: 'app-task-card',
  templateUrl: './task-card.component.html',
  styleUrls: ['./task-card.component.scss']
})
export class TaskCardComponent {
  @Input() task: Task = {} as Task;
  @Input() draggable: boolean = true;
  
  @Output() taskClicked = new EventEmitter<Task>();
  @Output() dragStarted = new EventEmitter<Task>();
  
  isDragging: boolean = false;
  
  onTaskClick(): void {
    this.taskClicked.emit(this.task);
  }
  
  onDragStart(): void {
    this.isDragging = true;
    this.dragStarted.emit(this.task);
  }
  
  onDragEnd(): void {
    this.isDragging = false;
  }
}
```

```typescript
@Component({
  selector: 'app-modal',
  templateUrl: './modal.component.html',
  styleUrls: ['./modal.component.scss']
})
export class ModalComponent implements OnInit {
  @Input() isOpen: boolean = false;
  @Input() mode: 'create' | 'edit' = 'create';
  @Input() task?: Task;
  
  @Output() taskSubmitted = new EventEmitter<Task>();
  @Output() modalClosed = new EventEmitter<void>();
  
  formData: TaskFormData = {
    title: '',
    description: '',
    priority: 'medium',
    assignee: '',
    dueDate: null,
    tags: []
  };
  
  isSubmitting: boolean = false;
  
  ngOnInit(): void {
    if (this.mode === 'edit' && this.task) {
      this.formData = { ...this.task };
    }
  }
  
  onSubmit(): void {
    if (this.isSubmitting) return;
    
    this.isSubmitting = true;
    
    // API call to POST /api/youth-accounts/{id}/fund-transfer
    // Transform task data to API format
    
    const taskData: Task = {
      id: this.mode === 'edit' ? this.task!.id : this.generateId(),
      ...this.formData,
      status: this.mode === 'edit' ? this.task!.status : 'todo',
      createdAt: this.mode === 'edit' ? this.task!.createdAt : new Date(),
      updatedAt: new Date()
    };
    
    setTimeout(() => {
      this.taskSubmitted.emit(taskData);
      this.isSubmitting = false;
      this.closeModal();
    }, 1000);
  }
  
  closeModal(): void {
    this.modalClosed.emit();
    this.resetForm();
  }
  
  private resetForm(): void {
    this.formData = {
      title: '',
      description: '',
      priority: 'medium',
      assignee: '',
      dueDate: null,
      tags: []
    };
  }
  
  private generateId(): string {
    return 'task_' + Date.now() + '_' + Math.random().toString(36).substr(2, 9);
  }
}
```
