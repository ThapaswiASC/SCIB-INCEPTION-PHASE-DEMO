### COMPONENT DEFINITIONS

**KanbanBoardComponent:**
- Purpose: Main container for kanban board with three-column layout
- Props: columns: KanbanColumn[], loading: boolean, error: string | null
- State: draggedTask: Task | null, columnData: KanbanColumn[]
- Events: taskMoved: EventEmitter<TaskMoveEvent>, taskCreated: EventEmitter<Task>, taskUpdated: EventEmitter<Task>
- API Binding: GET /api/youth-accounts/{id}/dashboard for initial data

**KanbanColumnComponent:**
- Purpose: Individual column container with tasks
- Props: column: KanbanColumn, tasks: Task[], allowDrop: boolean
- State: isDropTarget: boolean
- Events: taskDropped: EventEmitter<TaskDropEvent>, addTaskClicked: EventEmitter<string>
- API Binding: None (receives data from parent)

**TaskCardComponent:**
- Purpose: Individual task display with drag functionality
- Props: task: Task, draggable: boolean, priority: 'high' | 'medium' | 'low'
- State: isDragging: boolean
- Events: taskClicked: EventEmitter<Task>, taskDragStart: EventEmitter<Task>
- API Binding: None (receives data from parent)

**HeaderComponent:**
- Purpose: Top navigation with search and user menu
- Props: user: User, notifications: Notification[]
- State: searchQuery: string, isUserMenuOpen: boolean
- Events: searchSubmitted: EventEmitter<string>, themeToggled: EventEmitter<void>, sidebarToggled: EventEmitter<void>
- API Binding: None (receives data from parent)

**SidebarComponent:**
- Purpose: Navigation sidebar with menu items
- Props: navigationItems: NavigationItem[], isOpen: boolean
- State: activeRoute: string
- Events: navigationClicked: EventEmitter<string>, sidebarClosed: EventEmitter<void>
- API Binding: None (static navigation data)

**ModalComponent:**
- Purpose: Reusable modal container
- Props: isOpen: boolean, title: string, size: 'sm' | 'md' | 'lg'
- State: None
- Events: modalClosed: EventEmitter<void>
- API Binding: None (content provided via projection)

**TaskFormComponent:**
- Purpose: Form for creating/editing tasks
- Props: task: Task | null, mode: 'create' | 'edit'
- State: formData: TaskFormData, isSubmitting: boolean
- Events: taskSubmitted: EventEmitter<Task>, formCancelled: EventEmitter<void>
- API Binding: POST /api/youth-accounts/{id}/fund-transfer for task creation

### INTERACTION FLOWS

**User Action → UI → API:**

1. **Load Kanban Board**
   → KanbanBoardComponent.ngOnInit()
   → GET /api/youth-accounts/{id}/dashboard
   → Display columns and tasks

2. **Create New Task**
   → Click "Add Task" button
   → Open TaskFormComponent in Modal
   → Fill form and submit
   → POST /api/youth-accounts/{id}/fund-transfer
   → Update kanban board state

3. **Move Task Between Columns**
   → Drag task card
   → Drop in new column
   → PUT /api/youth-accounts/{id}/spending-limit
   → Update task status

4. **View Task Details**
   → Click task card
   → Open task detail modal
   → Display task information

5. **Search Tasks**
   → Type in search bar
   → Filter displayed tasks
   → No API call (client-side filtering)

6. **Toggle Theme**
   → Click theme button
   → Update CSS custom properties
   → Store preference in localStorage

```typescript
@Component({
  selector: 'app-kanban-board',
  templateUrl: './kanban-board.component.html',
  styleUrls: ['./kanban-board.component.scss']
})
export class KanbanBoardComponent implements OnInit {
  @Input() columns: KanbanColumn[] = [];
  @Input() loading: boolean = false;
  @Input() error: string | null = null;
  
  @Output() taskMoved = new EventEmitter<TaskMoveEvent>();
  @Output() taskCreated = new EventEmitter<Task>();
  @Output() taskUpdated = new EventEmitter<Task>();
  
  draggedTask: Task | null = null;
  columnData: KanbanColumn[] = [];
  isMobileView: boolean = false;
  activeTab: string = 'todo';
  
  constructor(
    private kanbanService: KanbanService,
    private breakpointObserver: BreakpointObserver
  ) {}
  
  ngOnInit(): void {
    this.initializeColumns();
    this.setupResponsiveObserver();
    this.loadKanbanData();
  }
  
  initializeColumns(): void {
    this.columnData = [
      {id: 'todo', title: 'To Do', cards: []},
      {id: 'inprogress', title: 'In Progress', cards: []},
      {id: 'done', title: 'Done', cards: []}
    ];
  }
  
  setupResponsiveObserver(): void {
    this.breakpointObserver.observe(['(max-width: 768px)'])
      .subscribe(result => {
        this.isMobileView = result.matches;
      });
  }
  
  loadKanbanData(): void {
    this.loading = true;
    this.kanbanService.getDashboardData()
      .subscribe({
        next: (data) => {
          this.columnData = this.transformApiData(data);
          this.loading = false;
        },
        error: (error) => {
          this.error = 'Failed to load kanban data';
          this.loading = false;
        }
      });
  }
  
  onTaskDragStart(task: Task): void {
    this.draggedTask = task;
  }
  
  onTaskDropped(event: TaskDropEvent): void {
    if (this.draggedTask) {
      this.moveTask(this.draggedTask, event.targetColumn);
      this.draggedTask = null;
    }
  }
  
  moveTask(task: Task, targetColumn: string): void {
    // Implementation for moving task between columns
    this.taskMoved.emit({ task, targetColumn });
  }
  
  onTabChange(tabId: string): void {
    this.activeTab = tabId;
  }
}
```

**TypeScript Specification:**
```typescript
@Component({
  selector: 'app-kanban-column',
  templateUrl: './kanban-column.component.html',
  styleUrls: ['./kanban-column.component.scss']
})
export class KanbanColumnComponent {
  @Input() column!: KanbanColumn;
  @Input() tasks: Task[] = [];
  @Input() allowDrop: boolean = true;
  
  @Output() taskDropped = new EventEmitter<TaskDropEvent>();
  @Output() addTaskClicked = new EventEmitter<string>();
  
  isDropTarget: boolean = false;
  
  onDragOver(event: DragEvent): void {
    if (this.allowDrop) {
      event.preventDefault();
      this.isDropTarget = true;
    }
  }
  
  onDragLeave(event: DragEvent): void {
    this.isDropTarget = false;
  }
  
  onDrop(event: DragEvent): void {
    event.preventDefault();
    this.isDropTarget = false;
    
    if (this.allowDrop) {
      this.taskDropped.emit({
        targetColumn: this.column.id,
        position: this.tasks.length
      });
    }
  }
  
  onAddTask(): void {
    this.addTaskClicked.emit(this.column.id);
  }
  
  trackByTaskId(index: number, task: Task): string {
    return task.id;
  }
}
```

```typescript
@Component({
  selector: 'app-task-card',

  styleUrls: ['./task-card.component.scss']
})
export class TaskCardComponent {
  @Input() task!: Task;
  @Input() draggable: boolean = true;
  @Input() priority: 'high' | 'medium' | 'low' = 'medium';
  
  @Output() taskClicked = new EventEmitter<Task>();
  @Output() taskDragStart = new EventEmitter<Task>();
  
  isDragging: boolean = false;
  
  onDragStart(event: DragEvent): void {
    if (this.draggable) {
      this.isDragging = true;
      this.taskDragStart.emit(this.task);
      
      // Set drag data
      event.dataTransfer?.setData('text/plain', this.task.id);
      event.dataTransfer!.effectAllowed = 'move';
    }
  }
  
  onDragEnd(): void {
    this.isDragging = false;
  }
  
  onClick(): void {
    this.taskClicked.emit(this.task);
  }
  
  onKeyDown(event: KeyboardEvent): void {
    if (event.key === 'Enter' || event.key === ' ') {
      event.preventDefault();
      this.onClick();
    }
  }
  
  getPriorityClass(): string {
    return `priority-${this.priority}`;
  }
  
  getFormattedDate(date: Date): string {
    return new Intl.DateTimeFormat('en-US', {
      month: 'short',
      day: 'numeric'
    }).format(date);
  }
}
```

```typescript
    this.isDarkTheme = this.themeService.isDarkTheme();

    if (this.searchQuery.trim()) {
      this.searchSubmitted.emit(this.searchQuery.trim());
    }
  }
  
  onSearchKeyDown(event: KeyboardEvent): void {
    if (event.key === 'Enter') {
      this.onSearchSubmit();
    }
  }
  
  toggleUserMenu(): void {
    this.isUserMenuOpen = !this.isUserMenuOpen;
    this.isNotificationMenuOpen = false;
  }
  
  toggleNotificationMenu(): void {
    this.isNotificationMenuOpen = !this.isNotificationMenuOpen;
    this.isUserMenuOpen = false;
  }
  
  toggleTheme(): void {
    this.isDarkTheme = !this.isDarkTheme;
    this.themeToggled.emit();
  }
  
  toggleSidebar(): void {
    this.sidebarToggled.emit();
  }
  
  getUnreadNotificationCount(): number {
    return this.notifications.filter(n => !n.read).length;
  }
  
  closeMenus(): void {
    this.isUserMenuOpen = false;
    this.isNotificationMenuOpen = false;
  }
}
```

```typescript
@Component({
  selector: 'app-modal',
  templateUrl: './modal.component.html',
  styleUrls: ['./modal.component.scss']
})
export class ModalComponent implements OnInit, OnDestroy {
  @Input() isOpen: boolean = false;
  @Input() title: string = '';
  @Input() size: 'sm' | 'md' | 'lg' = 'md';
  @Input() closable: boolean = true;
  @Input() backdrop: boolean = true;
  
  @Output() modalClosed = new EventEmitter<void>();
  @Output() modalOpened = new EventEmitter<void>();
  
  private previousFocusElement: HTMLElement | null = null;
  
  constructor(private document: Document) {}
  
  ngOnInit(): void {
    if (this.isOpen) {
      this.onModalOpen();
    }
  }
  
  ngOnDestroy(): void {
    this.restoreFocus();
  }
  
  ngOnChanges(changes: SimpleChanges): void {
    if (changes['isOpen']) {
      if (this.isOpen) {
        this.onModalOpen();
      } else {
        this.onModalClose();
      }
    }
  }
  
  onModalOpen(): void {
    this.previousFocusElement = this.document.activeElement as HTMLElement;
    this.document.body.style.overflow = 'hidden';
    
    // Focus first focusable element in modal
    setTimeout(() => {
      const firstFocusable = this.getFirstFocusableElement();
      if (firstFocusable) {
        firstFocusable.focus();
      }
    }, 100);
    
    this.modalOpened.emit();
  }
  
  onModalClose(): void {
    this.document.body.style.overflow = '';
    this.restoreFocus();
    this.modalClosed.emit();
  }
  
  closeModal(): void {
    if (this.closable) {
      this.onModalClose();
    }
  }
  
  onBackdropClick(event: MouseEvent): void {
    if (this.backdrop && this.closable && event.target === event.currentTarget) {
      this.closeModal();
    }
  }
  
  onKeyDown(event: KeyboardEvent): void {
    if (event.key === 'Escape' && this.closable) {
      this.closeModal();
    } else if (event.key === 'Tab') {
      this.handleTabKey(event);
    }
  }
  
  private handleTabKey(event: KeyboardEvent): void {
    const focusableElements = this.getFocusableElements();
    const firstElement = focusableElements[0];
    const lastElement = focusableElements[focusableElements.length - 1];
    
    if (event.shiftKey) {
      if (this.document.activeElement === firstElement) {
        event.preventDefault();
        lastElement.focus();
      }
    } else {
      if (this.document.activeElement === lastElement) {
        event.preventDefault();
        firstElement.focus();
      }
    }
  }
  
  private getFocusableElements(): HTMLElement[] {
    const modal = this.document.querySelector('.modal-content');
    if (!modal) return [];
    
    const focusableSelectors = [
      'button:not([disabled])',
      'input:not([disabled])',
      'textarea:not([disabled])',
      'select:not([disabled])',
      'a[href]',
      '[tabindex]:not([tabindex="-1"])'
    ].join(', ');
    
    return Array.from(modal.querySelectorAll(focusableSelectors)) as HTMLElement[];
  }
  
  private getFirstFocusableElement(): HTMLElement | null {
    const elements = this.getFocusableElements();
    return elements.length > 0 ? elements[0] : null;
  }
  
  private restoreFocus(): void {
    if (this.previousFocusElement) {
      this.previousFocusElement.focus();
      this.previousFocusElement = null;
    }
  }
  
  getSizeClass(): string {
    return `modal-${this.size}`;
  }
}
```

```typescript
// kanban.service.ts
@Injectable({
  providedIn: 'root'
})
export class KanbanService {
  constructor(private http: HttpClient) {}
  
  getDashboardData(): Observable<any> {
    return this.http.get('/api/youth-accounts/1/dashboard')
      .pipe(
        map(data => this.transformToKanbanData(data)),
        catchError(this.handleError)
      );
  }
  
  createTask(task: Task): Observable<Task> {
    return this.http.post('/api/youth-accounts/1/fund-transfer', {
      amount: this.mapPriorityToAmount(task.priority),
      description: task.title,
      recipient: task.assignee?.id
    }).pipe(
      map(response => this.transformToTask(response)),
      catchError(this.handleError)
    );
  }
  
  updateTask(task: Task): Observable<Task> {
    return this.http.put('/api/youth-accounts/1/spending-limit', {
      limit: this.mapStatusToLimit(task.status),
      description: task.title
    }).pipe(
      map(response => this.transformToTask(response)),
      catchError(this.handleError)
    );
  }
  
  private transformToKanbanData(apiData: any): KanbanColumn[] {
    // Transform API response to kanban format
    return [
      {
        id: 'todo',
        title: 'To Do',
        cards: apiData.pendingTransactions?.map(this.transformToTask) || []
      },
      {
        id: 'inprogress',
        title: 'In Progress',
        cards: apiData.processingTransactions?.map(this.transformToTask) || []
      },
      {
        id: 'done',
        title: 'Done',
        cards: apiData.completedTransactions?.map(this.transformToTask) || []
      }
    ];
  }
  
  private transformToTask(apiItem: any): Task {
    return {
      id: apiItem.id,
      title: apiItem.description || 'Untitled Task',
      description: apiItem.notes,
      status: this.mapAmountToStatus(apiItem.amount),
      priority: this.mapAmountToPriority(apiItem.amount),
      dueDate: new Date(apiItem.scheduledDate),
      assignee: {
        id: apiItem.recipientId,
        name: apiItem.recipientName,
        avatar: '/assets/default-avatar.jpg'
      },
      tags: this.extractTags(apiItem.category)
    };
  }
  
  private handleError(error: any): Observable<never> {
    console.error('API Error:', error);
    return throwError(() => new Error('Failed to process request'));
  }
}
```

```typescript
// Error handling in components
export class KanbanBoardComponent {
  error: string | null = null;
  loading: boolean = false;
  
  loadKanbanData(): void {
    this.loading = true;
    this.error = null;
    
    this.kanbanService.getDashboardData()
      .subscribe({
        next: (data) => {
          this.columnData = data;
          this.loading = false;
        },
        error: (error) => {
          this.error = this.getErrorMessage(error);
          this.loading = false;
        }
      });
  }
  
  private getErrorMessage(error: any): string {
    if (error.status === 404) {
      return 'Dashboard data not found';
    } else if (error.status === 403) {
      return 'Access denied';
    } else if (error.status === 500) {
      return 'Server error occurred';
    } else {
      return 'Failed to load kanban data';
    }
  }
  
  retryLoad(): void {
    this.loadKanbanData();
  }
}
```
