// src/app/features/kanban/components/kanban-board/kanban-board.component.ts
import { Component, OnInit, OnDestroy, ChangeDetectionStrategy, ChangeDetectorRef } from '@angular/core';
import { Subject, takeUntil, catchError, of } from 'rxjs';
import { KanbanService } from '../../services/kanban.service';
import { Task, CreateTaskRequest } from '../../models/task.model';
import { KanbanColumn } from '../../models/kanban-column.model';
import { CdkDragDrop, moveItemInArray, transferArrayItem } from '@angular/cdk/drag-drop';

@Component({
  selector: 'app-kanban-board',
  templateUrl: './kanban-board.component.html',
  styleUrls: ['./kanban-board.component.scss'],
  changeDetection: ChangeDetectionStrategy.OnPush
})
export class KanbanBoardComponent implements OnInit, OnDestroy {
  // Input Properties
  columns: KanbanColumn[] = [
    { id: 'todo', title: 'To Do', tasks: [] },
    { id: 'inprogress', title: 'In Progress', tasks: [] },
    { id: 'done', title: 'Done', tasks: [] }
  ];
  
  // State Properties
  loading = false;
  error: string | null = null;
  selectedTasks: string[] = [];
  draggedTask: Task | null = null;
  showCreateModal = false;
  
  // Private Properties
  private destroy$ = new Subject<void>();
  
  constructor(
    private kanbanService: KanbanService,
    private cdr: ChangeDetectorRef
  ) {}
  
  ngOnInit(): void {
    this.loadTasks();
  }
  
  ngOnDestroy(): void {
    this.destroy$.next();
    this.destroy$.complete();
  }
  
  // Public Methods
  loadTasks(): void {
    this.loading = true;
    this.error = null;
    
    // Note: GET /api/v1/tasks endpoint to be implemented
    // For now, using mock data or local state
    this.kanbanService.getTasks()
      .pipe(
        takeUntil(this.destroy$),
        catchError(error => {
          this.error = 'Failed to load tasks. Please try again.';
          this.loading = false;
          this.cdr.markForCheck();
          return of([]);
        })
      )
      .subscribe(tasks => {
        this.distributeTasksToColumns(tasks);
        this.loading = false;
        this.cdr.markForCheck();
      });
  }
  
  onTaskCreate(taskData: CreateTaskRequest): void {
    this.kanbanService.createTask(taskData)
      .pipe(
        takeUntil(this.destroy$),
        catchError(error => {
          this.error = 'Failed to create task. Please try again.';
          this.cdr.markForCheck();
          return of(null);
        })
      )
      .subscribe(newTask => {
        if (newTask) {
          const todoColumn = this.columns.find(col => col.id === 'todo');
          if (todoColumn) {
            todoColumn.tasks.push(newTask);
          }
          this.showCreateModal = false;
          this.cdr.markForCheck();
        }
      });
  }
  
  onTaskMove(event: CdkDragDrop<Task[]>): void {
    if (event.previousContainer === event.container) {
      moveItemInArray(event.container.data, event.previousIndex, event.currentIndex);
    } else {
      transferArrayItem(
        event.previousContainer.data,
        event.container.data,
        event.previousIndex,
        event.currentIndex
      );
      
      // Update task status based on target column
      const task = event.container.data[event.currentIndex];
      const targetColumn = this.columns.find(col => col.tasks === event.container.data);
      if (task && targetColumn) {
        task.status = this.mapColumnToStatus(targetColumn.id);
        // Note: PUT /api/v1/tasks/{id} endpoint to be implemented
        this.updateTaskStatus(task);
      }
    }
    this.cdr.markForCheck();
  }
  
  onTaskSelect(taskId: string, selected: boolean): void {
    if (selected) {
      this.selectedTasks.push(taskId);
    } else {
      this.selectedTasks = this.selectedTasks.filter(id => id !== taskId);
    }
    this.cdr.markForCheck();
  }
  
  openCreateModal(): void {
    this.showCreateModal = true;
  }
  
  closeCreateModal(): void {
    this.showCreateModal = false;
  }
  
  // Private Methods
  private distributeTasksToColumns(tasks: Task[]): void {
    this.columns.forEach(column => {
      column.tasks = tasks.filter(task => 
        this.mapStatusToColumn(task.status) === column.id
      );
    });
  }
  
  private mapColumnToStatus(columnId: string): string {
    const statusMap: { [key: string]: string } = {
      'todo': 'PENDING',
      'inprogress': 'IN_PROGRESS',
      'done': 'COMPLETED'
    };
    return statusMap[columnId] || 'PENDING';
  }
  
  private mapStatusToColumn(status: string): string {
    const columnMap: { [key: string]: string } = {
      'PENDING': 'todo',
      'IN_PROGRESS': 'inprogress',
      'COMPLETED': 'done',
      'CANCELLED': 'todo'
    };
    return columnMap[status] || 'todo';
  }
  
  private updateTaskStatus(task: Task): void {
    // Note: This will use PUT /api/v1/tasks/{id} when available
    this.kanbanService.updateTask(task.id, { status: task.status })
      .pipe(
        takeUntil(this.destroy$),
        catchError(error => {
          this.error = 'Failed to update task status.';
          this.cdr.markForCheck();
          return of(null);
        })
      )
      .subscribe();
  }
}

// src/app/layout/sidebar/sidebar.component.ts
import { Component, Input, Output, EventEmitter, ChangeDetectionStrategy } from '@angular/core';
import { Router } from '@angular/router';
import { User } from '../../shared/models/user.model';

interface MenuItem {
  id: string;
  label: string;
  icon: string;
  route: string;
  badge?: number;
  disabled?: boolean;
}

@Component({
  selector: 'app-sidebar',
  templateUrl: './sidebar.component.html',
  styleUrls: ['./sidebar.component.scss'],
  changeDetection: ChangeDetectionStrategy.OnPush
})
export class SidebarComponent {
  @Input() currentPage = 'board';
  @Input() user: User | null = null;
  @Input() collapsed = false;
  
  @Output() navigate = new EventEmitter<string>();
  @Output() toggleCollapse = new EventEmitter<void>();
  
  menuItems: MenuItem[] = [
    { id: 'board', label: 'Board', icon: 'dashboard', route: '/board' },
    { id: 'settings', label: 'Settings', icon: 'settings', route: '/settings' },
    { id: 'analytics', label: 'Analytics', icon: 'analytics', route: '/analytics' },
    { id: 'reports', label: 'Reports', icon: 'description', route: '/reports' }
  ];
  
  constructor(private router: Router) {}
  
  onMenuItemClick(item: MenuItem): void {
    if (!item.disabled) {
      this.navigate.emit(item.id);
      this.router.navigate([item.route]);
    }
  }
  
  onToggleCollapse(): void {
    this.toggleCollapse.emit();
  }
  
  trackByMenuItem(index: number, item: MenuItem): string {
    return item.id;
  }
}

// src/app/layout/header/header.component.ts
import { Component, Input, Output, EventEmitter, ChangeDetectionStrategy } from '@angular/core';
import { User } from '../../shared/models/user.model';

interface Notification {
  id: string;
  title: string;
  message: string;
  type: 'info' | 'warning' | 'error' | 'success';
  timestamp: Date;
  read: boolean;
}

@Component({
  selector: 'app-header',
  templateUrl: './header.component.html',
  styleUrls: ['./header.component.scss'],
  changeDetection: ChangeDetectionStrategy.OnPush
})
export class HeaderComponent {
  @Input() user: User | null = null;
  @Input() notifications: Notification[] = [];
  
  @Output() search = new EventEmitter<string>();
  @Output() userAction = new EventEmitter<string>();
  
  searchQuery = '';
  showNotifications = false;
  showUserMenu = false;
  
  get unreadNotifications(): number {
    return this.notifications.filter(n => !n.read).length;
  }
  
  onSearchInput(event: Event): void {
    const target = event.target as HTMLInputElement;
    this.searchQuery = target.value;
  }
  
  onSearchSubmit(): void {
    if (this.searchQuery.trim()) {
      this.search.emit(this.searchQuery.trim());
    }
  }
  
  onSearchClear(): void {
    this.searchQuery = '';
    this.search.emit('');
  }
  
  toggleNotifications(): void {
    this.showNotifications = !this.showNotifications;
    this.showUserMenu = false;
  }
  
  toggleUserMenu(): void {
    this.showUserMenu = !this.showUserMenu;
    this.showNotifications = false;
  }
  
  onUserActionClick(action: string): void {
    this.userAction.emit(action);
    this.showUserMenu = false;
  }
  
  onNotificationClick(notification: Notification): void {
    // Mark as read and handle notification action
    notification.read = true;
    this.userAction.emit(`notification:${notification.id}`);
  }
  
  trackByNotificationId(index: number, notification: Notification): string {
    return notification.id;
  }
}

// src/app/features/kanban/components/create-task-modal/create-task-modal.component.ts
import { Component, Input, Output, EventEmitter, OnInit, ChangeDetectionStrategy, ChangeDetectorRef } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { CreateTaskRequest } from '../../models/task.model';
import { User } from '../../../shared/models/user.model';

@Component({
  selector: 'app-create-task-modal',
  templateUrl: './create-task-modal.component.html',
  styleUrls: ['./create-task-modal.component.scss'],
  changeDetection: ChangeDetectionStrategy.OnPush
})
export class CreateTaskModalComponent implements OnInit {
  @Input() visible = false;
  @Input() users: User[] = [];
  
  @Output() taskCreate = new EventEmitter<CreateTaskRequest>();
  @Output() modalClose = new EventEmitter<void>();
  
  taskForm!: FormGroup;
  submitting = false;
  
  priorityOptions = [
    { value: 'LOW', label: 'Low', color: '#10b981' },
    { value: 'MEDIUM', label: 'Medium', color: '#f59e0b' },
    { value: 'HIGH', label: 'High', color: '#ef4444' },
    { value: 'CRITICAL', label: 'Critical', color: '#7c2d12' }
  ];
  
  constructor(
    private fb: FormBuilder,
    private cdr: ChangeDetectorRef
  ) {}
  
  ngOnInit(): void {
    this.initializeForm();
  }
  
  private initializeForm(): void {
    this.taskForm = this.fb.group({
      title: ['', [Validators.required, Validators.minLength(3), Validators.maxLength(100)]],
      description: ['', [Validators.maxLength(500)]],
      priority: ['MEDIUM', [Validators.required]],
      assigneeId: [''],
      dueDate: [''],
      tags: [[]]
    });
  }
  
  onSubmit(): void {
    if (this.taskForm.valid && !this.submitting) {
      this.submitting = true;
      
      const formValue = this.taskForm.value;
      const taskData: CreateTaskRequest = {
        title: formValue.title.trim(),
        description: formValue.description?.trim() || '',
        priority: formValue.priority,
        assigneeId: formValue.assigneeId || null,
        dueDate: formValue.dueDate ? new Date(formValue.dueDate) : null,
        tags: formValue.tags || []
      };
      
      this.taskCreate.emit(taskData);
      
      // Reset form after submission
      setTimeout(() => {
        this.submitting = false;
        this.resetForm();
        this.cdr.markForCheck();
      }, 1000);
    } else {
      this.markFormGroupTouched();
    }
  }
  
  onCancel(): void {
    this.resetForm();
    this.modalClose.emit();
  }
  
  onBackdropClick(event: Event): void {
    if (event.target === event.currentTarget) {
      this.onCancel();
    }
  }
  
  onEscapeKey(event: KeyboardEvent): void {
    if (event.key === 'Escape') {
      this.onCancel();
    }
  }
  
  private resetForm(): void {
    this.taskForm.reset({
      title: '',
      description: '',
      priority: 'MEDIUM',
      assigneeId: '',
      dueDate: '',
      tags: []
    });
    this.submitting = false;
  }
  
  private markFormGroupTouched(): void {
    Object.keys(this.taskForm.controls).forEach(key => {
      const control = this.taskForm.get(key);
      if (control) {
        control.markAsTouched();
      }
    });
    this.cdr.markForCheck();
  }
  
  getFieldError(fieldName: string): string | null {
    const field = this.taskForm.get(fieldName);
    if (field && field.errors && field.touched) {
      if (field.errors['required']) {
        return `${this.getFieldLabel(fieldName)} is required`;
      }
      if (field.errors['minlength']) {
        return `${this.getFieldLabel(fieldName)} must be at least ${field.errors['minlength'].requiredLength} characters`;
      }
      if (field.errors['maxlength']) {
        return `${this.getFieldLabel(fieldName)} must not exceed ${field.errors['maxlength'].requiredLength} characters`;
      }
    }
    return null;
  }
  
  private getFieldLabel(fieldName: string): string {
    const labels: { [key: string]: string } = {
      title: 'Title',
      description: 'Description',
      priority: 'Priority',
      assigneeId: 'Assignee',
      dueDate: 'Due Date'
    };
    return labels[fieldName] || fieldName;
  }
  
  trackByUserId(index: number, user: User): string {
    return user.id;
  }
  
  trackByPriorityValue(index: number, priority: any): string {
    return priority.value;
  }
}
