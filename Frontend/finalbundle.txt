### TaskCreationComponent Implementation

**TypeScript (task-creation.component.ts):**
```typescript
import { Component, EventEmitter, Input, Output, OnInit, OnDestroy } from '@angular/core';
import { FormBuilder, FormGroup, Validators, AbstractControl } from '@angular/forms';
import { Subject, takeUntil, debounceTime } from 'rxjs';
import { TaskService } from '../../services/task.service';
import { Task, CreateTaskRequest, Priority } from '../../models/task.model';
import { CustomValidators } from '../../validators/custom-validators';

@Component({
  selector: 'app-task-creation',
  templateUrl: './task-creation.component.html',
  styleUrls: ['./task-creation.component.scss']
})
export class TaskCreationComponent implements OnInit, OnDestroy {
  @Input() isVisible: boolean = false;
  @Input() initialData: Partial<Task> | null = null;
  @Input() loading: boolean = false;
  
  @Output() taskCreated = new EventEmitter<Task>();
  @Output() modalClosed = new EventEmitter<void>();
  @Output() formValidationChanged = new EventEmitter<boolean>();

  taskForm: FormGroup;
  isSubmitting: boolean = false;
  validationErrors: string[] = [];
  titleCharCount: number = 0;
  descriptionCharCount: number = 0;
  
  readonly maxTitleLength = 255;
  readonly maxDescriptionLength = 2000;
  readonly priorityOptions = [
    { value: Priority.LOW, label: 'Low' },
    { value: Priority.MEDIUM, label: 'Medium' },
    { value: Priority.HIGH, label: 'High' },
    { value: Priority.CRITICAL, label: 'Critical' }
  ];

  private destroy$ = new Subject<void>();

  constructor(
    private fb: FormBuilder,
    private taskService: TaskService
  ) {
    this.initializeForm();
  }

  ngOnInit(): void {
    this.setupFormValidation();
    this.setupCharacterCounters();
    
    if (this.initialData) {
      this.populateForm(this.initialData);
    }
  }

  ngOnDestroy(): void {
    this.destroy$.next();
    this.destroy$.complete();
  }

  private initializeForm(): void {
    this.taskForm = this.fb.group({
      title: ['', [Validators.required, Validators.maxLength(this.maxTitleLength)]],
      description: ['', [Validators.maxLength(this.maxDescriptionLength)]],
      priority: [Priority.MEDIUM, [Validators.required]],
      dueDate: ['', [Validators.required, CustomValidators.futureDate]],
      assignedTo: [''],
      tags: [[]],
      estimatedHours: [null, [Validators.min(0.1), Validators.max(999)]]
    });
  }

  private setupFormValidation(): void {
    this.taskForm.statusChanges
      .pipe(takeUntil(this.destroy$))
      .subscribe(status => {
        this.formValidationChanged.emit(status === 'VALID');
      });
  }

  private setupCharacterCounters(): void {
    this.taskForm.get('title')?.valueChanges
      .pipe(takeUntil(this.destroy$))
      .subscribe(value => {
        this.titleCharCount = value ? value.length : 0;
      });

    this.taskForm.get('description')?.valueChanges
      .pipe(takeUntil(this.destroy$))
      .subscribe(value => {
        this.descriptionCharCount = value ? value.length : 0;
      });
  }

  private populateForm(data: Partial<Task>): void {
    this.taskForm.patchValue({
      title: data.title || '',
      description: data.description || '',
      priority: data.priority || Priority.MEDIUM,
      dueDate: data.dueDate ? new Date(data.dueDate).toISOString().split('T')[0] : '',
      assignedTo: data.assignedTo?.id || '',
      tags: data.tags || [],
      estimatedHours: data.estimatedHours || null
    });
  }

  onSubmit(): void {
    if (this.taskForm.valid && !this.isSubmitting) {
      this.isSubmitting = true;
      this.validationErrors = [];
      
      const formValue = this.taskForm.value;
      const createTaskRequest: CreateTaskRequest = {
        title: formValue.title.trim(),
        description: formValue.description?.trim() || '',
        priority: formValue.priority,
        dueDate: new Date(formValue.dueDate).toISOString(),
        assignedTo: formValue.assignedTo || undefined,
        tags: formValue.tags || [],
        estimatedHours: formValue.estimatedHours || undefined
      };

      this.taskService.createTask(createTaskRequest)
        .pipe(takeUntil(this.destroy$))
        .subscribe({
          next: (task: Task) => {
            this.taskCreated.emit(task);
            this.resetForm();
            this.closeModal();
            this.isSubmitting = false;
          },
          error: (error) => {
            this.handleSubmissionError(error);
            this.isSubmitting = false;
          }
        });
    } else {
      this.markFormGroupTouched();
    }
  }

  private handleSubmissionError(error: any): void {
    if (error.status === 400 && error.error?.validationErrors) {
      this.validationErrors = error.error.validationErrors.map((err: any) => err.message);
    } else if (error.status === 401) {
      this.validationErrors = ['Authentication required. Please log in again.'];
    } else if (error.status === 403) {
      this.validationErrors = ['You do not have permission to create tasks.'];
    } else if (error.status === 409) {
      this.validationErrors = ['A task with this title already exists.'];
    } else {
      this.validationErrors = ['An unexpected error occurred. Please try again.'];
    }
  }

  private markFormGroupTouched(): void {
    Object.keys(this.taskForm.controls).forEach(key => {
      const control = this.taskForm.get(key);
      control?.markAsTouched();
    });
  }

  private resetForm(): void {
    this.taskForm.reset({
      priority: Priority.MEDIUM,
      tags: []
    });
    this.titleCharCount = 0;
    this.descriptionCharCount = 0;
    this.validationErrors = [];
  }

  closeModal(): void {
    this.modalClosed.emit();
  }

  getFieldError(fieldName: string): string | null {
    const field = this.taskForm.get(fieldName);
    if (field?.errors && field.touched) {
      if (field.errors['required']) return `${fieldName} is required`;
      if (field.errors['maxlength']) return `${fieldName} exceeds maximum length`;
      if (field.errors['futureDate']) return 'Due date must be in the future';
      if (field.errors['min']) return 'Value must be greater than 0';
      if (field.errors['max']) return 'Value exceeds maximum allowed';
    }
    return null;
  }

  isFieldInvalid(fieldName: string): boolean {
    const field = this.taskForm.get(fieldName);
    return !!(field?.errors && field.touched);
  }
}
```
