### Interaction Flows

**User Action → UI → API:**

1. **Load Kanban Board:**
   - User navigates to Kanban page
   → KanbanBoardComponent loads
   → Call GET /api/youth-accounts/{id}/dashboard
   → Display columns with tasks

2. **Create New Task:**
   - User clicks "Add Task" button
   → Open TaskModalComponent
   → User fills form and submits
   → Call POST /api/youth-accounts/{id}/fund-transfer
   → Update UI with new task

3. **Move Task Between Columns:**
   - User drags task to different column
   → TaskCardComponent emits drag events
   → KanbanBoardComponent handles drop
   → Call PUT /api/youth-accounts/{id}/spending-limit
   → Update task status in UI

4. **Edit Task:**
   - User clicks on task card
   → Open TaskModalComponent in edit mode
   → User modifies and saves
   → Call PUT /api/youth-accounts/{id}/spending-limit
   → Update task in UI

5. **Search Tasks:**
   - User enters search query in header
   → HeaderComponent emits search event
   → Call GET /api/youth-accounts/{id}/transactions with filters
   → Display filtered results

```typescript
import { Component, OnInit, Input, Output, EventEmitter } from '@angular/core';
import { CdkDragDrop, moveItemInArray, transferArrayItem } from '@angular/cdk/drag-drop';

export interface KanbanColumn {
  id: string;
  title: string;
  cards: Task[];
  maxCards?: number;
}

export interface Task {
  id: string;
  title: string;
  description?: string;
  priority: 'low' | 'medium' | 'high';
  assignee?: User;
  tags: string[];
  dueDate?: Date;
  createdAt: Date;
  updatedAt: Date;
}

export interface User {
  id: string;
  name: string;
  avatar?: string;
  email: string;
}

@Component({
  selector: 'app-kanban-board',
  templateUrl: './kanban-board.component.html',
  styleUrls: ['./kanban-board.component.scss']
})
export class KanbanBoardComponent implements OnInit {
  @Input() columns: KanbanColumn[] = [];
  @Input() loading: boolean = false;
  @Input() error: string | null = null;
  
  @Output() taskMoved = new EventEmitter<{task: Task, fromColumn: string, toColumn: string}>();
  @Output() taskCreated = new EventEmitter<Task>();
  @Output() taskUpdated = new EventEmitter<Task>();
  @Output() taskDeleted = new EventEmitter<string>();
  
  draggedTask: Task | null = null;
  columnData: KanbanColumn[] = [];
  isMobile: boolean = false;
  activeColumn: string = 'todo';
  
  ngOnInit(): void {
    this.initializeColumns();
    this.checkViewport();
    this.setupResizeListener();
  }
  
  private initializeColumns(): void {
    if (this.columns.length === 0) {
      this.columnData = [
        { id: 'todo', title: 'To Do', cards: [] },
        { id: 'inprogress', title: 'In Progress', cards: [] },
        { id: 'done', title: 'Done', cards: [] }
      ];
    } else {
      this.columnData = [...this.columns];
    }
  }
  
  private checkViewport(): void {
    this.isMobile = window.innerWidth < 768;
  }
  
  private setupResizeListener(): void {
    window.addEventListener('resize', () => {
      this.checkViewport();
    });
  }
  
  onDrop(event: CdkDragDrop<Task[]>, targetColumnId: string): void {
    const task = event.item.data;
    const sourceColumnId = event.previousContainer.id;
    
    if (event.previousContainer === event.container) {
      moveItemInArray(event.container.data, event.previousIndex, event.currentIndex);
    } else {
      transferArrayItem(
        event.previousContainer.data,
        event.container.data,
        event.previousIndex,
        event.currentIndex
      );
      
      this.taskMoved.emit({
        task: task,
        fromColumn: sourceColumnId,
        toColumn: targetColumnId
      });
    }
  }
  
  onTaskClick(task: Task): void {
    // Open task modal for editing
  }
  
  onAddTask(columnId: string): void {
    // Open task creation modal
  }
  
  onColumnChange(columnId: string): void {
    this.activeColumn = columnId;
  }
  
  getColumnTasks(columnId: string): Task[] {
    const column = this.columnData.find(col => col.id === columnId);
    return column ? column.cards : [];
  }
}
```

```typescript
// kanban.service.ts
import { Injectable } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { Observable, map } from 'rxjs';

@Injectable({
  providedIn: 'root'
})
export class KanbanService {
  private baseUrl = '/api/youth-accounts';
  
  constructor(private http: HttpClient) {}
  
  loadKanbanData(accountId: string): Observable<KanbanColumn[]> {
    return this.http.get<any>(`${this.baseUrl}/${accountId}/dashboard`)
      .pipe(
        map(response => this.transformToKanbanData(response))
      );
  }
  
  createTask(accountId: string, task: Partial<Task>): Observable<Task> {
    const payload = this.transformTaskToFundTransfer(task);
    return this.http.post<any>(`${this.baseUrl}/${accountId}/fund-transfer`, payload)
      .pipe(
```

```typescript
import { Component, Input, Output, EventEmitter, OnInit, OnDestroy } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';

export interface TaskFormData {
  title: string;
  description: string;
  priority: 'low' | 'medium' | 'high';
  assigneeId: string;
  tags: string[];
  dueDate: Date | null;
}

export interface ValidationError {
  field: string;
  message: string;
}

@Component({
  selector: 'app-task-modal',
  templateUrl: './task-modal.component.html',
  styleUrls: ['./task-modal.component.scss']
})
export class TaskModalComponent implements OnInit, OnDestroy {
  @Input() isOpen: boolean = false;
  @Input() task: Task | null = null;
  @Input() mode: 'create' | 'edit' = 'create';
  
  @Output() taskSaved = new EventEmitter<Task>();
  @Output() modalClosed = new EventEmitter<void>();
  
  taskForm!: FormGroup;
  isSubmitting: boolean = false;
  validationErrors: ValidationError[] = [];
  availableUsers: User[] = [];
  tagInput: string = '';
  
  constructor(private fb: FormBuilder) {}
  
  ngOnInit(): void {
    this.initializeForm();
    this.loadAvailableUsers();
    
    if (this.task && this.mode === 'edit') {
      this.populateForm();
    }
  }
  
  ngOnDestroy(): void {
    // Cleanup subscriptions
  }
  
  private initializeForm(): void {
    this.taskForm = this.fb.group({
      title: ['', [Validators.required, Validators.minLength(3)]],
      description: [''],
      priority: ['medium', Validators.required],
      assigneeId: [''],
      dueDate: [null],
      tags: [[]]
    });
  }
  
  private populateForm(): void {
    if (this.task) {
      this.taskForm.patchValue({
        title: this.task.title,
        description: this.task.description || '',
        priority: this.task.priority,
        assigneeId: this.task.assignee?.id || '',
        dueDate: this.task.dueDate,
        tags: this.task.tags
      });
    }
  }
  
  private loadAvailableUsers(): void {
    // Load users from service
    this.availableUsers = [
      { id: '1', name: 'John Doe', email: 'john@example.com' },
      { id: '2', name: 'Jane Smith', email: 'jane@example.com' }
    ];
  }
  
  onSubmit(): void {
    if (this.taskForm.valid && !this.isSubmitting) {
      this.isSubmitting = true;
      this.validationErrors = [];
      
      const formData = this.taskForm.value;
      const taskData: Task = {
        id: this.task?.id || Date.now().toString(),
        title: formData.title,
        description: formData.description,
        priority: formData.priority,
        assignee: this.availableUsers.find(u => u.id === formData.assigneeId),
        tags: formData.tags,
        dueDate: formData.dueDate,
        createdAt: this.task?.createdAt || new Date(),
        updatedAt: new Date()
      };
      
      // Simulate API call
      setTimeout(() => {
        this.taskSaved.emit(taskData);
        this.isSubmitting = false;
        this.closeModal();
      }, 1000);
    } else {
      this.validateForm();
    }
  }
  
  private validateForm(): void {
    this.validationErrors = [];
    
    Object.keys(this.taskForm.controls).forEach(key => {
      const control = this.taskForm.get(key);
      if (control && control.invalid && control.touched) {
        if (control.errors?.['required']) {
          this.validationErrors.push({
            field: key,
            message: `${key} is required`
          });
        }
        if (control.errors?.['minlength']) {
          this.validationErrors.push({
            field: key,
            message: `${key} must be at least ${control.errors['minlength'].requiredLength} characters`
          });
        }
      }
    });
  }
  
  addTag(): void {
    if (this.tagInput.trim() && !this.taskForm.value.tags.includes(this.tagInput.trim())) {
      const currentTags = this.taskForm.value.tags;
      this.taskForm.patchValue({
        tags: [...currentTags, this.tagInput.trim()]
      });
      this.tagInput = '';
    }
  }
  
  removeTag(tagToRemove: string): void {
    const currentTags = this.taskForm.value.tags;
    this.taskForm.patchValue({
      tags: currentTags.filter((tag: string) => tag !== tagToRemove)
    });
  }
  
  closeModal(): void {
    this.modalClosed.emit();
    this.taskForm.reset();
    this.validationErrors = [];
    this.tagInput = '';
  }
  
  onBackdropClick(event: Event): void {
    if (event.target === event.currentTarget) {
      this.closeModal();
    }
  }
  
  getModalTitle(): string {
    return this.mode === 'create' ? 'Create New Task' : 'Edit Task';
  }
  
  getSubmitButtonText(): string {
    if (this.isSubmitting) {
      return this.mode === 'create' ? 'Creating...' : 'Updating...';
    }
    return this.mode === 'create' ? 'Create Task' : 'Update Task';
  }
}
```

```typescript
import { Component, Input, Output, EventEmitter, OnInit } from '@angular/core';

export interface Breadcrumb {
  label: string;
  url?: string;
  active?: boolean;
}

@Component({
  selector: 'app-header',
  templateUrl: './header.component.html',
  styleUrls: ['./header.component.scss']
})
export class HeaderComponent implements OnInit {
  @Input() user!: User;
  @Input() breadcrumbs: Breadcrumb[] = [];
  
  @Output() searchSubmitted = new EventEmitter<string>();
  @Output() themeToggled = new EventEmitter<void>();
  @Output() userMenuClicked = new EventEmitter<void>();
  @Output() sidebarToggled = new EventEmitter<void>();
  
  searchQuery: string = '';
  notificationCount: number = 3;
  isUserMenuOpen: boolean = false;
  isDarkTheme: boolean = false;
  
  ngOnInit(): void {
    this.loadThemePreference();
  }
  
  private loadThemePreference(): void {
    const savedTheme = localStorage.getItem('theme');
    this.isDarkTheme = savedTheme === 'dark';
  }
  
  onSearchSubmit(): void {
    if (this.searchQuery.trim()) {
      this.searchSubmitted.emit(this.searchQuery.trim());
    }
  }
  
  onSearchKeyup(event: KeyboardEvent): void {
    if (event.key === 'Enter') {
      this.onSearchSubmit();
    }
  }
  
  toggleTheme(): void {
    this.isDarkTheme = !this.isDarkTheme;
    localStorage.setItem('theme', this.isDarkTheme ? 'dark' : 'light');
    this.themeToggled.emit();
  }
  
  toggleUserMenu(): void {
    this.isUserMenuOpen = !this.isUserMenuOpen;
    if (this.isUserMenuOpen) {
      this.userMenuClicked.emit();
    }
  }
  
  toggleSidebar(): void {
    this.sidebarToggled.emit();
  }
  
  closeUserMenu(): void {
    this.isUserMenuOpen = false;
  }
  
  getUserInitials(): string {
    return this.user.name.split(' ').map(n => n[0]).join('').toUpperCase();
  }
}
```

### API INTEGRATION MAPPING

**Component to API Mapping:**

1. **KanbanBoardComponent:**
   - **Load Data:** GET /api/youth-accounts/{id}/dashboard
   - **Transform Response:** Map account data to Kanban columns structure
   - **Error Handling:** Display error state when API fails

2. **TaskModalComponent:**
   - **Create Task:** POST /api/youth-accounts/{id}/fund-transfer
   - **Update Task:** PUT /api/youth-accounts/{id}/spending-limit
   - **Transform Payload:** Map task data to API request format

3. **TaskCardComponent:**
   - **Update Status:** PUT /api/youth-accounts/{id}/spending-limit
   - **Move Between Columns:** API call triggered by parent component

4. **HeaderComponent:**
   - **Search:** GET /api/youth-accounts/{id}/transactions (with filters)
   - **Transform Query:** Map search terms to transaction filters

5. **KanbanService:**
   - **Centralized API Logic:** Handle all API transformations
   - **Error Handling:** Consistent error response handling
   - **Data Mapping:** Transform between UI models and API responses

### INTERACTION FLOWS

**Complete User Interaction Mapping:**

1. **Application Load:**
   - App Component initializes
   - HeaderComponent loads user data
   - SidebarComponent renders navigation
   - KanbanBoardComponent calls API to load tasks

2. **Task Creation Flow:**
   - User clicks "Add Task" in KanbanColumnComponent
   - TaskModalComponent opens in create mode
   - User fills form and submits
   - API call to create task
   - UI updates with new task
   - Modal closes

3. **Drag and Drop Flow:**
   - User drags TaskCardComponent
   - CDK Drag Drop handles drag events
   - Drop event triggers in KanbanBoardComponent
   - API call to update task status
   - UI reflects new task position

4. **Search Flow:**
   - User enters query in HeaderComponent
   - Search event emitted to parent
   - API call with search filters
   - Results displayed in task list

5. **Mobile Responsive Flow:**
   - Viewport detection in KanbanBoardComponent
   - Mobile tabs rendered for column navigation
   - Single column view activated
   - Touch interactions enabled
