    Start[User Opens Application]

    InitBoard --> DisplayCols[Display Three Columns]
    DisplayCols --> TodoCol[To Do Column]
    DisplayCols --> ProgressCol[In Progress Column]

    TodoCol --> ViewTasks[View Existing Tasks]

    UserActions --> CreateTask[Create New Task]

    UserActions --> SearchTasks[Search Tasks]
    UserActions --> ViewProfile[Access User Profile]
    
    CreateTask --> OpenModal[Open Task Creation Modal]
    OpenModal --> FillForm[Fill Task Details]
    FillForm --> ValidateForm{Form Valid?}
    ValidateForm -->|No| ShowErrors[Show Validation Errors]
    ShowErrors --> FillForm
    ValidateForm -->|Yes| SubmitTask[Submit New Task]
    SubmitTask --> APICreate[POST API Call]
    APICreate --> CreateSuccess{API Success?}
    CreateSuccess -->|No| ShowAPIError[Show Error Message]
    ShowAPIError --> FillForm
    CreateSuccess -->|Yes| RefreshBoard[Refresh Kanban Board]
    RefreshBoard --> ViewTasks
    
    EditTask --> SelectTask[Click on Task Card]
    SelectTask --> OpenEditModal[Open Edit Modal]

    ShowUpdateErrors --> UpdateForm
    ValidateUpdate -->|Yes| SubmitUpdate[Submit Changes]
    SubmitUpdate --> APIUpdate[PUT API Call]
    APIUpdate --> UpdateSuccess{API Success?}
    UpdateSuccess -->|No| ShowUpdateAPIError[Show Error Message]
    ShowUpdateAPIError --> UpdateForm
    UpdateSuccess -->|Yes| RefreshBoard
    
    MoveTask --> DragStart[Start Dragging Task]
    DragStart --> DragOver[Drag Over Target Column]
    DragOver --> DropTask[Drop Task in Column]
    DropTask --> UpdateStatus[Update Task Status]
    UpdateStatus --> APIMoveCall[PUT API Call]
    APIMoveCall --> MoveSuccess{API Success?}
    MoveSuccess -->|No| RevertMove[Revert Task Position]
    RevertMove --> ViewTasks
    MoveSuccess -->|Yes| RefreshBoard
    
    SearchTasks --> EnterQuery[Enter Search Query]
    EnterQuery --> FilterResults[Filter Task Results]

    ViewProfile --> OpenUserMenu[Open User Menu]

    SettingsPage --> UserActions
