            <p>This diagram illustrates the complete user journey for creating tasks in the Kanban board application. The flow covers navigation, form interaction, validation, submission, and error handling scenarios.</p>
        </div>

        <div class="mermaid">
            flowchart TD
                Start[User Opens Kanban Board]
                Start --> LoadApp[Application Loads]
                LoadApp --> ViewDashboard[View Dashboard with Sidebar and Kanban Columns]
                
                ViewDashboard --> ClickAddTask[Click Add Task Button]
                ClickAddTask --> OpenModal[Task Creation Modal Opens]
                OpenModal --> InitializeForm[Initialize Empty Task Form]
                
                InitializeForm --> FillTitle[Fill Task Title Field]
                FillTitle --> TitleValidation{Title Valid?}
                TitleValidation -->|No| ShowTitleError[Show Title Error Message]
                ShowTitleError --> FillTitle
                TitleValidation -->|Yes| UpdateTitleCounter[Update Character Counter]
                
                UpdateTitleCounter --> FillDescription[Fill Description Field]
                FillDescription --> UpdateDescCounter[Update Description Counter]
                UpdateDescCounter --> SelectPriority[Select Task Priority]
                
                SelectPriority --> PriorityValidation{Priority Selected?}
                PriorityValidation -->|No| ShowPriorityError[Show Priority Required Error]
                ShowPriorityError --> SelectPriority
                PriorityValidation -->|Yes| SelectDueDate[Select Due Date]
                
                SelectDueDate --> DateValidation{Date Valid and Future?}
                DateValidation -->|No| ShowDateError[Show Date Error Message]
                ShowDateError --> SelectDueDate
                DateValidation -->|Yes| FormComplete[All Fields Completed]
                
                FormComplete --> ClickSubmit[Click Submit Button]
                ClickSubmit --> FormValidation{Form Valid?}
                FormValidation -->|No| ShowFormErrors[Show Form Validation Errors]
                ShowFormErrors --> FillTitle
                
                FormValidation -->|Yes| ShowLoading[Show Loading Spinner]
                ShowLoading --> CallAPI[Call POST /tasks API]
                
                CallAPI --> APIResponse{API Response}
                APIResponse -->|Success| TaskCreated[Task Successfully Created]
                APIResponse -->|Error| ShowAPIError[Show API Error Message]
                
                ShowAPIError --> RetryOption{User Chooses Retry?}
                RetryOption -->|Yes| CallAPI
                RetryOption -->|No| StayInModal[Stay in Modal for Correction]
                StayInModal --> FillTitle
                
                TaskCreated --> CloseModal[Close Task Creation Modal]
                CloseModal --> RefreshBoard[Refresh Kanban Board]
                RefreshBoard --> ShowNewTask[Display New Task in Appropriate Column]
                ShowNewTask --> ResetForm[Reset Form for Next Use]
                ResetForm --> ViewDashboard
                
                ViewDashboard --> ClickCancel[Click Cancel Button]
                ClickCancel --> ConfirmCancel{Unsaved Changes?}
                ConfirmCancel -->|Yes| ShowCancelWarning[Show Unsaved Changes Warning]
                ShowCancelWarning --> UserConfirm{User Confirms Cancel?}
                UserConfirm -->|No| InitializeForm
                UserConfirm -->|Yes| DiscardChanges[Discard Changes]
                ConfirmCancel -->|No| DiscardChanges
                DiscardChanges --> CloseModal
                
                ViewDashboard --> NavigateAway[Navigate to Other Sections]
                NavigateAway --> SidebarNavigation[Use Sidebar Navigation]
                SidebarNavigation --> ViewDashboard
        </div>
