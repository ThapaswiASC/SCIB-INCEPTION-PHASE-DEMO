flowchart TD
    Start[User Opens Application]
    Start --> LoadApp[Load Kanban Board App]
    LoadApp --> InitAuth[Initialize Authentication]
    InitAuth --> AuthCheck{User Authenticated?}
    
    AuthCheck -->|No| Login[Show Login Screen]
    Login --> AuthSuccess[Authentication Success]
    AuthSuccess --> LoadDashboard[Load Dashboard]
    
    AuthCheck -->|Yes| LoadDashboard
    LoadDashboard --> InitComponents[Initialize Components]
    InitComponents --> LoadSidebar[Load Sidebar Navigation]
    InitComponents --> LoadHeader[Load Header with Search]
    InitComponents --> LoadKanban[Load Kanban Board]
    
    LoadKanban --> FetchTasks[Fetch Tasks from API]
    FetchTasks --> TasksLoaded{Tasks Loaded Successfully?}
    
    TasksLoaded -->|No| ShowError[Show Error State]
    ShowError --> RetryLoad[User Clicks Retry]
    RetryLoad --> FetchTasks
    
    TasksLoaded -->|Yes| RenderBoard[Render Kanban Board]
    RenderBoard --> ShowColumns[Display Three Columns]
    ShowColumns --> TodoColumn[To Do Column]
    ShowColumns --> ProgressColumn[In Progress Column]
    ShowColumns --> DoneColumn[Done Column]
    
    TodoColumn --> RenderTasks[Render Task Cards]
    ProgressColumn --> RenderTasks
    DoneColumn --> RenderTasks
    
    RenderTasks --> UserReady[User Interface Ready]
    
    UserReady --> UserActions{User Action}
    
    UserActions --> SearchAction[Search Tasks]
    SearchAction --> SearchAPI[Call Search API]
    SearchAPI --> FilterResults[Filter and Display Results]
    FilterResults --> UserActions
    
    UserActions --> NavigateAction[Navigate via Sidebar]
    NavigateAction --> RouteChange[Change Route]
    RouteChange --> LoadNewPage[Load New Page]
    LoadNewPage --> UserActions
    
    UserActions --> CreateTask[Click Add Task]
    CreateTask --> OpenModal[Open Task Modal]
    OpenModal --> FillForm[Fill Task Form]
    FillForm --> ValidateForm{Form Valid?}
    
    ValidateForm -->|No| ShowValidation[Show Validation Errors]
    ShowValidation --> FillForm
    
    ValidateForm -->|Yes| SubmitTask[Submit Task]
    SubmitTask --> CreateAPI[Call Create Task API]
    CreateAPI --> TaskCreated{Task Created?}
    
    TaskCreated -->|No| ShowCreateError[Show Creation Error]
    ShowCreateError --> FillForm
    
    TaskCreated -->|Yes| CloseModal[Close Modal]
    CloseModal --> RefreshBoard[Refresh Kanban Board]
    RefreshBoard --> UserActions
    
    UserActions --> DragTask[Drag Task Card]
    DragTask --> DragOver[Drag Over Column]
    DragOver --> DropTask[Drop Task in Column]
    DropTask --> ValidateDrop{Valid Drop?}
    
    ValidateDrop -->|No| RevertDrag[Revert to Original Position]
    RevertDrag --> UserActions
    
    ValidateDrop -->|Yes| UpdateAPI[Call Update Task API]
    UpdateAPI --> TaskUpdated{Task Updated?}
    
    TaskUpdated -->|No| ShowUpdateError[Show Update Error]
    ShowUpdateError --> RevertDrag
    
    TaskUpdated -->|Yes| UpdateUI[Update UI State]
    UpdateUI --> UserActions
    
    UserActions --> ClickTask[Click Task Card]
    ClickTask --> OpenTaskDetails[Open Task Details]
    OpenTaskDetails --> EditTask[Edit Task]
    EditTask --> OpenModal
    
    UserActions --> FilterTasks[Apply Filters]
    FilterTasks --> UpdateFilters[Update Filter State]
    UpdateFilters --> FilterBoard[Filter Board Display]
    FilterBoard --> UserActions
    
    UserActions --> NotificationClick[Click Notifications]
    NotificationClick --> LoadNotifications[Load Notifications]
    LoadNotifications --> ShowNotifications[Display Notification Panel]
    ShowNotifications --> UserActions
    
    UserActions --> UserMenu[Click User Menu]
    UserMenu --> ShowUserOptions[Show User Options]
    ShowUserOptions --> ProfileAction[Profile/Settings/Logout]
    ProfileAction --> UserActions
    
    UserActions --> MobileView[Switch to Mobile View]
    MobileView --> ShowTabs[Show Column Tabs]
    ShowTabs --> TabNavigation[Navigate Between Tabs]
    TabNavigation --> UserActions
    
    UserActions --> Logout[Logout]
    Logout --> ClearSession[Clear User Session]
    ClearSession --> Start

            </div>

            </div>
            <div class="flow-section">
